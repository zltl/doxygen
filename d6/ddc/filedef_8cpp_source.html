319void FileDefImpl::writeTagFile(TextStream &tagFile)
320{
321 tagFile << " <compound kind=\"file\">\n";
322 tagFile << " <name>" << convertToXML(name()) << "</name>\n";
323 tagFile << " <path>" << convertToXML(stripFromPath(getPath())) << "</path>\n";
324 tagFile << " <filename>" << addHtmlExtensionIfMissing(getOutputFileBase()) << "</filename>\n";
325 for (const auto &ii : m_includeList)
326 {
327 const FileDef *fd=ii.fileDef;
328 if (fd && fd->isLinkable() && !fd->isReference())
329 {
330 bool isIDLorJava = FALSE;
331 SrcLangExt lang = fd->getLanguage();
332 isIDLorJava = lang==SrcLangExt_IDL || lang==SrcLangExt_Java;
333 const char *locStr = (ii.local || isIDLorJava) ? "yes" : "no";
334 const char *impStr = (ii.imported || isIDLorJava) ? "yes" : "no";
335 tagFile << " <includes id=\""
336 << convertToXML(fd->getOutputFileBase()) << "\" "
337 << "name=\"" << convertToXML(fd->name()) << "\" "
338 << "local=\"" << locStr << "\" "
339 << "imported=\"" << impStr << "\">"
340 << convertToXML(ii.includeName)
341 << "</includes>\n";
342 }
343 }
344 for (const auto &lde : LayoutDocManager::instance().docEntries(LayoutDocManager::File))
345 {
346 switch (lde->kind())
347 {
348 case LayoutDocEntry::FileClasses:
349 {
350 writeClassesToTagFile(tagFile, m_classes);
351 }
352 break;
353 case LayoutDocEntry::FileInterfaces:
354 {
355 writeClassesToTagFile(tagFile, m_interfaces);
356 }
357 break;
358 case LayoutDocEntry::FileStructs:
359 {
360 writeClassesToTagFile(tagFile, m_structs);
361 }
362 break;
363 case LayoutDocEntry::FileExceptions:
364 {
365 writeClassesToTagFile(tagFile, m_exceptions);
366 }
367 break;
368 case LayoutDocEntry::FileConcepts:
369 {
370 for (const auto *nd : m_concepts)
371 {
372 if (nd->isLinkableInProject())
373 {
374 tagFile << " <concept>" << convertToXML(nd->name()) << "</concept>\n";
375 }
376 }
377 }
378 break;
379 case LayoutDocEntry::FileNamespaces:
380 {
381 for (const auto *nd : m_namespaces)
382 {
383 if (nd->isLinkableInProject())
384 {
385 tagFile << " <namespace>" << convertToXML(nd->name()) << "</namespace>\n";
386 }
387 }
388 }
389 break;
390 case LayoutDocEntry::MemberDecl:
391 {
392 const LayoutDocEntryMemberDecl *lmd = dynamic_cast<const LayoutDocEntryMemberDecl*>(lde.get());
393 if (lmd)
394 {
395 MemberList * ml = getMemberList(lmd->type);
396 if (ml)
397 {
398 ml->writeTagFile(tagFile);
399 }
400 }
401 }
402 break;
403 case LayoutDocEntry::MemberGroups:
404 {
405 for (const auto &mg : m_memberGroups)
406 {
407 mg->writeTagFile(tagFile);
408 }
409 }
410 break;
411 default:
412 break;
413 }
414 }
415
416 writeDocAnchorsToTagFile(tagFile);
417 tagFile << " </compound>\n";
418}

420void FileDefImpl::writeDetailedDescription(OutputList &ol,const QCString &title)
421{
422 if (hasDetailedDescription())
423 {
424 ol.pushGeneratorState();
425 ol.disable(OutputType::Html);
426 ol.writeRuler();
427 ol.popGeneratorState();
428 ol.pushGeneratorState();
429 ol.disableAllBut(OutputType::Html);
430 ol.writeAnchor(QCString(),"details");
431 ol.popGeneratorState();
432 ol.startGroupHeader();
433 ol.parseText(title);
434 ol.endGroupHeader();
435
436 ol.startTextBlock();
437 if (!briefDescription().isEmpty() && Config_getBool(REPEAT_BRIEF))
438 {
439 ol.generateDoc(briefFile(),briefLine(),this,0,briefDescription(),FALSE,FALSE,
440 QCString(),FALSE,FALSE,Config_getBool(MARKDOWN_SUPPORT));
441 }
442 if (!briefDescription().isEmpty() && Config_getBool(REPEAT_BRIEF) &&
443 !documentation().isEmpty())
444 {
445 ol.pushGeneratorState();
446 ol.disable(OutputType::Man);
447 ol.disable(OutputType::RTF);
448 // ol.newParagraph(); // FIXME:PARA
449 ol.enableAll();
450 ol.disableAllBut(OutputType::Man);
451 ol.enable(OutputType::Latex);
452 ol.writeString("\n\n");
453 ol.popGeneratorState();
454 }
455 if (!documentation().isEmpty())
456 {
457 ol.generateDoc(docFile(),docLine(),this,0,documentation()+"\n",TRUE,FALSE,
458 QCString(),FALSE,FALSE,Config_getBool(MARKDOWN_SUPPORT));
459 }
460 //printf("Writing source ref for file %s\n",qPrint(name()));
461 if (Config_getBool(SOURCE_BROWSER))
462 {
463 ol.startParagraph("definition");
464 QCString refText = theTranslator->trDefinedInSourceFile();
465 int fileMarkerPos = refText.find("@0");
466 if (fileMarkerPos!=-1) // should always pass this.
467 {
468 ol.parseText(refText.left(fileMarkerPos)); //text left from marker 1
469 ol.writeObjectLink(QCString(),getSourceFileBase(),QCString(),name());
470 ol.parseText(refText.right(
471 refText.length()-fileMarkerPos-2)); // text right from marker 2
472 }
473 else
474 {
475 err("translation error: invalid marker in trDefinedInSourceFile()\n");
476 }
477 ol.endParagraph();
478 }
479 ol.endTextBlock();
480 }

533void FileDefImpl::writeIncludeFiles(OutputList &ol)
534{
535 if (!m_includeList.empty())
536 {
537 ol.startTextBlock(TRUE);
538 for (const auto &ii : m_includeList)
539 {
540 const FileDef *fd=ii.fileDef;
541 bool isIDLorJava = FALSE;
542 if (fd)
543 {
544 SrcLangExt lang = fd->getLanguage();
545 isIDLorJava = lang==SrcLangExt_IDL || lang==SrcLangExt_Java;
546 }
547 ol.startTypewriter();
548 if (isIDLorJava) // IDL/Java include
549 {
550 ol.docify("import ");
551 }
552 else if (ii.imported) // Objective-C include
553 {
554 ol.docify("#import ");
555 }
556 else // C/C++ include
557 {
558 ol.docify("#include ");
559 }
560 if (ii.local || isIDLorJava)
561 ol.docify("\"");
562 else
563 ol.docify("<");
564 ol.disable(OutputType::Html);
565 ol.docify(ii.includeName);
566 ol.enableAll();
567 ol.disableAllBut(OutputType::Html);
568
569 // Here we use the include file name as it appears in the file.
570 // we could also we the name as it is used within doxygen,
571 // then we should have used fd->docName() instead of ii->includeName
572 if (fd && fd->isLinkable())
573 {
574 ol.writeObjectLink(fd->getReference(),
575 fd->generateSourceFile() ? fd->includeName() : fd->getOutputFileBase(),
576 QCString(),ii.includeName);
577 }
578 else
579 {
580 ol.docify(ii.includeName);
581 }
582
583 ol.enableAll();
584 if (ii.local || isIDLorJava)
585 ol.docify("\"");
586 else
587 ol.docify(">");
588 if (isIDLorJava)
589 ol.docify(";");
590 ol.endTypewriter();
591 ol.lineBreak();
592 }
593 ol.endTextBlock();
594 }
