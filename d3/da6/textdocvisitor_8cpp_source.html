53void TextDocVisitor::filter(const QCString &str)
54{
55 if (str.isEmpty()) return;
56 //printf("TextDocVisitor::filter(%s)\n",str);
57 const char *p=str.data();
58 char c;
59 while (*p)
60 {
61 c=*p++;
62 switch(c)
63 {
64 case '\n': m_t << " "; break;
65 default: m_t << c;
66 }
67 }
68}
