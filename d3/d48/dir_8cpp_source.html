223std::string Dir::filePath(const std::string &path,bool acceptsAbsPath) const
224{
225 std::string result;
226 if (acceptsAbsPath && !isRelativePath(path))
227 {
228 result = path;
229 }
230 else
231 {
232 result = (p->path / path).string();
233 }
234 correctPath(result);
235 return result;
236}
