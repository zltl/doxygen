683 /*! used as the title of the HTML page of a class/struct/union */
684 virtual QCString trCompoundReference(const QCString &clName,
685 ClassDef::CompoundType compType,
686 bool isTemplate)
687 {
688 QCString result=clName+" ";
689 if (isTemplate) result+="Template-";
690 switch(compType)
691 {
692 case ClassDef::Class: result+="Klassen"; break;
693 case ClassDef::Struct: result+="Struktur"; break;
694 case ClassDef::Union: result+="Varianten"; break;
695 case ClassDef::Interface: result+="Schnittstellen"; break;
696 case ClassDef::Protocol: result+="Protokoll"; break;
697 case ClassDef::Category: result+="Kategorie"; break;
698 case ClassDef::Exception: result+="Ausnahmen"; break;
699 default: break;
700 }
701 result+="referenz";
702 return result;
703 }

1931 * @param dayOfWeek Day of the week: 1=Monday..7=Sunday
1932 * @param hour Hour of the day: 0..23
1933 * @param minutes Minutes in the hour: 0..59
1934 * @param seconds Seconds within the minute: 0..59
1935 * @param includeTime Include time in the result string?
1936 */
1937 virtual QCString trDateTime(int year,int month,int day,int dayOfWeek,
1938 int hour,int minutes,int seconds,
1939 DateTimeType includeTime)
1940 {
1941 static const char *days[] = { "Mon","Die","Mit","Don","Fre","Sam","Son" };
1942 static const char *months[] = { "Jan","Feb","Mär","Apr","Mai","Jun","Jul","Aug","Sep","Okt","Nov","Dez" };
1943 QCString sdate;
1944 if (includeTime == DateTimeType::DateTime || includeTime == DateTimeType::Date)
1945 {
1946 sdate.sprintf("%s %s %d %d",days[dayOfWeek-1],months[month-1],day,year);
1947 }
1948 if (includeTime == DateTimeType::DateTime) sdate += " ";
1949 if (includeTime == DateTimeType::DateTime || includeTime == DateTimeType::Time)
1950 {
1951 QCString stime;
1952 stime.sprintf("%.2d:%.2d:%.2d",hour,minutes,seconds);
1953 sdate+=stime;
1954 }
1955 return sdate;
