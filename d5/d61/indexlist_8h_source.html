54/** \brief A list of index interfaces.
55 *
56 * This class itself implements all methods of IndexIntf and
57 * just forwards the calls to all items in the list (composite design pattern).
58 */
59class IndexList
60{
61 using IndexVariant = std::variant<DocSets, EclipseHelp, FTVHelp, HtmlHelp, Qhp>;
62
63 template<template <class> class IndexT, class... As>
64 void foreach(As&&... args)
65 {
66 for (auto &v : m_indices)
67 {
68 dispatch_call<IndexT>(v,std::forward<As>(args)...);
69 }
70 }
71 template<template <class> class IndexT, class... As>
72 void foreach_locked(As&&... args)
73 {
74 std::lock_guard<std::mutex> lock(m_mutex);
75 for (auto &v : m_indices)
76 {
77 dispatch_call<IndexT>(v,std::forward<As>(args)...);
78 }
79 }
80
81 public:
82 /** disable the indices */
83 void disable()
84 { m_enabled = FALSE; }
85
86 /** enable the indices */
87 void enable()
88 { m_enabled = TRUE; }
89
90 /** returns true iff the indices are enabled */
91 bool isEnabled() const
92 { return m_enabled; }
93
94 /** Add an index generator to the list, using a syntax similar to std::make_unique<T>() */
95 template<class T,class... As>
96 void addIndex(As&&... args)
97 { m_indices.push_back(std::move(T{std::forward<As>(args)...})); }
98
99 void initialize()
100 { foreach<IndexIntf::initialize>(); }
101
102 void finalize()
103 { foreach<IndexIntf::finalize>(); }
104
105 void incContentsDepth()
106 { if (m_enabled) foreach_locked<IndexIntf::incContentsDepth>(); }
107
108 void decContentsDepth()
109 { if (m_enabled) foreach_locked<IndexIntf::decContentsDepth>(); }
110
111 void addContentsItem(bool isDir, const QCString &name, const QCString &ref,
112 const QCString &file, const QCString &anchor,bool separateIndex=FALSE,bool addToNavIndex=FALSE,
113 const Definition *def=0)
114 { if (m_enabled) foreach_locked<IndexIntf::addContentsItem>(isDir,name,ref,file,anchor,separateIndex,addToNavIndex,def); }
115
116 void addIndexItem(const Definition *context,const MemberDef *md,const QCString &sectionAnchor=QCString(),const QCString &title=QCString())
117 { if (m_enabled) foreach_locked<IndexIntf::addIndexItem>(context,md,sectionAnchor,title); }
118
119 void addIndexFile(const QCString &name)
120 { if (m_enabled) foreach_locked<IndexIntf::addIndexFile>(name); }
121
122 void addImageFile(const QCString &name)
123 { if (m_enabled) foreach_locked<IndexIntf::addImageFile>(name); }
124
125 void addStyleSheetFile(const QCString &name)
126 { if (m_enabled) foreach_locked<IndexIntf::addStyleSheetFile>(name); }
127
128 private:
129 bool m_enabled = true;
130 std::mutex m_mutex;
131 std::vector<IndexVariant> m_indices;
132};
