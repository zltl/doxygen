113void CitationManager::insertCrossReferencesForBibFile(const QCString &bibFile)
114{
115 // sanity checks
116 if (bibFile.isEmpty())
117 {
118 return;
119 }
120 FileInfo fi(bibFile.str());
121 if (!fi.exists())
122 {
123 err("bib file %s not found!\n",qPrint(bibFile));
124 return;
125 }
126 std::ifstream f = Portable::openInputStream(bibFile);
127 if (!f.is_open())
128 {
129 err("could not open file %s for reading\n",qPrint(bibFile));
130 return;
131 }
132
133 // search for citation cross references
134 QCString citeName;
135
136 std::string lineStr;
137 while (getline(f,lineStr))
138 {
139 int i;
140 QCString line(lineStr);
141 if (line.stripWhiteSpace().startsWith("@"))
142 {
143 // assumption entry like: "@book { name," or "@book { name" (spaces optional)
144 int j = line.find('{');
145 // when no {, go hunting for it
146 while (j==-1 && getline(f,lineStr))
147 {
148 line = lineStr;
149 j = line.find('{');
150 }
151 // search for the name
152 citeName = "";
153 if (!f.eof() && j!=-1) // to prevent something like "@manual ," and no { found
154 {
155 int k = line.find(',',j);
156 j++;
157 // found a line "@....{.....,...." or "@.....{....."
158 // ^=j ^=k ^=j k=-1
159 while (!f.eof() && citeName.isEmpty())
160 {
161 if (k!=-1)
162 {
163 citeName = line.mid(static_cast<size_t>(j),static_cast<size_t>(k-j));
164 }
165 else
166 {
167 citeName = line.mid(static_cast<size_t>(j));
168 }
169 citeName = citeName.stripWhiteSpace();
170 j = 0;
171 if (citeName.isEmpty() && getline(f,lineStr))
172 {
173 line = lineStr;
174 k = line.find(',');
175 }
176 }
177 }
178 //printf("citeName = #%s#\n",qPrint(citeName));
179 }
180 else if ((i=line.find("crossref"))!=-1 && !citeName.isEmpty()) /* assumption cross reference is on one line and the only item */
181 {
182 int j = line.find('{',i);
183 int k = line.find('}',i);
184 if (j>i && k>j)
185 {
186 QCString crossrefName = line.mid(static_cast<size_t>(j+1),static_cast<uint>(k-j-1));
187 // check if the reference with the cross reference is used
188 // insert cross reference when cross reference has not yet been added.
189 if ((p->entries.find(citeName.str())!=p->entries.end()) &&
190 (p->entries.find(crossrefName.str())==p->entries.end())) // not found yet
191 {
192 insert(crossrefName);
193 }
194 }
195 }
196 }

199void CitationManager::generatePage()
200{
201 //printf("** CitationManager::generatePage() count=%d\n",m_ordering.count());
202
203 // do not generate an empty citations page
204 if (isEmpty()) return; // nothing to cite
205
206 bool citeDebug = Debug::isFlagSet(Debug::Cite);
207
208 // 0. add cross references from the bib files to the cite dictionary
209 const StringVector &citeDataList = Config_getList(CITE_BIB_FILES);
210 for (const auto &bibdata : citeDataList)
211 {
212 QCString bibFile = getBibFile(QCString(bibdata));
213 insertCrossReferencesForBibFile(bibFile);
214 }
215
216 // 1. generate file with markers and citations to OUTPUT_DIRECTORY
217 QCString outputDir = Config_getString(OUTPUT_DIRECTORY);
218 QCString citeListFile = outputDir+"/citelist.doc";
219 {
220 std::ofstream t = Portable::openOutputStream(citeListFile);
221 if (!t.is_open())
222 {
223 err("could not open file %s for writing\n",qPrint(citeListFile));
224 }
225 t << "<!-- BEGIN CITATIONS -->\n";
226 t << "<!--\n";
227 for (const auto &it : p->entries)
228 {
229 t << "\\citation{" << it.second->label() << "}\n";
230 }
231 t << "-->\n";
232 t << "<!-- END CITATIONS -->\n";
233 t << "<!-- BEGIN BIBLIOGRAPHY -->\n";
234 t << "<!-- END BIBLIOGRAPHY -->\n";
235 t.close();
236 }
237
238 // 2. generate bib2xhtml
239 QCString bib2xhtmlFile = outputDir+"/bib2xhtml.pl";
240 ResourceMgr::instance().copyResource("bib2xhtml.pl",outputDir);
241
242 // 3. generate doxygen.bst
243 QCString doxygenBstFile = outputDir+"/doxygen.bst";
244 ResourceMgr::instance().copyResource("doxygen.bst",outputDir);
245
246 // 4. for all formats we just copy the bib files to as special output directory
247 // so bibtex can find them without path (bibtex doesn't support paths or
248 // filenames with spaces!)
249 // Strictly not required when only latex is generated
250 QCString bibOutputDir = outputDir+"/"+bibTmpDir;
251 QCString bibOutputFiles = "";
252 Dir thisDir;
253 if (!thisDir.exists(bibOutputDir.str()) && !thisDir.mkdir(bibOutputDir.str()))
254 {
255 err("Failed to create temporary output directory '%s', skipping citations\n",qPrint(bibOutputDir));
256 return;
257 }
258 int i = 0;
259 for (const auto &bibdata : citeDataList)
260 {
261 QCString bibFile = getBibFile(QCString(bibdata));
262 FileInfo fi(bibFile.str());
263 if (fi.exists())
264 {
265 if (!bibFile.isEmpty())
266 {
267 ++i;
268 copyFile(bibFile,bibOutputDir + bibTmpFile + QCString().setNum(i) + ".bib");
269 bibOutputFiles = bibOutputFiles + " " + bibTmpDir + bibTmpFile + QCString().setNum(i) + ".bib";
270 }
271 }
272 }
273
274 std::string oldDir = Dir::currentDirPath();
275 Dir::setCurrent(outputDir.str());
276
277 // 5. run bib2xhtml perl script on the generated file which will insert the
278 // bibliography in citelist.doc
279 int exitCode;
280 QCString perlArgs = "\""+bib2xhtmlFile+"\" "+bibOutputFiles+" \""+ citeListFile+"\"";
281 if (citeDebug) perlArgs+=" -d";
282 if ((exitCode=Portable::system("perl",perlArgs)) != 0)
283 {
284 err("Problems running bibtex. Verify that the command 'perl --version' works from the command line. Exit code: %d\n",
285 exitCode);
286 }
287
288 Dir::setCurrent(oldDir);
289
290 // 6. read back the file
291 QCString doc;
292 {
293 std::ifstream f = Portable::openInputStream(citeListFile);
294 if (!f.is_open())
295 {
296 err("could not open file %s for reading\n",qPrint(citeListFile));
297 }
298
299 bool insideBib=FALSE;
300 //printf("input=[%s]\n",qPrint(input));
301 std::string lineStr;
302 while (getline(f,lineStr))
303 {
304 QCString line(lineStr);
305 //printf("pos=%d s=%d line=[%s]\n",pos,s,qPrint(line));
306
307 if (line.find("<!-- BEGIN BIBLIOGRAPHY")!=-1) insideBib=TRUE;
308 else if (line.find("<!-- END BIBLIOGRAPH")!=-1) insideBib=FALSE;
309 // determine text to use at the location of the @cite command
310 if (insideBib && ((i=line.find("name=\"CITEREF_"))!=-1 || (i=line.find("name=\"#CITEREF_"))!=-1))
311 {
312 int j=line.find("\">[");
313 int k=line.find("]</a>");
314 if (j!=-1 && k!=-1)
315 {
316 size_t ui=static_cast<size_t>(i);
317 size_t uj=static_cast<size_t>(j);
318 size_t uk=static_cast<size_t>(k);
319 QCString label = line.mid(ui+14,uj-ui-14);
320 QCString number = line.mid(uj+2,uk-uj-1);
321 line = line.left(ui+14) + label + line.right(line.length()-uj);
322 auto it = p->entries.find(label.str());
323 //printf("label='%s' number='%s' => %p\n",qPrint(label),qPrint(number),it->second.get());
324 if (it!=p->entries.end())
325 {
326 it->second->setText(number);
327 }
328 }
329 }
330 if (insideBib) doc+=line+"\n";
331 }
332 //printf("doc=[%s]\n",qPrint(doc));
333 }
334
335 // 7. add it as a page
336 addRelatedPage(fileName(),theTranslator->trCiteReferences(),doc,fileName(),1,1);
337
338 // 8. for latex we just copy the bib files to the output and let
339 // latex do this work.
340 if (Config_getBool(GENERATE_LATEX))
341 {
342 // copy bib files to the latex output dir
343 QCString latexOutputDir = Config_getString(LATEX_OUTPUT)+"/";
344 i = 0;
345 for (const auto &bibdata : citeDataList)
346 {
347 QCString bibFile = getBibFile(QCString(bibdata));
348 FileInfo fi(bibFile.str());
349 if (fi.exists())
350 {
351 if (!bibFile.isEmpty())
352 {
353 // bug_700510, multiple times the same name were overwriting; creating new names
354 // also for names with spaces
355 ++i;
356 copyFile(bibFile,latexOutputDir + bibTmpFile + QCString().setNum(i) + ".bib");
357 }
358 }
359 else
360 {
361 err("bib file %s not found!\n",qPrint(bibFile));
362 }
363 }
364 }
365
366 // 9. Remove temporary files
367 if (!citeDebug)
368 {
369 thisDir.remove(citeListFile.str());
370 thisDir.remove(doxygenBstFile.str());
371 thisDir.remove(bib2xhtmlFile.str());
372 // we might try to remove too many files as empty files didn't get a corresponding new file
373 // but the remove function does not emit an error for it and we don't catch the error return
374 // so no problem.
375 for (size_t j = 1; j <= citeDataList.size(); j++)
376 {
377 QCString bibFile = bibOutputDir + bibTmpFile + QCString().setNum(static_cast<ulong>(j)) + ".bib";
378 thisDir.remove(bibFile.str());
379 }
380 thisDir.rmdir(bibOutputDir.str());
381 }
382}

385{
386 QCString result;

388 int i = 0;

390 {

394 {
395 if (!bibFile.isEmpty())
