157 const std::unique_ptr<MemberList> &get(MemberListType lt,MemberListContainer con)
158 {
159 // find the list with the given type
160 auto it = std::find_if(begin(),end(),[&lt](const auto &ml) { return ml->listType()==lt; });
161 if (it!=end()) return *it;
162 // or create a new list if it is not found
163 emplace_back(std::make_unique<MemberList>(lt,con));
164 return back();
165 }
