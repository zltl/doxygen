537 // end of member sections
538 virtual QCString trWriteList(int numEntries)
539 {
540 // this function is used to produce a comma-separated list of items.
541 // use generateMarker(i) to indicate where item i should be put.
542 QCString result;
543 int i;
544 // the inherits list contain `numEntries' classes
545 for (i=0;i<numEntries;i++)
546 {
547 // use generateMarker to generate placeholders for the class links!
548 result+=generateMarker(i); // generate marker for entry i in the list
549 // (order is left to right)
550
551 if (i!=numEntries-1) // not the last entry, so we need a separator
552 {
553 if (i<numEntries-2) // not the fore last entry
554 result+="," CN_SPC;
555 else // the fore last entry
556 result+=CN_SPC ", 以及" CN_SPC;
557 }
558 }
559 return result;
560 }
