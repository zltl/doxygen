37 EdgeInfo(Colors color,Styles style,const QCString &lab,const QCString &url,int labColor)

45 static constexpr Colors protectionToColor(Protection prot)
46 {
47 switch (prot)
48 {
49 case Protection::Public: return Blue;
50 case Protection::Protected: return Green;
51 case Protection::Private: return Red;
52 case Protection::Package: return Purple;
53 }
54 return Blue;
55 }
