111DirDefImpl::DirDefImpl(const QCString &path) : DefinitionMixin(path,1,1,path)
112{
113 bool fullPathNames = Config_getBool(FULL_PATH_NAMES);
114 // get display name (stripping the paths mentioned in STRIP_FROM_PATH)
115 // get short name (last part of path)
116 m_shortName = path;
117 m_diskName = path;
118 if (m_shortName.at(m_shortName.length()-1)=='/')
119 { // strip trailing /
120 m_shortName = m_shortName.left(m_shortName.length()-1);
121 }
122 int pi=m_shortName.findRev('/');
123 if (pi!=-1)
124 { // remove everything till the last /
125 m_shortName = m_shortName.mid(pi+1);
126 }
127 setLocalName(m_shortName);
128 m_dispName = fullPathNames ? stripFromPath(path) : m_shortName;
129 if (m_dispName.length()>0 && m_dispName.at(m_dispName.length()-1)=='/')
130 { // strip trailing /
131 m_dispName = m_dispName.left(m_dispName.length()-1);
132 }
133
134 m_level=-1;
135 m_parent=0;
136}

643/** Add as "uses" dependency between \a this dir and \a dir,
644 * that was caused by a dependency on file \a fd.
645 * srcDirect and dstDirect indicate if it is a direct dependencies (true) or if
646 * the dependencies was indirect (e.g. a parent dir that has a child dir that has the dependencies)
647 */
648void DirDefImpl::addUsesDependency(const DirDef *dir,const FileDef *srcFd,
649 const FileDef *dstFd,bool srcDirect, bool dstDirect)
650{
651 if (this==dir) return; // do not add self-dependencies
652 //static int count=0;
653 //printf(" %d add dependency %s->%s due to %s->%s\n",
654 // count++,qPrint(shortName()),
655 // qPrint(dir->shortName()),
656 // qPrint(srcFd->name()),
657 // qPrint(dstFd->name()));
658
659 // levels match => add direct dependency
660 bool added=FALSE;
661 UsedDir *usedDir = m_usedDirs.find(dir->getOutputFileBase());
662 if (usedDir) // dir dependency already present
663 {
664 const FilePair *usedPair = usedDir->findFilePair(FilePair::key(srcFd,dstFd));
665 if (usedPair==0) // new file dependency
666 {
667 //printf(" => new file\n");
668 usedDir->addFileDep(srcFd,dstFd, srcDirect, dstDirect);
669 added=TRUE;
670 }
671 else
672 {
673 // dir & file dependency already added
674 }
675 }
676 else // new directory dependency
677 {
678 //printf(" => new file\n");
679 auto newUsedDir = std::make_unique<UsedDir>(dir);
680 newUsedDir->addFileDep(srcFd,dstFd, srcDirect, dstDirect);
681 m_usedDirs.add(dir->getOutputFileBase(),std::move(newUsedDir));
682 added=TRUE;
683 }
684 if (added)
685 {
686 if (dir->parent())
687 {
688 // add relation to parent of used dir
689 addUsesDependency(dir->parent(),
690 srcFd,
691 dstFd,
692 srcDirect,
693 false); // indirect dependency on dest dir
694 }
695 if (parent())
696 {
697 // add relation for the parent of this dir as well
698 parent()->addUsesDependency(dir,
699 srcFd,
700 dstFd,
701 false, // indirect dependency from source dir
702 dstDirect);
703 }
704 }

707/** Computes the dependencies between directories
708 */
709void DirDefImpl::computeDependencies()
710{
711 for (const auto &fd : m_fileList)
712 {
713 //printf(" File %s\n",qPrint(fd->name()));
714 //printf("** dir=%s file=%s\n",qPrint(shortName()),qPrint(fd->name()));
715 for (const auto &ii : fd->includeFileList())
716 {
717 //printf(" > %s\n",qPrint(ii->includeName));
718 //printf(" #include %s\n",qPrint(ii->includeName));
719 if (ii.fileDef && ii.fileDef->isLinkable()) // linkable file
720 {
721 DirDef *usedDir = ii.fileDef->getDirDef();
722 if (usedDir)
723 {
724 // add dependency: thisDir->usedDir
725 //static int count=0;
726 //printf(" %d: add dependency %s->%s\n",count++,qPrint(name()),qPrint(usedDir->name()));
727 addUsesDependency(usedDir,fd,ii.fileDef,true,true);
728 }
729 }
730 }
731 }
732
733 std::sort(m_usedDirs.begin(),m_usedDirs.end(),
734 [](const auto &u1,const auto &u2)
735 { return qstricmp(u1->dir()->getOutputFileBase(),u2->dir()->getOutputFileBase())<0; });
736
737 for (const auto& usedDirectory : m_usedDirs)
738 {
739 usedDirectory->sort();
740 }

937/** In order to create stable, but unique directory names,
938 * we compute the common part of the path shared by all directories.
939 */
940static void computeCommonDirPrefix()
941{
942 QCString path;
943 auto it = Doxygen::dirLinkedMap->begin();
944 if (!Doxygen::dirLinkedMap->empty()) // we have at least one dir
945 {
946 // start will full path of first dir
947 path=(*it)->name();
948 int i=path.findRev('/',path.length()-2);
949 path=path.left(i+1);
950 bool done=FALSE;
951 if (i==-1)
952 {
953 path="";
954 }
955 else
956 {
957 while (!done)
958 {
959 uint l = path.length();
960 size_t count=0;
961 for (const auto &dir : *Doxygen::dirLinkedMap)
962 {
963 QCString dirName = dir->name();
964 if (dirName.length()>path.length())
965 {
966 if (dirName.left(l)!=path) // dirName does not start with path
967 {
968 i=path.findRev('/',l-2);
969 if (i==-1) // no unique prefix -> stop
970 {
971 path="";
972 done=TRUE;
973 }
974 else // restart with shorter path
975 {
976 path=path.left(i+1);
977 break;
978 }
979 }
980 }
981 else // dir is shorter than path -> take path of dir as new start
982 {
983 path=dir->name();
984 l=path.length();
985 i=path.findRev('/',l-2);
986 if (i==-1) // no unique prefix -> stop
987 {
988 path="";
989 done=TRUE;
990 }
991 else // restart with shorter path
992 {
993 path=path.left(i+1);
994 }
995 break;
996 }
997 count++;
998 }
999 if (count==Doxygen::dirLinkedMap->size())
1000 // path matches for all directories -> found the common prefix
1001 {
1002 done=TRUE;
1003 }
1004 }
1005 }
1006 }
1007 for (const auto &dir : *Doxygen::dirLinkedMap)
1008 {
1009 QCString diskName = dir->name().right(dir->name().length()-path.length());
1010 dir->setDiskName(diskName);
1011 //printf("set disk name: %s -> %s\n",qPrint(dir->name()),qPrint(diskName));
