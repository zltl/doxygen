301void ConceptDefImpl::writeBriefDescription(OutputList &ol) const
302{
303 if (hasBriefDescription())
304 {
305 auto parser { createDocParser() };
306 auto ast { validatingParseDoc(
307 *parser.get(),briefFile(),briefLine(),this,0,
308 briefDescription(),TRUE,FALSE,
309 QCString(),TRUE,FALSE,Config_getBool(MARKDOWN_SUPPORT)) };
310 if (!ast->isEmpty())
311 {
312 ol.startParagraph();
313 ol.pushGeneratorState();
314 ol.disableAllBut(OutputType::Man);
315 ol.writeString(" - ");
316 ol.popGeneratorState();
317 ol.writeDoc(ast.get(),this,0);
318 ol.pushGeneratorState();
319 ol.disable(OutputType::RTF);
320 ol.writeString(" \n");
321 ol.enable(OutputType::RTF);
322
323 if (hasDetailedDescription())
324 {
325 ol.disableAllBut(OutputType::Html);
326 ol.startTextLink(getOutputFileBase(),"details");
327 ol.parseText(theTranslator->trMore());
328 ol.endTextLink();
329 }
330 ol.popGeneratorState();
331 ol.endParagraph();
332 }
333 }
334 ol.writeSynopsis();
335}

337void ConceptDefImpl::writeIncludeFiles(OutputList &ol) const
338{
339 if (m_incInfo)
340 {
341 QCString nm=m_incInfo->includeName.isEmpty() ?
342 (m_incInfo->fileDef ?
343 m_incInfo->fileDef->docName() : QCString()
344 ) :
345 m_incInfo->includeName;
346 if (!nm.isEmpty())
347 {
348 ol.startParagraph();
349 ol.startTypewriter();
350 ol.docify("#include ");
351 if (m_incInfo->local)
352 ol.docify("\"");
353 else
354 ol.docify("<");
355 ol.pushGeneratorState();
356 ol.disable(OutputType::Html);
357 ol.docify(nm);
358 ol.disableAllBut(OutputType::Html);
359 ol.enable(OutputType::Html);
360 if (m_incInfo->fileDef)
361 {
362 ol.writeObjectLink(QCString(),m_incInfo->fileDef->includeName(),QCString(),nm);
363 }
364 else
365 {
366 ol.docify(nm);
367 }
368 ol.popGeneratorState();
369 if (m_incInfo->local)
370 ol.docify("\"");
371 else
372 ol.docify(">");
373 ol.endTypewriter();
374 ol.endParagraph();
375 }

379static QCString templateSpec(const ArgumentList &al)
380{
381 TextStream t;
382 t << "template<";
383 bool first=true;
384 for (const auto &a : al)
385 {
386 if (!first) t << ", ";
387 t << a.type;
388 if (!a.name.isEmpty())
389 {
390 t << " " << a.name;
391 }
392 if (a.defval.length()!=0)
393 {
394 t << " " << a.defval;
395 }
396 first=false;
397 }
398 t << ">";
399 return t.str();
400}

412void ConceptDefImpl::writeDefinition(OutputList &ol,const QCString &title) const
413{
414 ol.startGroupHeader();
415 ol.parseText(title);
416 ol.endGroupHeader();
417
418 auto intf = Doxygen::parserManager->getCodeParser(".cpp");
419 intf->resetCodeParserState();
420 auto &codeOL = ol.codeGenerators();
421 codeOL.startCodeFragment("DoxyCode");
422 QCString scopeName;
423 if (getOuterScope()!=Doxygen::globalScope) scopeName=getOuterScope()->name();
424 TextStream conceptDef;
425 conceptDef << m_initializer;
426 intf->parseCode(codeOL,scopeName,conceptDef.str(),SrcLangExt_Cpp,false,QCString(),
427 m_fileDef, -1,-1,true,0,false,this);
428 codeOL.endCodeFragment("DoxyCode");
429}

431void ConceptDefImpl::writeDetailedDescription(OutputList &ol,const QCString &title) const
432{
433 bool repeatBrief = Config_getBool(REPEAT_BRIEF);
434 if (hasDetailedDescription())
435 {
436 ol.pushGeneratorState();
437 ol.disable(OutputType::Html);
438 ol.writeRuler();
439 ol.popGeneratorState();
440
441 ol.pushGeneratorState();
442 ol.disableAllBut(OutputType::Html);
443 ol.writeAnchor(QCString(),"details");
444 ol.popGeneratorState();
445
446 ol.startGroupHeader();
447 ol.parseText(title);
448 ol.endGroupHeader();
449
450 ol.startTextBlock();
451 // repeat brief description
452 if (!briefDescription().isEmpty() && repeatBrief)
453 {
454 ol.generateDoc(briefFile(),briefLine(),this,0,briefDescription(),FALSE,FALSE,
455 QCString(),FALSE,FALSE,Config_getBool(MARKDOWN_SUPPORT));
456 }
457 if (!briefDescription().isEmpty() && repeatBrief &&
458 !documentation().isEmpty())
459 {
460 ol.pushGeneratorState();
461 ol.disable(OutputType::Html);
462 ol.writeString("\n\n");
463 ol.popGeneratorState();
464 }
465
466 // write documentation
467 if (!documentation().isEmpty())
468 {
469 ol.generateDoc(docFile(),docLine(),this,0,documentation(),TRUE,FALSE,
470 QCString(),FALSE,FALSE,Config_getBool(MARKDOWN_SUPPORT));
471 }
472
473 writeSourceDef(ol,name());
474 ol.endTextBlock();
475 }

490void ConceptDefImpl::writeDocumentation(OutputList &ol)
491{
492 bool generateTreeView = Config_getBool(GENERATE_TREEVIEW);
493 QCString pageTitle = theTranslator->trConceptReference(displayName());
494 startFile(ol,getOutputFileBase(),name(),pageTitle,HighlightedItem::ConceptVisible,!generateTreeView);
495
496 // ---- navigation part
497 if (!generateTreeView)
498 {
499 if (getOuterScope()!=Doxygen::globalScope)
500 {
501 writeNavigationPath(ol);
502 }
503 ol.endQuickIndices();
504 }
505
506 // ---- title part
507 startTitle(ol,getOutputFileBase(),this);
508 ol.parseText(pageTitle);
509 addGroupListToTitle(ol,this);
510 endTitle(ol,getOutputFileBase(),displayName());
511
512 // ---- contents part
513
514 ol.startContents();
515
516 //---------------------------------------- start flexible part -------------------------------
517
518 for (const auto &lde : LayoutDocManager::instance().docEntries(LayoutDocManager::Concept))
519 {
520 const LayoutDocEntrySection *ls = dynamic_cast<const LayoutDocEntrySection*>(lde.get());
521 switch (lde->kind())
522 {
523 case LayoutDocEntry::BriefDesc:
524 writeBriefDescription(ol);
525 break;
526 case LayoutDocEntry::ConceptDefinition:
527 if (ls) writeDefinition(ol,ls->title(getLanguage()));
528 break;
529 case LayoutDocEntry::DetailedDesc:
530 if (ls) writeDetailedDescription(ol,ls->title(getLanguage()));
531 break;
532 case LayoutDocEntry::AuthorSection:
533 writeAuthorSection(ol);
534 break;
535 case LayoutDocEntry::ClassIncludes:
536 writeIncludeFiles(ol);
537 break;
538 case LayoutDocEntry::MemberDeclStart:
539 case LayoutDocEntry::NamespaceClasses:
540 case LayoutDocEntry::NamespaceInterfaces:
541 case LayoutDocEntry::NamespaceStructs:
542 case LayoutDocEntry::NamespaceExceptions:
543 case LayoutDocEntry::NamespaceConcepts:
544 case LayoutDocEntry::NamespaceNestedNamespaces:
545 case LayoutDocEntry::NamespaceNestedConstantGroups:
546 case LayoutDocEntry::MemberGroups:
547 case LayoutDocEntry::MemberDecl:
548 case LayoutDocEntry::MemberDeclEnd:
549 case LayoutDocEntry::MemberDefStart:
550 case LayoutDocEntry::NamespaceInlineClasses:
551 case LayoutDocEntry::MemberDef:
552 case LayoutDocEntry::MemberDefEnd:
553 case LayoutDocEntry::ClassInheritanceGraph:
554 case LayoutDocEntry::ClassNestedClasses:
555 case LayoutDocEntry::ClassCollaborationGraph:
556 case LayoutDocEntry::ClassAllMembersLink:
557 case LayoutDocEntry::ClassUsedFiles:
558 case LayoutDocEntry::ClassInlineClasses:
559 case LayoutDocEntry::FileClasses:
560 case LayoutDocEntry::FileConcepts:
561 case LayoutDocEntry::FileInterfaces:
562 case LayoutDocEntry::FileStructs:
563 case LayoutDocEntry::FileExceptions:
564 case LayoutDocEntry::FileNamespaces:
565 case LayoutDocEntry::FileConstantGroups:
566 case LayoutDocEntry::FileIncludes:
567 case LayoutDocEntry::FileIncludeGraph:
568 case LayoutDocEntry::FileIncludedByGraph:
569 case LayoutDocEntry::FileInlineClasses:
570 case LayoutDocEntry::FileSourceLink:
571 case LayoutDocEntry::GroupClasses:
572 case LayoutDocEntry::GroupConcepts:
573 case LayoutDocEntry::GroupInlineClasses:
574 case LayoutDocEntry::GroupNamespaces:
575 case LayoutDocEntry::GroupDirs:
576 case LayoutDocEntry::GroupNestedGroups:
577 case LayoutDocEntry::GroupFiles:
578 case LayoutDocEntry::GroupGraph:
579 case LayoutDocEntry::GroupPageDocs:
580 case LayoutDocEntry::DirSubDirs:
581 case LayoutDocEntry::DirFiles:
582 case LayoutDocEntry::DirGraph:
583 err("Internal inconsistency: member %d should not be part of "
584 "LayoutDocManager::Namespace entry list\n",lde->kind());
585 break;
586 }
587 }
588
589 //---------------------------------------- end flexible part -------------------------------
590
591 ol.endContents();
592
593 endFileWithNavPath(ol,this);
594}

596void ConceptDefImpl::writeDeclarationLink(OutputList &ol,bool &found,const QCString &header,bool localNames) const
597{
598 if (isLinkable())
599 {
600 if (!found) // first concept
601 {
602 ol.startMemberHeader("concepts");
603 if (!header.isEmpty())
604 {
605 ol.parseText(header);
606 }
607 else
608 {
609 theTranslator->trConcept(true,false);
610 }
611 ol.endMemberHeader();
612 ol.startMemberList();
613 found=TRUE;
614 }
615 ol.startMemberDeclaration();
616 ol.startMemberItem(anchor(),OutputGenerator::MemberItemType::Normal);
617 ol.writeString("concept ");
618 QCString cname = displayName(!localNames);
619 ol.insertMemberAlign();
620 if (isLinkable())
621 {
622 ol.writeObjectLink(getReference(),
623 getOutputFileBase(),
624 anchor(),
625 cname
626 );
627 }
628 else
629 {
630 ol.startBold();
631 ol.docify(cname);
632 ol.endBold();
633 }
634 ol.endMemberItem(OutputGenerator::MemberItemType::Normal);
635 // add the brief description if available
636 if (!briefDescription().isEmpty() && Config_getBool(BRIEF_MEMBER_DESC))
637 {
638 auto parser { createDocParser() };
639 auto ast { validatingParseDoc(
640 *parser.get(),briefFile(),briefLine(),this,0,
641 briefDescription(),FALSE,FALSE,
642 QCString(),TRUE,FALSE,Config_getBool(MARKDOWN_SUPPORT)) };
643 if (!ast->isEmpty())
644 {
645 ol.startMemberDescription(anchor());
646 ol.writeDoc(ast.get(),this,0);
647 ol.endMemberDescription();
648 }
649 }
650 ol.endMemberDeclaration(anchor(),QCString());
651 }
652}
