81inline void writeDocbookCodeString(TextStream &t,const QCString &str, int &col)
82{
83 if (str.isEmpty()) return;
84 const char *s = str.data();
85 char c;
86 while ((c=*s++))
87 {
88 switch(c)
89 {
90 case '\t':
91 {
92 int tabSize = Config_getInt(TAB_SIZE);
93 int spacesToNextTabStop = tabSize - (col%tabSize);
94 col+=spacesToNextTabStop;
95 while (spacesToNextTabStop--) t << "&#32;";
96 break;
97 }
98 case ' ': t << "&#32;"; col++; break;
99 case '<': t << "&lt;"; col++; break;
100 case '>': t << "&gt;"; col++; break;
101 case '&': t << "&amp;"; col++; break;
102 case '\'': t << "&apos;"; col++; break;
103 case '"': t << "&quot;"; col++; break;
104 default:
105 {
106 uchar uc = static_cast<uchar>(c);
107 static const char *hex="0123456789ABCDEF";
108 if (uc<32)
109 {
110 t << "&#x24" << hex[uc>>4] << hex[uc&0xF] << ";";
111 }
112 else
113 {
114 t << c;
115 }
116 col++;
117 }
118 break;
119 }
120 }
121}

224void DocbookCodeGenerator::writeLineNumber(const QCString &ref,const QCString &fileName,
225 const QCString &anchor,int l,bool writeLineAnchor)
226{
227 Docbook_DB(("(writeLineNumber)\n"));
228 m_insideCodeLine = TRUE;
229 if (Config_getBool(SOURCE_BROWSER))
230 {
231 QCString lineNumber;
232 lineNumber.sprintf("%05d",l);
233
234 if (!m_sourceFileName.isEmpty())
235 {
236 writeCodeLinkLine(CodeSymbolType::Default,ref,m_sourceFileName,anchor,lineNumber,QCString(),writeLineAnchor);
237 }
238 if (!fileName.isEmpty())
239 {
240 writeCodeLink(CodeSymbolType::Default,ref,fileName,anchor,lineNumber,QCString());
241 }
242 else
243 {
244 codify(lineNumber);
245 }
246 m_t << " ";
247 }
248 else
249 {
250 m_t << l << " ";
251 }
252 m_col=0;
253}

1333/*! Converts a string to an DocBook-encoded string */
1334QCString convertToDocBook(const QCString &s, const bool retainNewline)
1335{
1336 if (s.isEmpty()) return s;
1337 GrowBuf growBuf;
1338 const char *q;
1339 int cnt;
1340 const char *p=s.data();
1341 char c;
1342 while ((c=*p++))
1343 {
1344 switch (c)
1345 {
1346 case '\n': if (retainNewline) growBuf.addStr("<literallayout>&#160;&#xa;</literallayout>"); growBuf.addChar(c); break;
1347 case '<': growBuf.addStr("&lt;"); break;
1348 case '>': growBuf.addStr("&gt;"); break;
1349 case '&': // possibility to have a special symbol
1350 q = p;
1351 cnt = 2; // we have to count & and ; as well
1352 while ((*q >= 'a' && *q <= 'z') || (*q >= 'A' && *q <= 'Z') || (*q >= '0' && *q <= '9'))
1353 {
1354 cnt++;
1355 q++;
1356 }
1357 if (*q == ';')
1358 {
1359 --p; // we need & as well
1360 HtmlEntityMapper::SymType res = HtmlEntityMapper::instance().name2sym(QCString(p).left(cnt));
1361 if (res == HtmlEntityMapper::Sym_Unknown)
1362 {
1363 p++;
1364 growBuf.addStr("&amp;");
1365 }
1366 else
1367 {
1368 growBuf.addStr(HtmlEntityMapper::instance().docbook(res));
1369 q++;
1370 p = q;
1371 }
1372 }
1373 else
1374 {
1375 growBuf.addStr("&amp;");
1376 }
1377 break;
1378 case '\'': growBuf.addStr("&apos;"); break;
1379 case '"': growBuf.addStr("&quot;"); break;
1380 case 1: case 2: case 3: case 4: case 5: case 6: case 7: case 8:
1381 case 11: case 12: case 14: case 15: case 16: case 17: case 18:
1382 case 19: case 20: case 21: case 22: case 23: case 24: case 25: case 26:
1383 case 27: case 28: case 29: case 30: case 31:
1384 growBuf.addStr("&#x24");
1385 growBuf.addChar(hex[static_cast<uchar>(c)>>4]);
1386 growBuf.addChar(hex[static_cast<uchar>(c)&0xF]);
1387 growBuf.addChar(';');
1388 break;
1389 default:
1390 growBuf.addChar(c);
1391 break;
1392 }
1393 }
1394 growBuf.addChar(0);
1395 return growBuf.get();
1396}
