305{
306 public:
307 enum Type {
308 None = 0, // initial value
309 Html = 0, // index / also to be used as bit position in mask (1 << Html)
310 Latex = 1, // ...
311 Xml = 2, // ...
312 Docbook = 3, // ...
313 numTocTypes = 4 // number of enum values
314 };
315 LocalToc() : m_mask(None) { memset(m_level,0,sizeof(m_level)); }
316
317 // setters
318 void enableHtml(int level)
319 {
320 m_mask|=(1<<Html);
321 m_level[Html]=level;
322 }
323 void enableLatex(int level)
324 {
325 m_mask|=(1<<Latex);
326 m_level[Latex]=level;
327 }
328 void enableXml(int level)
329 {
330 m_mask|=(1<<Xml);
331 m_level[Xml]=level;
332 }
333 void enableDocbook(int level)
334 {
335 m_mask|=(1<<Docbook);
336 m_level[Docbook]=level;
337 }
338
339 // getters
340 bool isHtmlEnabled() const { return (m_mask & (1<<Html))!=0; }
341 bool isLatexEnabled() const { return (m_mask & (1<<Latex))!=0; }
342 bool isXmlEnabled() const { return (m_mask & (1<<Xml))!=0; }
343 bool isDocbookEnabled() const { return (m_mask & (1<<Docbook))!=0; }
344 bool nothingEnabled() const { return m_mask == None; }
345 int htmlLevel() const { return m_level[Html]; }
346 int latexLevel() const { return m_level[Latex]; }
347 int xmlLevel() const { return m_level[Xml]; }
348 int docbookLevel() const { return m_level[Docbook]; }
349 int mask() const { return m_mask; }
350
351 private:
352 int m_mask;
353 int m_level[numTocTypes];
354};
