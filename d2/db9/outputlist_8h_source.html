556 // For each output format that is enabled (OutputGenerator::isEnabled()) we forward

558 // We use C++11 variadic templates and perfect forwarding to implement forall() generically,
559 // and split the types of the methods from the arguments passed to allow implicit conversions.
560 template<typename T,class... Ts,class... As>
561 void forall(void (T::*methodPtr)(Ts...),As&&... args)
562 {

564 {
565 if (og->isEnabled()) (og.get()->*methodPtr)(std::forward<As>(args)...);
