297 QCString trWriteList(int numEntries)
298 {
299 // this function is used to produce a comma-separated list of items.
300 // use generateMarker(i) to indicate where item i should be put.
301 QCString result;
302 int i;
303 // the inherits list contain `numEntries' classes
304 for (i=0;i<numEntries;i++)
305 {
306 // use generateMarker to generate placeholders for the class links!
307 result+=generateMarker(i); // generate marker for entry i in the list
308 // (order is left to right)
309
310 if (i!=numEntries-1) // not the last entry, so we need a separator
311 {
312 if (i<numEntries-2) // not the fore last entry
313 result+=", ";
314 else // the fore last entry
315 result+=" in ";
316 }
317 }
318 return result;
319 }

384 /*! This is put at the bottom of a class documentation page and is
385 * followed by a list of files that were used to generate the page.
386 */
387 QCString trGeneratedFromFiles(ClassDef::CompoundType compType,
388 bool single)
389 { // here s is one of " Class", " Struct" or " Union"
390 // single is true implies a single file
391 QCString result="Opis ";
392 switch(compType)
393 {
394 case ClassDef::Class: result+="razreda"; break;
395 case ClassDef::Struct: result+="strukture"; break;
396 case ClassDef::Union: result+="unije"; break;
397 case ClassDef::Interface: result+="vmesnika (interface)"; break;
398 case ClassDef::Protocol: result+="protocol"; break; // translate me!
399 case ClassDef::Category: result+="category"; break; // translate me!
400 case ClassDef::Exception: result+="prekinitve (exception)"; break;
401 default: break;
402 }
403 result+=" je zgrajen na podlagi naslednj";
404 if (single) result+="e "; else result+="ih";
405 result+=" datotek";
406 if (single) result+="e :"; else result+=" :";
407 return result;
408 }
