24/*! @brief Buffer used to store strings
25 *
26 * This buffer is used append characters and strings. It will automatically
27 * resize itself, yet provide efficient random access to the content.
28 */
29class BufStr
30{
31 public:
32 BufStr(size_t size)
33 : m_size(size), m_writeOffset(0), m_spareRoom(10240), m_buf(0)
34 {
35 m_buf = static_cast<char *>(calloc(size,1));
36 }
37 ~BufStr()
38 {
39 free(m_buf);
40 }
41 void addChar(char c)
42 {
43 makeRoomFor(1);
44 m_buf[m_writeOffset++]=c;
45 }
46 void addArray(const char *a,size_t len)
47 {
48 makeRoomFor(len);
49 memcpy(m_buf+m_writeOffset,a,len);
50 m_writeOffset+=len;
51 }
52 void skip(size_t s)
53 {
54 makeRoomFor(s);
55 m_writeOffset+=s;
56 }
57 void shrink( size_t newlen )
58 {
59 m_writeOffset=newlen;
60 resize(newlen);
61 }
62 void resize( size_t newlen )
63 {
64 size_t oldsize = m_size;
65 m_size=newlen;
66 if (m_writeOffset>=m_size) // offset out of range -> enlarge
67 {
68 m_size=m_writeOffset+m_spareRoom;
69 }
70 m_buf = static_cast<char *>(realloc(m_buf,m_size));
71 if (m_size>oldsize)
72 {
73 memset(m_buf+oldsize,0,m_size-oldsize);
74 }
75 }
76 size_t size() const
77 {
78 return m_size;
79 }
80 const char *data() const
81 {
82 return m_buf;
83 }
84 char *data()
85 {
86 return m_buf;
87 }
88 const char &at(size_t i) const
89 {
90 return m_buf[i];
91 }
92 char &at(size_t i)
93 {
94 return m_buf[i];
95 }
96 bool isEmpty() const
97 {
98 return m_writeOffset==0;
99 }
100 operator const char *() const
101 {
102 return m_buf;
103 }
104 size_t curPos() const
105 {
106 return m_writeOffset;
107 }
108 void dropFromStart(size_t bytes)
109 {
110 if (bytes>m_size) bytes=m_size;
111 if (bytes>0) qmemmove(m_buf,m_buf+bytes,m_size-bytes);
112 m_size-=bytes;
113 m_writeOffset-=bytes;
114 }
115 void addTerminalCharIfMissing(char c)
116 {
117 if (m_buf && m_writeOffset>1 && m_buf[m_writeOffset-2]!=c && m_buf[m_writeOffset-1]=='\0')
118 {
119 // add missing terminal character and 0 terminator
120 m_buf[m_writeOffset-1] = '\n';
121 addChar('\0');
122 }
123 else if (m_buf && m_writeOffset>0 && m_buf[m_writeOffset-1]!='\0' && m_buf[m_writeOffset-1]!=c)
124 {
125 // add missing terminal character without 0 terminator
126 addChar(c);
127 }
128 }
129 private:
130 void makeRoomFor(size_t size)
131 {
132 if (m_writeOffset+size>=m_size)
133 {
134 resize(m_size+size+m_spareRoom);
135 }
136 }
137 size_t m_size;
138 size_t m_writeOffset;
139 const size_t m_spareRoom; // 10Kb extra room to avoid frequent resizing
140 char *m_buf;
141};
