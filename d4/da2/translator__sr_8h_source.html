632 /*! this function is used to produce a comma-separated list of items.
633 * use generateMarker(i) to indicate where item i should be put.
634 */
635 virtual QCString trWriteList(int numEntries)
636 {
637 QCString result;
638 int i;
639 // the inherits list contain `numEntries' classes
640 for (i=0;i<numEntries;i++)
641 {
642 // use generateMarker to generate placeholders for the class links!
643 result+=generateMarker(i); // generate marker for entry i in the list
644 // (order is left to right)
645
646 if (i!=numEntries-1) // not the last entry, so we need a separator
647 {
648 if (i<numEntries-2) // not the fore last entry
649 result+=", ";
650 else // the fore last entry
651 result+=" i ";
652 }
653 }
654 return result;
655 }
