634 /*! this function is used to produce a comma-separated list of items.
635 * use generateMarker(i) to indicate where item i should be put.
636 */
637 virtual QCString trWriteList(int numEntries)
638 {
639 QCString result;
640 int i;
641 // the inherits list contain `numEntries' classes
642 for (i=0;i<numEntries;i++)
643 {
644 // use generateMarker to generate placeholders for the class links!
645 result+=generateMarker(i); // generate marker for entry i in the list
646 // (order is left to right)
647
648 if (i!=numEntries-1) // not the last entry, so we need a separator
649 {
650 if (i<numEntries-2) // not the fore last entry
651 result+=", ";
652 else // the fore last entry
653 result+=" и ";
654 }
655 }
656 return result;
657 }
