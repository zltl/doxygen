435 QCString trWriteList(int numEntries)
436 {
437 // this function is used to produce a comma-separated list of items.
438 // use generateMarker(i) to indicate where item i should be put.
439 QCString result;
440 int i;
441 // the inherits list contain `numEntries' classes
442 for (i=0;i<numEntries;i++)
443 {
444 // use generateMarker to generate placeholders for the class links!
445 result+=generateMarker(i); // generate marker for entry i in the list
446 // (order is left to right)
447
448 if (i!=numEntries-1) // not the last entry, so we need a separator
449 {
450 if (i<numEntries-2) // not the fore last entry
451 result+=", ";
452 else // the fore last entry
453 result+=" en ";
454 }
455 }
456 return result;
457 }
