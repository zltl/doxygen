142/*!
143 It is not necessary to include "translator.h" or
144 "translator_adapter.h" here. The files are included in the
145 language.cpp correctly. Not including any of the mentioned
146 files frees the maintainer from thinking about whether the
147 first, the second, or both files should be included or not, and
148 why. This holds namely for localized translators because their
149 base class is changed occasionally to adapter classes when the
150 Translator class changes the interface, or back to the
151 Translator class (by the local maintainer) when the localized
152 translator is made up-to-date again.
153*/

716 /*! this function is used to produce a comma-separated list of items.
717 * use generateMarker(i) to indicate where item i should be put.
718 */
719 virtual QCString trWriteList(int numEntries)
720 {
721 QCString result;
722 int i;
723 // the inherits list contain `numEntries' classes
724 for (i = 0; i < numEntries; ++i)
725 {
726 // use generateMarker to generate placeholders for the class links!
727 result += generateMarker(i); // generate marker for entry i in the list
728 // (order is left to right)
729 if (i != numEntries - 1) // not the last entry, so we need a separator
730 {
731 if (i < numEntries - 2) // not the fore last entry
732 result += ", ";
733 else // the fore last entry
734 result += ", और ";
735 }
736 }
737 return result;

1762 /** Compiles a date string.

1764 * @param month Month of the year: 1=January
1765 * @param day Day of the Month: 1..31
1766 * @param dayOfWeek Day of the week: 1=Monday..7=Sunday
1767 * @param hour Hour of the day: 0..23
1768 * @param minutes Minutes in the hour: 0..59
1769 * @param seconds Seconds within the minute: 0..59
1770 * @param includeTime Include time in the result string?
1771 */
1772 virtual QCString trDateTime(
1773 int year, int month, int day, int dayOfWeek,
1774 int hour, int minutes, int seconds, DateTimeType includeTime)
1775 {
1776 static const char *days[] = { "सोमवार", "मंगलवार", "बुधवार", "गुरुवार",

1779 "जुलाई", "अगस्त", "सितम्बर", "अक्टूबर", "नवम्बर", "दिसम्बर" };
1780 QCString sdate;
1781 if (includeTime == DateTimeType::DateTime || includeTime == DateTimeType::Date)

1786 if (includeTime == DateTimeType::DateTime || includeTime == DateTimeType::Time)
1787 {
1788 QCString stime;

1790 sdate += stime;
1791 }
1792 return sdate;
