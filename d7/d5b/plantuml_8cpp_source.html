27QCString PlantumlManager::writePlantUMLSource(const QCString &outDirArg,const QCString &fileName,
28 const QCString &content,OutputFormat format, const QCString &engine,
29 const QCString &srcFile,int srcLine)
30{
31 QCString baseName;
32 QCString puName;
33 QCString imgName;
34 QCString outDir(outDirArg);
35 static int umlindex=1;
36
37 Debug::print(Debug::Plantuml,0,"*** %s fileName: %s\n","writePlantUMLSource",qPrint(fileName));
38 Debug::print(Debug::Plantuml,0,"*** %s outDir: %s\n","writePlantUMLSource",qPrint(outDir));
39
40 // strip any trailing slashes and backslashes
41 uint l;
42 while ((l=outDir.length())>0 && (outDir.at(l-1)=='/' || outDir.at(l-1)=='\\'))
43 {
44 outDir = outDir.left(l-1);
45 }
46
47 if (fileName.isEmpty()) // generate name
48 {
49 puName = "inline_umlgraph_"+QCString().setNum(umlindex);
50 baseName = outDir+"/inline_umlgraph_"+QCString().setNum(umlindex++);
51 }
52 else // user specified name
53 {
54 baseName = fileName;
55 int i=baseName.findRev('.');
56 if (i!=-1) baseName = baseName.left(i);
57 puName = baseName;
58 baseName.prepend(outDir+"/");
59 }
60
61 switch (format)
62 {
63 case PUML_BITMAP:
64 imgName =puName+".png";
65 break;
66 case PUML_EPS:
67 imgName =puName+".eps";
68 break;
69 case PUML_SVG:
70 imgName =puName+".svg";
71 break;
72 }
73
74 Debug::print(Debug::Plantuml,0,"*** %s baseName: %s\n","writePlantUMLSource",qPrint(baseName));
75 Debug::print(Debug::Plantuml,0,"*** %s puName: %s\n","writePlantUMLSource",qPrint(puName));
76 Debug::print(Debug::Plantuml,0,"*** %s imgName: %s\n","writePlantUMLSource",qPrint(imgName));
77
78 QCString text = "@start"+engine+" "+imgName+"\n";
79 text+=content;
80 text+="\n@end"+engine+"\n";
81
82 QCString qcOutDir(outDir);
83 uint pos = qcOutDir.findRev("/");
84 QCString generateType(qcOutDir.right(qcOutDir.length() - (pos + 1)) );
85 Debug::print(Debug::Plantuml,0,"*** %s generateType: %s\n","writePlantUMLSource",qPrint(generateType));
86 PlantumlManager::instance().insert(generateType.str(),puName.str(),outDir,format,text,srcFile,srcLine);
87 Debug::print(Debug::Plantuml,0,"*** %s generateType: %s\n","writePlantUMLSource",qPrint(generateType));
88
89 return baseName;
90}

92void PlantumlManager::generatePlantUMLOutput(const QCString &baseName,const QCString &outDir,OutputFormat format)
93{
94 QCString imgName = baseName;
95 // The basename contains path, we need to strip the path from the filename in order
96 // to create the image file name which should be included in the index.qhp (Qt help index file).
97 int i;
98 if ((i=imgName.findRev('/'))!=-1) // strip path
99 {
100 imgName=imgName.right(imgName.length()-i-1);
101 }
102 switch (format)
103 {
104 case PUML_BITMAP:
105 imgName+=".png";
106 break;
107 case PUML_EPS:
108 imgName+=".eps";
109 break;
110 case PUML_SVG:
111 imgName+=".svg";
112 break;
113 }
114
115 Doxygen::indexList->addImageFile(imgName);
116}

131static void runPlantumlContent(const PlantumlManager::FilesMap &plantumlFiles,
132 const PlantumlManager::ContentMap &plantumlContent,
133 PlantumlManager::OutputFormat format)
134{
135 /* example : running: java -Djava.awt.headless=true
136 -jar "/usr/local/bin/plantuml.jar"
137 -o "test_doxygen/DOXYGEN_OUTPUT/html"
138 -tpng
139 "test_doxygen/DOXYGEN_OUTPUT/html/A.pu"
140 -charset UTF-8
141 outDir:test_doxygen/DOXYGEN_OUTPUT/html
142 test_doxygen/DOXYGEN_OUTPUT/html/A
143 */
144 int exitCode;
145 QCString plantumlJarPath = Config_getString(PLANTUML_JAR_PATH);
146 QCString plantumlConfigFile = Config_getString(PLANTUML_CFG_FILE);
147
148 QCString pumlExe = "java";
149 QCString pumlArgs = "";
150 QCString pumlType = "";
151 QCString pumlOutDir = "";
152
153 const StringVector &pumlIncludePathList = Config_getList(PLANTUML_INCLUDE_PATH);
154 {
155 auto it = pumlIncludePathList.begin();
156 if (it!=pumlIncludePathList.end())
157 {
158 pumlArgs += "-Dplantuml.include.path=\"";
159 pumlArgs += it->c_str();
160 ++it;
161 }
162 while (it!=pumlIncludePathList.end())
163 {
164 pumlArgs += Portable::pathListSeparator();
165 pumlArgs += it->c_str();
166 ++it;
167 }
168 }
169 if (!pumlIncludePathList.empty()) pumlArgs += "\" ";
170 pumlArgs += "-Djava.awt.headless=true -jar \""+plantumlJarPath+"\" ";
171 if (!plantumlConfigFile.isEmpty())
172 {
173 pumlArgs += "-config \"";
174 pumlArgs += plantumlConfigFile;
175 pumlArgs += "\" ";
176 }
177 // the -graphvizdot option expects a relative or absolute path to the dot executable, so
178 // we need to use the unverified DOT_PATH option and check if it points to an existing file.
179 QCString dotPath = Config_getString(DOT_PATH);
180 FileInfo dp(dotPath.str());
181 if (Config_getBool(HAVE_DOT) && dp.exists() && dp.isFile())
182 {
183 pumlArgs += "-graphvizdot \"";
184 pumlArgs += dotPath;
185 pumlArgs += "\" ";
186 }
187 switch (format)
188 {
189 case PlantumlManager::PUML_BITMAP:
190 pumlType="png";
191 break;
192 case PlantumlManager::PUML_EPS:
193 pumlType="eps";
194 break;
195 case PlantumlManager::PUML_SVG:
196 pumlType="svg";
197 break;
198 }
199
200 {
201 for (const auto &kv : plantumlContent)
202 {
203 const PlantumlContent &nb = kv.second;
204 if (nb.content.isEmpty()) continue;
205
206 QCString pumlArguments = pumlArgs;
207 msg("Generating PlantUML %s Files in %s\n",qPrint(pumlType),kv.first.c_str());
208 pumlArguments+="-o \"";
209 pumlArguments+=nb.outDir;
210 pumlArguments+="\" ";
211 pumlArguments+="-charset UTF-8 -t";
212 pumlArguments+=pumlType;
213 pumlArguments+=" ";
214
215 QCString puFileName("");
216 puFileName+=nb.outDir;
217 puFileName+="/";
218 pumlOutDir=puFileName;
219 puFileName+="inline_umlgraph_";
220 puFileName+=pumlType;
221 puFileName+=kv.first.c_str();
222 puFileName+=".pu";
223
224 pumlArguments+="\"";
225 pumlArguments+=puFileName;
226 pumlArguments+="\" ";
227
228
229 QCString cachedContent;
230 FileInfo fi(puFileName.str());
231 if (fi.exists())
232 {
233 cachedContent = fileToString(puFileName);
234 }
235
236 std::ofstream file = Portable::openOutputStream(puFileName);
237 if (!file.is_open())
238 {
239 err_full(nb.srcFile,nb.srcLine,"Could not open file %s for writing\n",puFileName.data());
240 }
241 file.write( nb.content.data(), nb.content.length() );
242 file.close();
243 Debug::print(Debug::Plantuml,0,"*** %s Running Plantuml arguments:%s\n","PlantumlManager::runPlantumlContent",qPrint(pumlArguments));
244
245 if (cachedContent == nb.content) continue;
246
247 if ((exitCode=Portable::system(pumlExe.data(),pumlArguments.data(),TRUE))!=0)
248 {
249 err_full(nb.srcFile,nb.srcLine,"Problems running PlantUML. Verify that the command 'java -jar \"%s\" -h' works from the command line. Exit code: %d\n",
250 plantumlJarPath.data(),exitCode);
251 }
252
253 if ( (format==PlantumlManager::PUML_EPS) && (Config_getBool(USE_PDFLATEX)) )
254 {
255 Debug::print(Debug::Plantuml,0,"*** %s Running epstopdf\n","PlantumlManager::runPlantumlContent");
256 auto files_kv = plantumlFiles.find(kv.first);
257 if (files_kv!=plantumlFiles.end())
258 {
259 for (const auto &str : files_kv->second)
260 {
261 const int maxCmdLine = 40960;
262 QCString epstopdfArgs(maxCmdLine);
263 epstopdfArgs.sprintf("\"%s%s.eps\" --outfile=\"%s%s.pdf\"",
264 pumlOutDir.data(),str.c_str(), pumlOutDir.data(),str.c_str());
265 if ((exitCode=Portable::system("epstopdf",epstopdfArgs.data()))!=0)
266 {
267 err_full(nb.srcFile,nb.srcLine,"Problems running epstopdf. Check your TeX installation! Exit code: %d\n",exitCode);
268 }
269 }
270 }
271 }
272 }
273 }

334void PlantumlManager::insert(const std::string &key, const std::string &value,
335 const QCString &outDir,OutputFormat format,const QCString &puContent,
336 const QCString &srcFile,int srcLine)
337{
338 Debug::print(Debug::Plantuml,0,"*** %s key:%s ,value:%s\n","PlantumlManager::insert",qPrint(key),qPrint(value));
339
340 switch (format)
341 {
342 case PUML_BITMAP:
343 addPlantumlFiles(m_pngPlantumlFiles,key,value);
344 print(m_pngPlantumlFiles);
345 addPlantumlContent(m_pngPlantumlContent,key,outDir,puContent,srcFile,srcLine);
346 print(m_pngPlantumlContent);
347 break;
348 case PUML_EPS:
349 addPlantumlFiles(m_epsPlantumlFiles,key,value);
350 print(m_epsPlantumlFiles);
351 addPlantumlContent(m_epsPlantumlContent,key,outDir,puContent,srcFile,srcLine);
352 print(m_epsPlantumlContent);
353 break;
354 case PUML_SVG:
355 addPlantumlFiles(m_svgPlantumlFiles,key,value);
356 print(m_svgPlantumlFiles);
357 addPlantumlContent(m_svgPlantumlContent,key,outDir,puContent,srcFile,srcLine);
358 print(m_svgPlantumlContent);
359 break;
360 }
361}
