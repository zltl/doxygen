38static QCString makeDisplayName(const NamespaceDef *nd,bool includeScope)
39{
40 QCString result=includeScope ? nd->name() : nd->localName();
41 SrcLangExt lang = nd->getLanguage();
42 QCString sep = getLanguageSpecificSeparator(lang);
43 if (sep!="::")
44 {
45 result = substitute(result,"::",sep);
46 }
47 //printf("makeDisplayName() %s->%s lang=%d\n",qPrint(name()),qPrint(result),lang);
48 return result;
49}

354{
355 //printf("%s:NamespaceDefImpl::addInnerCompound(%s)\n",qPrint(name()),qPrint(d->name()));

405void NamespaceDefImpl::addMembersToMemberGroup()
406{
407 for (auto &ml : m_memberLists)
408 {
409 if (ml->listType()&MemberListType_declarationLists)
410 {
411 ::addMembersToMemberGroup(ml.get(),&m_memberGroups,this);
412 }
413 }
414
415 // add members inside sections to their groups
416 for (const auto &mg : m_memberGroups)
417 {
418 if (mg->allMembersInSameSection() && m_subGrouping)
419 {
420 //printf("----> addToDeclarationSection(%s)\n",qPrint(mg->header()));
421 mg->addToDeclarationSection();
422 }

426void NamespaceDefImpl::insertMember(MemberDef *md)
427{
428 //printf("%s::insertMember(%s) isInline=%d hasDocs=%d\n",qPrint(name()),qPrint(md->name()),
429 // isInline(),hasDocumentation());
430 if (md->isHidden()) return;
431 MemberDefMutable *mdm = toMemberDefMutable(md);
432
433 // if this is an inline namespace that is not documented, then insert the
434 // member in the parent scope instead
435 if (isInline() && !hasDocumentation())
436 {
437 Definition *outerScope = getOuterScope();
438 if (outerScope)
439 {
440 if (outerScope->definitionType()==Definition::TypeNamespace)
441 {
442 NamespaceDefMutable *nd = toNamespaceDefMutable(outerScope);
443 if (nd)
444 {
445 nd->insertMember(md);
446 if (mdm)
447 {
448 mdm->setNamespace(nd);
449 }
450 }
451 }
452 else if (outerScope->definitionType()==Definition::TypeFile)
453 {
454 FileDef *fd = toFileDef(outerScope);
455 fd->insertMember(md);
456 if (mdm)
457 {
458 mdm->setFileDef(fd);
459 mdm->setOuterScope(fd);
460 }
461 }
462 }
463 }
464 else // member is a non-inline namespace or a documented inline namespace
465 {
466 MemberList *allMemberList = getMemberList(MemberListType_allMembersList);
467 if (allMemberList==0)
468 {
469 m_memberLists.emplace_back(std::make_unique<MemberList>(MemberListType_allMembersList,MemberListContainer::Namespace));
470 allMemberList = m_memberLists.back().get();
471 }
472 allMemberList->push_back(md);
473 //printf("%s::m_allMembersDict->append(%s)\n",qPrint(name()),qPrint(md->localName()));
474 m_allMembers.add(md->localName(),md);
475 switch(md->memberType())
476 {
477 case MemberType_Variable:
478 addMemberToList(MemberListType_decVarMembers,md);
479 addMemberToList(MemberListType_docVarMembers,md);
480 break;
481 case MemberType_Function:
482 addMemberToList(MemberListType_decFuncMembers,md);
483 addMemberToList(MemberListType_docFuncMembers,md);
484 break;
485 case MemberType_Typedef:
486 addMemberToList(MemberListType_decTypedefMembers,md);
487 addMemberToList(MemberListType_docTypedefMembers,md);
488 break;
489 case MemberType_Sequence:
490 addMemberToList(MemberListType_decSequenceMembers,md);
491 addMemberToList(MemberListType_docSequenceMembers,md);
492 break;
493 case MemberType_Dictionary:
494 addMemberToList(MemberListType_decDictionaryMembers,md);
495 addMemberToList(MemberListType_docDictionaryMembers,md);
496 break;
497 case MemberType_Enumeration:
498 addMemberToList(MemberListType_decEnumMembers,md);
499 addMemberToList(MemberListType_docEnumMembers,md);
500 break;
501 case MemberType_EnumValue:
502 break;
503 case MemberType_Define:
504 addMemberToList(MemberListType_decDefineMembers,md);
505 addMemberToList(MemberListType_docDefineMembers,md);
506 break;
507 default:
508 err("NamespaceDefImpl::insertMembers(): "
509 "member '%s' with unexpected type id %d and class scope '%s' inserted in namespace scope '%s'!\n",
510 qPrint(md->name()),
511 md->memberType(),
512 md->getClassDef() ? qPrint(md->getClassDef()->name()) : "",
513 qPrint(name()));
514 }
515 // if this is an inline namespace, then insert an alias of this member in the outer scope.
516 if (isInline())
517 {
518 Definition *outerScope = getOuterScope();
519 if (outerScope)
520 {
521 std::unique_ptr<MemberDef> aliasMd;
522 if (outerScope->definitionType()==Definition::TypeNamespace)
523 {
524 aliasMd.reset(createMemberDefAlias(outerScope,md));
525 NamespaceDefMutable *ndm = toNamespaceDefMutable(outerScope);
526 if (ndm)
527 {
528 ndm->insertMember(aliasMd.get());
529 }
530 }
531 else if (outerScope->definitionType()==Definition::TypeFile)
532 {
533 aliasMd.reset(createMemberDefAlias(outerScope,md));
534 toFileDef(outerScope)->insertMember(aliasMd.get());
535 }
536 if (aliasMd)
537 {
538 MemberName *mn;
539 QCString name = md->name();
540 mn = Doxygen::functionNameLinkedMap->add(name);
541 mn->push_back(std::move(aliasMd));
542 }
543 }
544 }
545 }
546}

561void NamespaceDefImpl::writeTagFile(TextStream &tagFile)
562{
563 tagFile << " <compound kind=\"namespace\">\n";
564 tagFile << " <name>" << convertToXML(name()) << "</name>\n";
565 tagFile << " <filename>" << addHtmlExtensionIfMissing(getOutputFileBase()) << "</filename>\n";
566 QCString idStr = id();
567 if (!idStr.isEmpty())
568 {
569 tagFile << " <clangid>" << convertToXML(idStr) << "</clangid>\n";
570 }
571 for (const auto &lde : LayoutDocManager::instance().docEntries(LayoutDocManager::Namespace))
572 {
573 switch (lde->kind())
574 {
575 case LayoutDocEntry::NamespaceNestedNamespaces:
576 {
577 for (const auto &nd : namespaces)
578 {
579 if (nd->isLinkableInProject())
580 {
581 tagFile << " <namespace>" << convertToXML(nd->name()) << "</namespace>\n";
582 }
583 }
584 }
585 break;
586 case LayoutDocEntry::NamespaceClasses:
587 {
588 writeClassesToTagFile(tagFile, classes);
589 }
590 break;
591 case LayoutDocEntry::NamespaceInterfaces:
592 {
593 writeClassesToTagFile(tagFile, interfaces);
594 }
595 break;
596 case LayoutDocEntry::NamespaceStructs:
597 {
598 writeClassesToTagFile(tagFile, structs);
599 }
600 break;
601 case LayoutDocEntry::NamespaceExceptions:
602 {
603 writeClassesToTagFile(tagFile, exceptions);
604 }
605 break;
606 case LayoutDocEntry::NamespaceConcepts:
607 {
608 writeConceptsToTagFile(tagFile);
609 }
610 break;
611 case LayoutDocEntry::MemberDecl:
612 {
613 const LayoutDocEntryMemberDecl *lmd = dynamic_cast<const LayoutDocEntryMemberDecl*>(lde.get());
614 if (lmd)
615 {
616 MemberList * ml = getMemberList(lmd->type);
617 if (ml)
618 {
619 ml->writeTagFile(tagFile);
620 }
621 }
622 }
623 break;
624 case LayoutDocEntry::MemberGroups:
625 {
626 for (const auto &mg : m_memberGroups)
627 {
628 mg->writeTagFile(tagFile);
629 }
630 }
631 break;
632 default:
633 break;
634 }
635 }
636 writeDocAnchorsToTagFile(tagFile);
637 tagFile << " </compound>\n";
638}

640void NamespaceDefImpl::writeDetailedDescription(OutputList &ol,const QCString &title)
641{
642 if (hasDetailedDescription())
643 {
644 ol.pushGeneratorState();
645 ol.disable(OutputType::Html);
646 ol.writeRuler();
647 ol.popGeneratorState();
648 ol.pushGeneratorState();
649 ol.disableAllBut(OutputType::Html);
650 ol.writeAnchor(QCString(),"details");
651 ol.popGeneratorState();
652 ol.startGroupHeader();
653 ol.parseText(title);
654 ol.endGroupHeader();
655
656 ol.startTextBlock();
657 if (!briefDescription().isEmpty() && Config_getBool(REPEAT_BRIEF))
658 {
659 ol.generateDoc(briefFile(),briefLine(),this,0,briefDescription(),FALSE,FALSE,
660 QCString(),FALSE,FALSE,Config_getBool(MARKDOWN_SUPPORT));
661 }
662 if (!briefDescription().isEmpty() && Config_getBool(REPEAT_BRIEF) &&
663 !documentation().isEmpty())
664 {
665 ol.pushGeneratorState();
666 ol.disable(OutputType::Man);
667 ol.disable(OutputType::RTF);
668 //ol.newParagraph(); // FIXME:PARA
669 ol.enableAll();
670 ol.disableAllBut(OutputType::Man);
671 ol.enable(OutputType::Latex);
672 ol.writeString("\n\n");
673 ol.popGeneratorState();
674 }
675 if (!documentation().isEmpty())
676 {
677 ol.generateDoc(docFile(),docLine(),this,0,documentation()+"\n",TRUE,FALSE,
678 QCString(),FALSE,FALSE,Config_getBool(MARKDOWN_SUPPORT));
679 }
680 ol.endTextBlock();
681 }

1470{
1471 int i = name().findRev("::");
1472 if (i==-1) i=0; else i+=2;
1473 bool extractAnonNs = Config_getBool(EXTRACT_ANON_NSPACES);
1474 if (extractAnonNs && // extract anonymous ns
1475 name().mid(i,20)=="anonymous_namespace{" // correct prefix

1479 }
