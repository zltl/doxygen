245 /// returns a copy of this string with leading and trailing whitespace removed
246 QCString stripWhiteSpace() const
247 {
248 size_t sl = m_rep.size();
249 if (sl==0 || (!qisspace(m_rep[0]) && !qisspace(m_rep[sl-1]))) return *this;
250 size_t start=0,end=sl-1;
251 while (start<sl && qisspace(m_rep[start])) start++;
252 if (start==sl) return QCString(); // only whitespace
253 while (end>start && qisspace(m_rep[end])) end--;
254 return QCString(m_rep.substr(start,1+end-start));
255 }
256
257 // Returns a quoted copy of this string, unless it is already quoted.
258 // Note that trailing and leading whitespace is removed.
259 QCString quoted() const
260 {
261 size_t start=0, sl=m_rep.size(), end=sl-1;
262 while (start<sl && qisspace(m_rep[start])) start++; // skip over leading whitespace
263 if (start==sl) return QCString(); // only whitespace
264 while (end>start && qisspace(m_rep[end])) end--; // skip over trailing whitespace
265 bool needsQuotes=false;
266 size_t i=start;
267 if (i<end && m_rep[i]!='"') // stripped string has at least non-whitespace unquoted character
268 {
269 while (i<end && !needsQuotes) // check if the to be quoted part has at least one whitespace character
270 {
271 needsQuotes = m_rep[i] =='-';
272 needsQuotes |= qisspace(m_rep[i++]);
273 }
274 }
275 QCString result(m_rep.substr(start,1+end-start));
276 if (needsQuotes)
277 {
278 result.prepend("\"");
279 result.append("\"");
280 }
281 return result;
282 }
283
284 /// returns a copy of this string with all whitespace removed
285 QCString removeWhiteSpace() const
286 {
287 size_t sl = m_rep.size();
288 if (sl==0) return *this;
289 std::string result = m_rep;
290 size_t src=0,dst=0;
291 while (src<sl)
292 {
293 if (!qisspace(m_rep[src])) result[dst++]=m_rep[src];
294 src++;
295 }
296 if (dst<m_rep.size()) result.resize(dst);
297 return QCString(result);
298 }

304 QCString &insert( size_t index, const QCString &s )
305 {
306 if (s.length()>0)
307 {
308 size_t ol = m_rep.size();
309 if (index>ol) // insert beyond end of string and fill gap with spaces
310 {
311 m_rep.resize(index+s.length());
312 std::memset(&m_rep[ol],' ',index-ol);
313 std::memcpy(&m_rep[index],s.data(),s.length()+1);
314 }
315 else // insert inside the string
316 {
317 m_rep.insert(index,s.str());
318 }
319 }
320 return *this;
321 }

323 {
324 size_t len = s ? qstrlen(s) : 0;
325 if (len>0)

333 }
334 else // insert inside the string
335 {
336 m_rep.insert(index,s);
337 }
