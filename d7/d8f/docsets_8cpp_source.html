48void DocSets::initialize()
49{
50 // -- get config options
51 QCString projectName = Config_getString(PROJECT_NAME);
52 if (projectName.isEmpty()) projectName="root";
53 QCString bundleId = Config_getString(DOCSET_BUNDLE_ID);
54 if (bundleId.isEmpty()) bundleId="org.doxygen.Project";
55 QCString feedName = Config_getString(DOCSET_FEEDNAME);
56 if (feedName.isEmpty()) feedName="FeedName";
57 QCString feedURL = Config_getString(DOCSET_FEEDURL);
58 if (feedURL.isEmpty()) feedURL="FeedUrl";
59 QCString publisherId = Config_getString(DOCSET_PUBLISHER_ID);
60 if (publisherId.isEmpty()) publisherId="PublisherId";
61 QCString publisherName = Config_getString(DOCSET_PUBLISHER_NAME);
62 if (publisherName.isEmpty()) publisherName="PublisherName";
63 QCString projectNumber = Config_getString(PROJECT_NUMBER);
64 if (projectNumber.isEmpty()) projectNumber="ProjectNumber";
65
66 // -- write Makefile
67 {
68 QCString mfName = Config_getString(HTML_OUTPUT) + "/Makefile";
69 std::ofstream ts = Portable::openOutputStream(mfName);
70 if (!ts.is_open())
71 {
72 term("Could not open file %s for writing\n",qPrint(mfName));
73 }
74
75 ts << "DOCSET_NAME=" << bundleId << ".docset\n"
76 "DOCSET_CONTENTS=$(DOCSET_NAME)/Contents\n"
77 "DOCSET_RESOURCES=$(DOCSET_CONTENTS)/Resources\n"
78 "DOCSET_DOCUMENTS=$(DOCSET_RESOURCES)/Documents\n"
79 "DESTDIR=~/Library/Developer/Shared/Documentation/DocSets\n"
80 "XCODE_INSTALL=\"$(shell xcode-select -print-path)\"\n"
81 "\n"
82 "all: docset\n"
83 "\n"
84 "docset:\n"
85 "\tmkdir -p $(DOCSET_DOCUMENTS)\n"
86 "\tcp Nodes.xml $(DOCSET_RESOURCES)\n"
87 "\tcp Tokens.xml $(DOCSET_RESOURCES)\n"
88 "\tcp Info.plist $(DOCSET_CONTENTS)\n"
89 "\ttar --exclude $(DOCSET_NAME) \\\n"
90 "\t --exclude Nodes.xml \\\n"
91 "\t --exclude Tokens.xml \\\n"
92 "\t --exclude Info.plist \\\n"
93 "\t --exclude Makefile -c -f - . \\\n"
94 "\t | (cd $(DOCSET_DOCUMENTS); tar xvf -)\n"
95 "\t$(XCODE_INSTALL)/usr/bin/docsetutil index $(DOCSET_NAME)\n"
96 "\trm -f $(DOCSET_DOCUMENTS)/Nodes.xml\n"
97 "\trm -f $(DOCSET_DOCUMENTS)/Info.plist\n"
98 "\trm -f $(DOCSET_DOCUMENTS)/Makefile\n"
99 "\trm -f $(DOCSET_RESOURCES)/Nodes.xml\n"
100 "\trm -f $(DOCSET_RESOURCES)/Tokens.xml\n"
101 "\n"
102 "clean:\n"
103 "\trm -rf $(DOCSET_NAME)\n"
104 "\n"
105 "install: docset\n"
106 "\tmkdir -p $(DESTDIR)\n"
107 "\tcp -R $(DOCSET_NAME) $(DESTDIR)\n"
108 "\n"
109 "uninstall:\n"
110 "\trm -rf $(DESTDIR)/$(DOCSET_NAME)\n"
111 "\n"
112 "always:\n";
113 }
114

116 {

119 if (!ts.is_open())
120 {
121 term("Could not open file %s for writing\n",qPrint(plName));

124 ts << "<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n"

179 p->tts << "<Tokens version=\"1.0\">\n";

182void DocSets::finalize()
183{
184 if (!p->indentStack.top())
185 {
186 p->nts << p->indent() << " </Node>\n";
187 }
188 p->indentStack.pop();
189 p->nts << " </Subnodes>\n";
190 p->nts << " </Node>\n";
191 p->nts << " </TOC>\n";
192 p->nts << "</DocSetNodes>\n";
193 p->nts.flush();
194 p->ntf.close();
195
196 p->tts << "</Tokens>\n";
197 p->tts.flush();
198 p->ttf.close();
199}

226void DocSets::addContentsItem(bool isDir,
227 const QCString &name,
228 const QCString &ref,
229 const QCString &file,
230 const QCString &anchor,
231 bool /* separateIndex */,
232 bool /* addToNavIndex */,
233 const Definition * /*def*/)
234{
235 (void)isDir;
236 //printf("DocSets::addContentsItem(%s) depth=%zu\n",name,p->indentStack.size());
237 if (ref==0)
238 {
239 if (!p->indentStack.top())
240 {
241 p->nts << p->indent() << " </Node>\n";
242 }
243 p->indentStack.top()=false;
244 p->nts << p->indent() << " <Node>\n";
245 p->nts << p->indent() << " <Name>" << convertToXML(name) << "</Name>\n";
246 if (!file.isEmpty() && file[0]=='^') // URL marker
247 {
248 p->nts << p->indent() << " <URL>" << convertToXML(&file[1])
249 << "</URL>\n";
250 }
251 else // relative file
252 {
253 p->nts << p->indent() << " <Path>";
254 if (!file.isEmpty() && file[0]=='!') // user specified file
255 {
256 p->nts << convertToXML(&file[1]);
257 }
258 else if (!file.isEmpty()) // doxygen generated file
259 {
260 p->nts << addHtmlExtensionIfMissing(file);
261 }
262 p->nts << "</Path>\n";
263 if (!file.isEmpty() && !anchor.isEmpty())
264 {
265 p->nts << p->indent() << " <Anchor>" << anchor << "</Anchor>\n";
266 }
267 }
268 }

271void DocSets::addIndexItem(const Definition *context,const MemberDef *md,
272 const QCString &,const QCString &)
273{
274 if (md==0 && context==0) return;
275
276 const FileDef *fd = 0;
277 const ClassDef *cd = 0;
278 const NamespaceDef *nd = 0;
279
280 if (md)
281 {
282 fd = md->getFileDef();
283 cd = md->getClassDef();
284 nd = md->getNamespaceDef();
285 if (!md->isLinkable()) return; // internal symbol
286 }
287
288 QCString scope;
289 QCString type;
290 QCString decl;
291
292 // determine language
293 QCString lang;
294 SrcLangExt langExt = SrcLangExt_Cpp;
295 if (md)
296 {
297 langExt = md->getLanguage();
298 }
299 else if (context)
300 {
301 langExt = context->getLanguage();
302 }
303 switch (langExt)
304 {
305 case SrcLangExt_Cpp:
306 case SrcLangExt_ObjC:
307 {
308 if (md && (md->isObjCMethod() || md->isObjCProperty()))
309 lang="occ"; // Objective C/C++
310 else if (fd && fd->name().lower().endsWith(".c"))
311 lang="c"; // Plain C
312 else if (cd==0 && nd==0)
313 lang="c"; // Plain C symbol outside any class or namespace
314 else
315 lang="cpp"; // C++
316 }
317 break;
318 case SrcLangExt_IDL: lang="idl"; break; // IDL
319 case SrcLangExt_CSharp: lang="csharp"; break; // C#
320 case SrcLangExt_PHP: lang="php"; break; // PHP4/5
321 case SrcLangExt_D: lang="d"; break; // D
322 case SrcLangExt_Java: lang="java"; break; // Java
323 case SrcLangExt_JS: lang="javascript"; break; // JavaScript
324 case SrcLangExt_Python: lang="python"; break; // Python
325 case SrcLangExt_Fortran: lang="fortran"; break; // Fortran
326 case SrcLangExt_VHDL: lang="vhdl"; break; // VHDL
327 case SrcLangExt_XML: lang="xml"; break; // DBUS XML
328 case SrcLangExt_SQL: lang="sql"; break; // Sql
329 case SrcLangExt_Markdown:lang="markdown"; break; // Markdown
330 case SrcLangExt_Slice: lang="slice"; break; // Slice
331 case SrcLangExt_Lex: lang="lex"; break; // Lex
332 case SrcLangExt_Unknown: lang="unknown"; break; // should not happen!
333 }
334
335 if (md)
336 {
337 if (context==0)
338 {
339 if (md->getGroupDef())
340 context = md->getGroupDef();
341 else if (md->getFileDef())
342 context = md->getFileDef();
343 }
344 if (context==0) return; // should not happen
345
346 switch (md->memberType())
347 {
348 case MemberType_Define:
349 type="macro"; break;
350 case MemberType_Function:
351 if (cd && (cd->compoundType()==ClassDef::Interface ||
352 cd->compoundType()==ClassDef::Class))
353 {
354 if (md->isStatic())
355 type="clm"; // class member
356 else
357 type="instm"; // instance member
358 }
359 else if (cd && cd->compoundType()==ClassDef::Protocol)
360 {
361 if (md->isStatic())
362 type="intfcm"; // interface class member
363 else
364 type="intfm"; // interface member
365 }
366 else
367 type="func";
368 break;
369 case MemberType_Variable:
370 type="data"; break;
371 case MemberType_Typedef:
372 type="tdef"; break;
373 case MemberType_Enumeration:
374 type="enum"; break;
375 case MemberType_EnumValue:
376 type="econst"; break;
377 //case MemberDef::Prototype:
378 // type="prototype"; break;
379 case MemberType_Signal:
380 type="signal"; break;
381 case MemberType_Slot:
382 type="slot"; break;
383 case MemberType_Friend:
384 type="ffunc"; break;
385 case MemberType_DCOP:
386 type="dcop"; break;
387 case MemberType_Property:
388 if (cd && cd->compoundType()==ClassDef::Protocol)
389 type="intfp"; // interface property
390 else
391 type="instp"; // instance property
392 break;
393 case MemberType_Event:
394 type="event"; break;
395 case MemberType_Interface:
396 type="ifc"; break;
397 case MemberType_Service:
398 type="svc"; break;
399 case MemberType_Sequence:
400 type="sequence"; break;
401 case MemberType_Dictionary:
402 type="dictionary"; break;
403 }
404 cd = md->getClassDef();
405 nd = md->getNamespaceDef();
406 if (cd)
407 {
408 scope = cd->qualifiedName();
409 }
410 else if (nd)
411 {
412 scope = nd->name();
413 }
414 fd = md->getFileDef();
415 if (fd)
416 {
417 decl = fd->name();
418 }
419 writeToken(p->tts,md,type,lang,scope,md->anchor(),decl);
420 }
421 else if (context && context->isLinkable())
422 {
423 if (fd==0 && context->definitionType()==Definition::TypeFile)
424 {
425 fd = toFileDef(context);
426 }
427 if (cd==0 && context->definitionType()==Definition::TypeClass)
428 {
429 cd = toClassDef(context);
430 }
431 if (nd==0 && context->definitionType()==Definition::TypeNamespace)
432 {
433 nd = toNamespaceDef(context);
434 }
435 if (fd)
436 {
437 type="file";
438 }
439 else if (cd)
440 {
441 scope = cd->qualifiedName();
442 if (cd->isTemplate())
443 {
444 type="tmplt";
445 }
446 else if (cd->compoundType()==ClassDef::Protocol)
447 {
448 type="intf";
449 if (scope.endsWith("-p")) scope=scope.left(scope.length()-2);
450 }
451 else if (cd->compoundType()==ClassDef::Interface)
452 {
453 type="cl";
454 }
455 else if (cd->compoundType()==ClassDef::Category)
456 {
457 type="cat";
458 }
459 else
460 {
461 type = "cl";
462 }
463 const IncludeInfo *ii = cd->includeInfo();
464 if (ii)
465 {
466 decl=ii->includeName;
467 }
468 }
469 else if (nd)
470 {
471 scope = nd->name();
472 type = "ns";
473 }
474 if (p->scopes.find(context->getOutputFileBase().str())==p->scopes.end())
475 {
476 writeToken(p->tts,context,type,lang,scope,QCString(),decl);
477 p->scopes.insert(context->getOutputFileBase().str());
478 }
479 }
480}

482void DocSets::writeToken(TextStream &t,
483 const Definition *d,
484 const QCString &type,
485 const QCString &lang,
486 const QCString &scope,
487 const QCString &anchor,
488 const QCString &decl)
489{
490 t << " <Token>\n";
491 t << " <TokenIdentifier>\n";
492 QCString name = d->name();
493 if (name.endsWith("-p")) name=name.left(name.length()-2);
494 t << " <Name>" << convertToXML(name) << "</Name>\n";
495 if (!lang.isEmpty())
496 {
497 t << " <APILanguage>" << lang << "</APILanguage>\n";
498 }
499 if (!type.isEmpty())
500 {
501 t << " <Type>" << type << "</Type>\n";
502 }
503 if (!scope.isEmpty())
504 {
505 t << " <Scope>" << convertToXML(scope) << "</Scope>\n";
506 }
507 t << " </TokenIdentifier>\n";
508 t << " <Path>" << addHtmlExtensionIfMissing(d->getOutputFileBase()) << "</Path>\n";
509 if (!anchor.isEmpty())
510 {
511 t << " <Anchor>" << anchor << "</Anchor>\n";
512 }
513 QCString tooltip = d->briefDescriptionAsTooltip();
514 if (!tooltip.isEmpty())
515 {
516 t << " <Abstract>" << convertToXML(tooltip) << "</Abstract>\n";
517 }
518 if (!decl.isEmpty())
519 {
520 t << " <DeclaredIn>" << convertToXML(decl) << "</DeclaredIn>\n";
521 }
522 t << " </Token>\n";
523}
