54void TooltipManager::addTooltip(const Definition *d)
55{
56 bool sourceTooltips = Config_getBool(SOURCE_TOOLTIPS);
57 if (!sourceTooltips) return;
58
59 QCString id = d->getOutputFileBase();
60 int i=id.findRev('/');
61 if (i!=-1)
62 {
63 id = id.right(id.length()-i-1); // strip path (for CREATE_SUBDIRS=YES)
64 }
65 // In case an extension is present translate this extension to something understood by the tooltip handler
66 // otherwise extend t with a translated htmlFileExtension.
67 QCString currentExtension = getFileNameExtension(id);
68 if (currentExtension.isEmpty())
69 {
70 id += escapeId(Doxygen::htmlFileExtension);
71 }
72 else
73 {
74 id = stripExtensionGeneral(id,currentExtension) + escapeId(currentExtension);
75 }
76
77 QCString anc = d->anchor();
78 if (!anc.isEmpty())
79 {
80 id+="_"+anc;
81 }
82 id = "a" + id;
83 p->tooltipInfo.insert(std::make_pair(id.str(),d));
84 //printf("%p: addTooltip(%s)\n",this,id.data());
85}

87void TooltipManager::writeTooltips(CodeOutputInterface &ol)
88{
89 // critical section
90 std::lock_guard<std::mutex> lock(g_tooltipsMutex);
91
92 int id = ol.id();
93 auto it = g_tooltipsWrittenPerFile.find(id);
94 if (it==g_tooltipsWrittenPerFile.end()) // new file
95 {
96 it = g_tooltipsWrittenPerFile.insert(std::make_pair(id,std::set<std::string>())).first;
97 }
98
99 for (const auto &kv : p->tooltipInfo)
100 {
101 bool written = it->second.find(kv.first)!=it->second.end();
102 if (!written) // only write tooltips once
103 {
104 //printf("%p: writeTooltips(%s) ol=%d\n",this,kv.first.c_str(),ol.id());
105 const Definition *d = kv.second;
106 DocLinkInfo docInfo;
107 docInfo.name = d->qualifiedName();
108 docInfo.ref = d->getReference();
109 docInfo.url = d->getOutputFileBase();
110 docInfo.anchor = d->anchor();
111 SourceLinkInfo defInfo;
112 if (d->getBodyDef() && d->getStartBodyLine()!=-1)
113 {
114 defInfo.file = d->getBodyDef()->name();
115 defInfo.line = d->getStartBodyLine();
116 defInfo.url = d->getSourceFileBase();
117 defInfo.anchor = d->getSourceAnchor();
118 }
119 SourceLinkInfo declInfo; // TODO: fill in...
120 QCString decl;
121 if (d->definitionType()==Definition::TypeMember)
122 {
123 const MemberDef *md = toMemberDef(d);
124 if (!md->isAnonymous())
125 {
126 decl = md->declaration();
127 }
128 }
129 ol.writeTooltip(kv.first.c_str(), // id
130 docInfo, // symName
131 decl, // decl
132 d->briefDescriptionAsTooltip(), // desc
133 defInfo,
134 declInfo
135 );
136 it->second.insert(kv.first); // remember we wrote this tooltip for the given file id
137 }
138 }
139}
