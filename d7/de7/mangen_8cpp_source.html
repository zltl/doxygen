35static QCString getExtension()
36{
37 /*
38 * [.][number][rest]
39 * in case of . missing, just ignore it
40 * in case number missing, just place a 3 in front of it
41 */
42 QCString ext = Config_getString(MAN_EXTENSION);
43 if (ext.isEmpty())
44 {
45 ext = "3";
46 }
47 else
48 {
49 if (ext.at(0)=='.')
50 {
51 if (ext.length()==1)
52 {
53 ext = "3";
54 }
55 else // strip .
56 {
57 ext = ext.mid(1);
58 }
59 }
60 if (ext.at(0)<'0' || ext.at(0)>'9')
61 {
62 ext.prepend("3");
63 }
64 }
65 return ext;
66}

78static QCString docifyToString(const QCString &str)
79{
80 QCString result;
81 result.reserve(str.length());
82 if (!str.isEmpty())
83 {
84 const char *p=str.data();
85 char c=0;
86 while ((c=*p++))
87 {
88 switch(c)
89 {
90 case '-': result += "\\-"; break; // see bug747780
91 case '.': result += "\\&."; break; // see bug652277
92 case '\\': result += "\\\\"; break;
93 case '\n': result += "\n"; break;
94 case '\"': c = '\''; // no break!
95 default: result += c; break;
96 }
97 }
98 //printf("%s",str);fflush(stdout);
99 }
100 return result;
101}

148void ManCodeGenerator::codify(const QCString &str)
149{
150 //static char spaces[]=" ";
151 if (!str.isEmpty())
152 {
153 const char *p=str.data();
154 char c;
155 int spacesToNextTabStop;
156 while (*p)
157 {
158 c=*p++;
159 switch(c)
160 {
161 case '-': m_t << "\\-"; break; // see bug747780
162 case '.': m_t << "\\&."; break; // see bug652277
163 case '\t': spacesToNextTabStop =
164 Config_getInt(TAB_SIZE) - (m_col%Config_getInt(TAB_SIZE));
165 m_t << Doxygen::spaces.left(spacesToNextTabStop);
166 m_col+=spacesToNextTabStop;
167 break;
168 case '\n': m_t << "\n"; m_col=0; break;
169 case '\\': m_t << "\\\\"; m_col++; break;
170 case '\"': // no break!
171 default: p=writeUTF8Char(m_t,p-1); m_col++; break;
172 }
173 }
174 //printf("%s",str);fflush(stdout);

232static QCString buildFileName(const QCString &name)
233{
234 QCString fileName;
235 if (name.isEmpty()) return "noname";
236
237 const char *p=name.data();
238 char c;
239 while ((c=*p++))
240 {
241 switch (c)
242 {
243 case ':':
244 fileName+="_";
245 if (*p==':') p++;
246 break;
247 case '<':
248 case '>':
249 case '&':
250 case '*':
251 case '!':
252 case '^':
253 case '~':
254 case '%':
255 case '+':
256 case '/':
257 fileName+="_";
258 break;
259 default:
260 fileName+=c;
261 }
262 }
263
264 QCString manExtension = "." + getExtension();
265 if (fileName.right(manExtension.length())!=manExtension)
266 {
267 fileName+=manExtension;
268 }
269
270 return fileName;
271}

475void ManGenerator::startDoxyAnchor(const QCString &,const QCString &manName,
476 const QCString &, const QCString &name,
477 const QCString &)
478{
479 // something to be done?
480 if( !Config_getBool(MAN_LINKS) )
481 {
482 return; // no
483 }
484
485 // the name of the link file is derived from the name of the anchor:
486 // - truncate after an (optional) ::
487 QCString baseName = name;
488 int i=baseName.findRev("::");
489 if (i!=-1) baseName=baseName.right(baseName.length()-i-2);
490
491 //printf("Converting man link '%s'->'%s'->'%s'\n",
492 // name,qPrint(baseName),qPrint(buildFileName(baseName)));
493
494 // - remove dangerous characters and append suffix, then add dir prefix
495 QCString fileName=dir()+"/"+buildFileName( baseName );
496 FileInfo fi(fileName.str());
497 if (!fi.exists())
498 {
499 std::ofstream linkStream = Portable::openOutputStream(fileName);
500 if (linkStream.is_open())
501 {
502 linkStream << ".so " << getSubdir() << "/" << buildFileName( manName ) << "\n";
503 }
504 }
505}
