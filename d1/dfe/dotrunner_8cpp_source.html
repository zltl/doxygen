51// since dot silently reproduces the input file when it does not
52// support the PNG format, we need to check the result.
53static void checkPngResult(const QCString &imgName)
54{
55 FILE *f = Portable::fopen(imgName,"rb");
56 if (f)
57 {
58 char data[4];
59 if (fread(data,1,4,f)==4)
60 {
61 if (!(data[1]=='P' && data[2]=='N' && data[3]=='G'))
62 {
63 err("Image '%s' produced by dot is not a valid PNG!\n"
64 "You should either select a different format "
65 "(DOT_IMAGE_FORMAT in the config file) or install a more "
66 "recent version of graphviz (1.7+)\n",qPrint(imgName)
67 );
68 }
69 }
70 else
71 {
72 err("Could not read image '%s' generated by dot!\n",qPrint(imgName));
73 }
74 fclose(f);
75 }
76 else
77 {
78 err("Could not open image '%s' generated by dot!\n",qPrint(imgName));
79 }
80}

82static bool resetPDFSize(const int width,const int height, const QCString &base)
83{
84 QCString tmpName = base+".tmp";
85 QCString patchFile = base+".dot";
86 Dir thisDir;
87 if (!thisDir.rename(patchFile.str(),tmpName.str()))
88 {
89 err("Failed to rename file %s to %s!\n",qPrint(patchFile),qPrint(tmpName));
90 return FALSE;
91 }
92 std::ifstream fi = Portable::openInputStream(tmpName);
93 std::ofstream t = Portable::openOutputStream(patchFile);
94 if (!fi.is_open())
95 {
96 err("problem opening file %s for patching!\n",qPrint(tmpName));
97 thisDir.rename(tmpName.str(),patchFile.str());
98 return FALSE;
99 }
100 if (!t.is_open())
101 {
102 err("problem opening file %s for patching!\n",qPrint(patchFile));
103 thisDir.rename(tmpName.str(),patchFile.str());
104 return FALSE;
105 }
106 std::string line;
107 while (getline(fi,line)) // foreach line
108 {
109 if (line.find("LATEX_PDF_SIZE") != std::string::npos)
110 {
111 double scale = (width > height ? width : height)/double(MAX_LATEX_GRAPH_INCH);
112 t << " size=\""<<width/scale << "," <<height/scale << "\";\n";
113 }
114 else
115 t << line << "\n";
116 }
117 fi.close();
118 t.close();
119 // remove temporary file
120 thisDir.remove(tmpName.str());
121 return TRUE;
122}

124bool DotRunner::readBoundingBox(const QCString &fileName,int *width,int *height,bool isEps)
125{
126 std::ifstream f = Portable::openInputStream(fileName);
127 if (!f.is_open())
128 {
129 err("Failed to open file %s for extracting bounding box\n",qPrint(fileName));
130 return false;
131 }
132
133 // read file contents into string 'contents'
134 std::stringstream buffer;
135 buffer << f.rdbuf();
136 std::string contents = buffer.str();
137
138 // start of bounding box marker we are looking for
139 const std::string boundingBox = isEps ? "%%PageBoundingBox:" : "/MediaBox [";
140
141 // helper routine to extract the bounding boxes width and height
142 auto extractBoundingBox = [&fileName,&boundingBox,&width,&height](const char *s) -> bool
143 {
144 int x,y;
145 double w,h;
146 if (sscanf(s+boundingBox.length(),"%d %d %lf %lf",&x,&y,&w,&h)==4)
147 {
148 *width = static_cast<int>(std::ceil(w));
149 *height = static_cast<int>(std::ceil(h));
150 return true;
151 }
152 err("Failed to extract bounding box from generated diagram file %s\n",qPrint(fileName));
153 return false;
154 };
155
156 // compressed segment start and end markers
157 const std::string streamStart = "stream\n";
158 const std::string streamEnd = "\nendstream";
159
160 auto detectDeflateStreamStart = [&streamStart](const char *s)
161 {
162 size_t len = streamStart.length();
163 bool streamOK = strncmp(s,streamStart.c_str(),len)==0;
164 if (streamOK) // ASCII marker matches, check stream header bytes as well
165 {
166 unsigned short header1 = static_cast<unsigned char>(s[len])<<8; // CMF byte
167 if (header1) // not end of string
168 {
169 unsigned short header = (static_cast<unsigned char>(s[len+1])) | header1; // FLG byte
170 // check for correct header (see https://www.rfc-editor.org/rfc/rfc1950)
171 return ((header&0x8F20)==0x0800) && (header%31)==0;
172 }
173 }
174 return false;
175 };
176
177 const size_t l = contents.length();
178 size_t i=0;
179 while (i<l)
180 {
181 if (!isEps && contents[i]=='s' && detectDeflateStreamStart(&contents[i]))
182 { // compressed stream start
183 int col=17;
184 i+=streamStart.length();
185 const size_t start=i;
186 DBG(("---- start stream at offset %08x\n",(int)i));
187 while (i<l)
188 {
189 if (contents[i]=='\n' && strncmp(&contents[i],streamEnd.c_str(),streamEnd.length())==0)
190 { // compressed block found in range [start..i]
191 DBG(("\n---- end stream at offset %08x\n",(int)i));
192 // decompress it into decompressBuf
193 std::vector<char> decompressBuf;
194 const char *source = &contents[start];
195 const size_t sourceLen = i-start;
196 size_t sourcePos = 0;
197 decompressBuf.reserve(sourceLen*2);
198 auto getter = [source,&sourcePos,sourceLen]() -> int {
199 return sourcePos<sourceLen ? static_cast<unsigned char>(source[sourcePos++]) : EOF;
200 };
201 auto putter = [&decompressBuf](const char c) -> int {
202 decompressBuf.push_back(c); return c;
203 };
204 Deflate(getter,putter);
205 // convert decompression buffer to string
206 std::string s(decompressBuf.begin(), decompressBuf.end());
207 DBG(("decompressed_data=[[[\n%s\n]]]\n",s.c_str()));
208 // search for bounding box marker
209 const size_t idx = s.find(boundingBox);
210 if (idx!=std::string::npos) // found bounding box in uncompressed data
211 {
212 return extractBoundingBox(s.c_str()+idx);
213 }
214 // continue searching after end stream marker
215 i+=streamEnd.length();
216 break;
217 }
218 else // compressed stream character
219 {
220 if (col>16) { col=0; DBG(("\n%08x: ",static_cast<int>(i))); }
221 DBG(("%02x ",static_cast<unsigned char>(contents[i])));
222 col++;
223 i++;
224 }
225 }
226 }
227 else if (((isEps && contents[i]=='%') || (!isEps && contents[i]=='/')) &&
228 strncmp(&contents[i],boundingBox.c_str(),boundingBox.length())==0)
229 { // uncompressed bounding box
230 return extractBoundingBox(&contents[i]);
231 }
232 else // uncompressed stream character
233 {
234 i++;
235 }
236 }
237 err("Failed to find bounding box in generated diagram file %s\n",qPrint(fileName));
238 // nothing found
239 return false;
240}

253void DotRunner::addJob(const QCString &format, const QCString &output,
254 const QCString &srcFile,int srcLine)
255{
256
257 for (auto& s: m_jobs)
258 {
259 if (s.format != format) continue;
260 if (s.output != output) continue;
261 // we have this job already
262 return;
263 }
264 auto args = QCString("-T") + format + " -o \"" + output + "\"";
265 m_jobs.emplace_back(format, output, args, srcFile, srcLine);
266}

275bool DotRunner::run()
276{
277 int exitCode=0;
278
279 QCString dotArgs;
280
281 QCString srcFile;
282 int srcLine=-1;
283
284 // create output
285 if (Config_getBool(DOT_MULTI_TARGETS))
286 {
287 dotArgs=QCString("\"")+m_file+"\"";
288 for (auto& s: m_jobs)
289 {
290 dotArgs+=' ';
291 dotArgs+=s.args;
292 }
293 if (!m_jobs.empty())
294 {
295 srcFile = m_jobs.front().srcFile;
296 srcLine = m_jobs.front().srcLine;
297 }
298 if ((exitCode=Portable::system(m_dotExe,dotArgs,FALSE))!=0) goto error;
299 }
300 else
301 {
302 for (auto& s : m_jobs)
303 {
304 srcFile = s.srcFile;
305 srcLine = s.srcLine;
306 dotArgs=QCString("\"")+m_file+"\" "+s.args;
307 if ((exitCode=Portable::system(m_dotExe,dotArgs,FALSE))!=0) goto error;
308 }
309 }
310
311 // check output
312 // As there should be only one pdf file be generated, we don't need code for regenerating multiple pdf files in one call
313 for (auto& s : m_jobs)
314 {
315 if (s.format.startsWith("pdf"))
316 {
317 int width=0,height=0;
318 if (!readBoundingBox(s.output,&width,&height,FALSE)) goto error;
319 if ((width > MAX_LATEX_GRAPH_SIZE) || (height > MAX_LATEX_GRAPH_SIZE))
320 {
321 if (!resetPDFSize(width,height,getBaseNameOfOutput(s.output))) goto error;
322 dotArgs=QCString("\"")+m_file+"\" "+s.args;
323 if ((exitCode=Portable::system(m_dotExe,dotArgs,FALSE))!=0) goto error;
324 }
325 }
326
327 if (s.format.startsWith("png"))
328 {
329 checkPngResult(s.output);
330 }
331 }
332
333 // remove .dot files
334 if (m_cleanUp)
335 {
336 //printf("removing dot file %s\n",qPrint(m_file));
337 Portable::unlink(m_file);
338 }
339
340 // create checksum file
341 if (!m_md5Hash.isEmpty())
342 {
343 QCString md5Name = getBaseNameOfOutput(m_file) + ".md5";
344 FILE *f = Portable::fopen(md5Name,"w");
345 if (f)
346 {
347 fwrite(m_md5Hash.data(),1,32,f);
348 fclose(f);
349 }
350 }
351 return TRUE;
352error:
353 err_full(srcFile,srcLine,"Problems running dot: exit code=%d, command='%s', arguments='%s'\n",
354 exitCode,qPrint(m_dotExe),qPrint(dotArgs));
355 return FALSE;
356}

361void DotRunnerQueue::enqueue(DotRunner *runner)
362{
363 std::lock_guard<std::mutex> locker(m_mutex);
364 m_queue.push(runner);
365 m_bufferNotEmpty.notify_all();
366}

368DotRunner *DotRunnerQueue::dequeue()
369{
370 std::unique_lock<std::mutex> locker(m_mutex);
371
372 // wait until something is added to the queue
373 m_bufferNotEmpty.wait(locker, [this]() { return !m_queue.empty(); });
374
375 DotRunner *result = m_queue.front();
376 m_queue.pop();
377 return result;
378}
