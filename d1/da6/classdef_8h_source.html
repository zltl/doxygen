50/** Class that contains information about an inheritance relation.
51 */
52struct BaseClassDef
53{
54 BaseClassDef(ClassDef *cd,const QCString &n,Protection p, Specifier v,const QCString &t) :
55 classDef(cd), usedName(n), prot(p), virt(v), templSpecifiers(t) {}
56
57 /** Class definition that this relation inherits from. */
58 ClassDef *classDef;
59
60 /** name used in the inheritance list
61 * (may be a typedef name instead of the class name)
62 */
63 QCString usedName;
64
65 /** Protection level of the inheritance relation:
66 * Public, Protected, or Private
67 */
68 Protection prot;
69
70 /** Virtualness of the inheritance relation:
71 * Normal, or Virtual
72 */
73 Specifier virt;
74
75 /** Template arguments used for the base class */
76 QCString templSpecifiers;
77};

493/** Class that contains information about a usage relation.
494 */
495struct UsesClassDef
496{
497 UsesClassDef(ClassDef *cd) : classDef(cd)
498 {
499 }
500 ~UsesClassDef()
501 {
502 }
503 void addAccessor(const QCString &s)
504 {
505 if (accessors.find(s.str())==accessors.end())
506 {
507 accessors.insert(s.str());
508 }
509 }
510 /** Class definition that this relation uses. */
511 ClassDef *classDef;
512
513 /** Dictionary of member variable names that form the edge labels of the
514 * usage relation.
515 */
516 StringSet accessors;
517
518 /** Template arguments used for the base class */
519 QCString templSpecifiers;
520
521 bool containment = true;
522};
