88void VHDLOutlineParser::Private::parseVhdlfile(const QCString &fileName,
89 const char* inputBuffer,bool inLine)
90{
91 JAVACC_STRING_TYPE s =inputBuffer;
92 CharStream *stream = new CharStream(s.c_str(), (int)s.size(), 1, 1);
93 VhdlParserTokenManager *tokenManager = new VhdlParserTokenManager(stream);
94 VhdlTokenManagerErrorHandler *tokErrHandler=new VhdlTokenManagerErrorHandler(fileName.data());
95 vhdlParser=new VhdlParser(tokenManager);
96 vhdlParser->setOutlineParser(thisParser);
97 vhdlParser->setSharedState(&shared);
98 tokenManager->setLexParser(vhdlParser);
99 tokenManager->ReInit(stream,0);
100 tokenManager->setErrorHandler(tokErrHandler);
101 VhdlErrorHandler *parserErrHandler=new VhdlErrorHandler(fileName.data());
102 vhdlParser->setErrorHandler(parserErrHandler);
103 try
104 {
105 if(inLine)
106 {
107 vhdlParser->parseInline();
108 }
109 else
110 {
111 vhdlParser->design_file();
112 }
113 }
114 catch( std::exception &){ /* fprintf(stderr,"\n[%s]",e.what()); */ }
115 // fprintf(stderr,"\n\nparsed lines: %d\n",yyLineNr);
116 // fprintf(stderr,"\n\nerrors : %d\n\n",myErr->getErrorCount());
117 //
118 delete vhdlParser;
119 delete tokenManager;
120 delete stream;
121}

131void VHDLOutlineParser::parseInput(const QCString &fileName,const char *fileBuf,
132 const std::shared_ptr<Entry> &root, ClangTUParser *)
133{
134 VhdlParser::SharedState *s = &p->shared;
135 p->thisParser=this;
136 p->inputString=fileBuf;
137
138 // fprintf(stderr,"\n ============= %s\n ==========\n",fileBuf);
139
140 bool inLine = fileName.isEmpty();
141
142 p->yyFileName=fileName;
143
144 bool xilinx_ucf=isConstraintFile(p->yyFileName,".ucf");
145 bool altera_qsf=isConstraintFile(p->yyFileName,".qsf");
146
147 // support XILINX(ucf) and ALTERA (qsf) file
148
149 if (xilinx_ucf)
150 {
151 VhdlDocGen::parseUCF(fileBuf,root.get(),p->yyFileName,FALSE);
152 return;
153 }
154 if (altera_qsf)
155 {
156 VhdlDocGen::parseUCF(fileBuf,root.get(),p->yyFileName,TRUE);
157 return;
158 }
159 p->yyLineNr=1;
160 s->current_root=root;
161 s->lastCompound=0;
162 s->lastEntity=0;
163 s->lastEntity=0;
164 p->oldEntry = 0;
165 s->current=std::make_shared<Entry>();
166 initEntry(s->current.get());
167 p->commentScanner.enterFile(fileName,p->yyLineNr);
168 p->lineParse.reserve(200);
169 p->parseVhdlfile(fileName,fileBuf,inLine);
170 p->commentScanner.leaveFile(fileName,p->yyLineNr);
171
172 s->current.reset();
173
174 if (!inLine)
175 mapLibPackage(root.get());
176
177 p->yyFileName.resize(0);
178 p->libUse.clear();
179}

272int VHDLOutlineParser::checkInlineCode(QCString &doc)
273{
274 static const reg::Ex csRe(R"([\\@]code)");
275 static const reg::Ex cendRe(R"(\s*[\\@]endcode)");
276 static const reg::Ex cbriefRe(R"([\\@]brief)");
277
278 // helper to simulate behavior of QString.find(const QRegExp &re,int pos)
279 auto findRe = [](const QCString &str,const reg::Ex &re,int pos=0) -> int
280 {
281 if ((int)str.length()<pos) return -1;
282 reg::Match match;
283 const std::string s = str.str();
284 if (reg::search(s,match,re,pos)) // match found
285 {
286 return (int)match.position();
287 }
288 else // not found
289 {
290 return -1;
291 }
292 };
293 auto replaceRe = [](const QCString &str,const reg::Ex &re,const QCString &replacement) -> QCString
294 {
295 return reg::replace(str.str(), re, replacement.str());
296 };
297
298 int index = findRe(doc,csRe);
299
300 if (findRe(doc,cendRe)!=-1)
301 return 1;
302
303 if (index < 0)
304 return index;
305
306 VhdlParser::SharedState *s = &p->shared;
307 p->strComment += doc;
308 p->code = findRe(p->inputString,csRe, p->code + 1);
309 int com = p->inputString.find(p->strComment.data());
310 int ref = findRe(p->inputString,cendRe, p->code + 1);
311 int len = p->strComment.size();
312
313 int ll = com + len;
314 int diff = ref - ll - 3;
315 QCString code = p->inputString.mid(ll, diff);
316 int iLine = 0;
317 code = stripLeadingAndTrailingEmptyLines(code, iLine);
318 int val = code.contains('\n');
319 VhdlDocGen::prepareComment(p->strComment);
320 StringVector ql = split(p->strComment.str(),"\n");
321
322 QCString co;
323 QCString na;
324 for (const auto &qcs_ : ql)
325 {
326 QCString qcs(qcs_);
327 qcs = qcs.simplifyWhiteSpace();
328 if (findRe(qcs,csRe)!=-1)
329 {
330 int i = qcs.find('{');
331 int j = qcs.find('}');
332 if (i > 0 && j > 0 && j > i)
333 {
334 na = qcs.mid(i + 1, (j - i - 1));
335 }
336 continue;
337 }
338 qcs = replaceRe(qcs,cbriefRe, "");
339 co += qcs;
340 co += '\n';
341 }
342
343 VhdlDocGen::prepareComment(co);
344
345 Entry gBlock;
346 if (!na.isEmpty())
347 gBlock.name = na;
348 else
349 gBlock.name = "misc" + VhdlDocGen::getRecordNumber();
350 gBlock.startLine = p->yyLineNr+iLine-1;
351 gBlock.bodyLine = p->yyLineNr+iLine-1 ;
352 gBlock.doc = code;
353 gBlock.inbodyDocs = code;
354 gBlock.brief = co;
355 gBlock.section = Entry::VARIABLE_SEC;
356 gBlock.spec = VhdlDocGen::MISCELLANEOUS;
357 gBlock.fileName = p->yyFileName;
358 gBlock.endBodyLine = p->yyLineNr + val +iLine;
359 gBlock.lang = SrcLangExt_VHDL;
360 std::shared_ptr<Entry> compound;
361
362 if (s->lastEntity)
363 compound = s->lastEntity;
364 else if (s->lastCompound)
365 compound = s->lastCompound;
366 else
367 compound = 0;
368
369 if (compound)
370 {
371 compound->copyToSubEntry(&gBlock);
372 }
373 else
374 {
375 gBlock.type = "misc"; // global code like library ieee...
376 s->current_root->copyToSubEntry(&gBlock);
377 }
378 p->strComment.resize(0);
379 return 1;
380}

382void VHDLOutlineParser::handleCommentBlock(const QCString &doc1, bool brief)
383{
384 int position = 0;
385 bool needsEntry = FALSE;
386 VhdlParser::SharedState *s = &p->shared;
387 QCString doc = doc1;
388
389 if (doc.isEmpty())
390 return;
391
392 if (checkMultiComment(doc, p->yyLineNr))
393 {
394 p->strComment.resize(0);
395 return;
396 }
397
398 if (checkInlineCode(doc) > 0)
399 {
400 return;
401 }
402
403 Protection protection = Protection::Public;
404 VhdlDocGen::prepareComment(doc);
405
406 if (p->oldEntry == s->current.get())
407 {
408 p->str_doc.doc = doc;
409 p->str_doc.iDocLine = p->iDocLine;
410 p->str_doc.brief = brief;
411 p->str_doc.pending = TRUE;
412 return;
413 }
414
415 p->oldEntry = s->current.get();
416
417 if (brief)
418 {
419 s->current->briefLine = p->yyLineNr;
420 }
421 else
422 {
423 s->current->docLine = p->yyLineNr;
424 }
425
426
427
428 Markdown markdown(p->yyFileName,p->iDocLine);
429 int lineNr = p->iDocLine;
430 QCString processedDoc = Config_getBool(MARKDOWN_SUPPORT) ? markdown.process(doc,lineNr) : doc;
431
432 while (p->commentScanner.parseCommentBlock(
433 p->thisParser,
434 s->current.get(),
435 processedDoc, // text
436 p->yyFileName, // file
437 lineNr, // line of block start
438 brief,
439 0,
440 FALSE,
441 protection,
442 position,
443 needsEntry,
444 Config_getBool(MARKDOWN_SUPPORT)))
445 {
446 if (needsEntry)
447 newEntry();
448 }
449 if (needsEntry)
450 {
451 if (p->varr)
452 {
453 p->varr = FALSE;
454 s->current->name = p->varName;
455 s->current->section = Entry::VARIABLEDOC_SEC;
456 p->varName = "";
457 }
458 newEntry();
459 }
460 p->iDocLine = -1;
461 p->strComment.resize(0);
462}

470void VHDLOutlineParser::addCompInst(const char *n, const char* instName, const char* comp,int iLine)
471{
472 VhdlParser::SharedState *s = &p->shared;
473 s->current->spec=VhdlDocGen::INSTANTIATION;
474 s->current->section=Entry::VARIABLE_SEC;
475 s->current->startLine=iLine;
476 s->current->bodyLine=iLine;
477 s->current->type=instName; // foo:instname e.g proto or work. proto(ttt)
478 s->current->exception=s->genLabels.lower(); // |arch|label1:label2...
479 s->current->name=n; // foo
480 if (s->lastCompound)
481 {
482 s->current->args=s->lastCompound->name; // architecture name
483 }
484 s->current->includeName=comp; // component/entity/configuration
485 int u=s->genLabels.find("|",1);
486 if (u>0)
487 {
488 s->current->write=s->genLabels.right(s->genLabels.length()-u);
489 s->current->read=s->genLabels.left(u);
490 }
491 //printf (" \n genlabel: [%s] inst: [%s] name: [%s] %d\n",n,instName,comp,iLine);
492
493 if (s->lastCompound)
494 {
495 s->current->args=s->lastCompound->name;
496 if (true) // !findInstant(current->type))
497 {
498 initEntry(s->current.get());
499 // TODO: protect with mutex
500 g_instFiles.emplace_back(std::make_shared<Entry>(*s->current));
501 // TODO: end protect with mutex
502 }
503
504 s->current=std::make_shared<Entry>();
505 }
506 else
507 {
508 newEntry();
509 }

512void VHDLOutlineParser::addVhdlType(const char *n,int startLine,int section,
513 uint64 spec,const char* args,const char* type,Protection prot)
514{
515 VhdlParser::SharedState *s = &p->shared;
516 QCString name(n);
517 if (isFuncProcProced() || VhdlDocGen::getFlowMember()) return;
518
519 if (s->parse_sec==GEN_SEC)
520 {
521 spec= VhdlDocGen::GENERIC;
522 }
523
524 StringVector ql=split(name.str(),",");
525
526 for (size_t u=0;u<ql.size();u++)
527 {
528 s->current->name=ql[u];
529 s->current->startLine=startLine;
530 s->current->bodyLine=startLine;
531 s->current->section=section;
532 s->current->spec=spec;
533 s->current->fileName=p->yyFileName;
534 if (s->current->args.isEmpty())
535 {
536 s->current->args=args;
537 }
538 s->current->type=type;
539 s->current->protection=prot;
540
541 if (!s->lastCompound && (section==Entry::VARIABLE_SEC) && (spec == VhdlDocGen::USE || spec == VhdlDocGen::LIBRARY) )
542 {
543 p->libUse.emplace_back(std::make_shared<Entry>(*s->current));
544 s->current->reset();
545 }
546 newEntry();
547 }

550void VHDLOutlineParser::createFunction(const char *imp,uint64 spec,const char *fn)
551{
552 VhdlParser::SharedState *s = &p->shared;
553 QCString impure(imp);
554 QCString fname(fn);
555 s->current->spec=spec;
556 s->current->section=Entry::FUNCTION_SEC;
557
558 if (impure=="impure" || impure=="pure")
559 {
560 s->current->exception=impure;
561 }
562
563 if (s->parse_sec==GEN_SEC)
564 {
565 s->current->spec= VhdlDocGen::GENERIC;
566 s->current->section=Entry::FUNCTION_SEC;
567 }
568
569 if (s->currP==VhdlDocGen::PROCEDURE)
570 {
571 s->current->name=impure;
572 s->current->exception="";
573 }
574 else
575 {
576 s->current->name=fname;
577 }
578
579 if (spec==VhdlDocGen::PROCESS)
580 {
581 s->current->args=fname;
582 s->current->name=impure;
583 VhdlDocGen::deleteAllChars(s->current->args,' ');
584 if (!fname.isEmpty())
585 {
586 StringVector q1=split(fname.str(),",");
587 for (const auto &name : q1)
588 {
589 Argument arg;
590 arg.name=name;
591 s->current->argList.push_back(arg);
592 }
593 }
594 }
595}

672 * entity xxx

674 * library
675 * package

680void VHDLOutlineParser::mapLibPackage( Entry* root)
681{
682 for (const auto &rt : p->libUse)
683 {
684 if (addLibUseClause(rt->name))
685 {
686 bool bFound=FALSE;
687 for (const auto &current : root->children())
688 {
689 if (VhdlDocGen::isVhdlClass(current.get()))
690 {
691 if (current->startLine > rt->startLine)
692 {
693 bFound=TRUE;
694 current->copyToSubEntry(rt);
695 break;
696 }
697 }
698 }//for
699 if (!bFound)
700 {
701 root->copyToSubEntry(rt);
702 }
703 } //if
704 }// for
705}//MapLib

712 {
713 if (type.lower().stripPrefix("ieee")) return FALSE;
714 if (type.lower().stripPrefix("std")) return FALSE;
715 }
716 return TRUE;

739void VHDLOutlineParser::createFlow()
740{
741 VhdlParser::SharedState *s = &p->shared;
742 if (!VhdlDocGen::getFlowMember())
743 {
744 return;
745 }
746 QCString q,ret;
747
748 if (s->currP==VhdlDocGen::FUNCTION)
749 {
750 q=":function( ";
751 FlowChart::alignFuncProc(q,s->tempEntry->argList,true);
752 q+=")";
753 }
754 else if (s->currP==VhdlDocGen::PROCEDURE)
755 {
756 q=":procedure (";
757 FlowChart::alignFuncProc(q,s->tempEntry->argList,false);
758 q+=")";
759 }
760 else
761 {
762 q=":process( "+s->tempEntry->args;
763 q+=")";
764 }
765
766 q.prepend(VhdlDocGen::getFlowMember()->name());
767
768 FlowChart::addFlowChart(FlowChart::START_NO,q,QCString());
769
770 if (s->currP==VhdlDocGen::FUNCTION)
771 {
772 ret="end function ";
773 }
774 else if (s->currP==VhdlDocGen::PROCEDURE)
775 {
776 ret="end procedure";
777 }
778 else
779 {
780 ret="end process ";
781 }
782
783 FlowChart::addFlowChart(FlowChart::END_NO,ret,QCString());
784 // FlowChart::printFlowList();
785 FlowChart::writeFlowChart();
786 s->currP=0;
787}

805bool VHDLOutlineParser::checkMultiComment(QCString& qcs,int line)
806{
807 VhdlParser::SharedState *s = &p->shared;
808 insertEntryAtLine(s->current_root,line);
809
810 if (p->lineEntry.empty()) return false;
811
812 VhdlDocGen::prepareComment(qcs);
813 while (!p->lineEntry.empty())
814 {
815 std::shared_ptr<Entry> e=p->lineEntry.back();
816 e->briefLine=line;
817 e->brief+=qcs;
818
819 p->lineEntry.pop_back();
820 }
821 return true;
822}

861QCString filter2008VhdlComment(const char *s)
862{
863 GrowBuf growBuf;
864 const char *p=s+3; // skip /*!
865 char c='\0';
866 while (*p == ' ' || *p == '\t') p++;
867 while ((c=*p++))
868 {
869 growBuf.addChar(c);
870 if (c == '\n')
871 {
872 // special handling of space followed by * at beginning of line
873 while (*p == ' ' || *p == '\t') p++;
874 while (*p == '*') p++;
875 // special attention in case character at end is /
876 if (*p == '/') p++;
877 }
878 }
879 // special attention in case */ at end of last line
880 size_t len = growBuf.getPos();
881 if (growBuf.at(len-1) == '/' && growBuf.at(len-2) == '*')
882 {
883 len -= 2;
884 while (growBuf.at(len-1) == '*') len--;
885 c = growBuf.at(len-1);
886 while ((c = growBuf.at(len-1)) == ' ' || c == '\t') len--;
887 growBuf.setPos(len);
888 }
889 growBuf.addChar(0);
890 return growBuf.get();
891}
