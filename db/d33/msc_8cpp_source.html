31static bool convertMapFile(TextStream &t,const QCString &mapName,const QCString &relPath,
32 const QCString &context)
33{
34 std::ifstream f = Portable::openInputStream(mapName);
35 if (!f.is_open())
36 {
37 err("failed to open map file %s for inclusion in the docs!\n"
38 "If you installed Graphviz/dot after a previous failing run, \n"
39 "try deleting the output directory and rerun doxygen.\n",qPrint(mapName));
40 return false;
41 }
42 const int maxLineLen=1024;
43 char url[maxLineLen];
44 char ref[maxLineLen];
45 int x1,y1,x2,y2;
46 std::string line;
47 while (getline(f,line))
48 {
49 bool isRef = false;
50 //printf("ReadLine '%s'\n",line.c_str());
51 if (qstrncmp(line.c_str(),"rect",4)==0)
52 {
53 // obtain the url and the coordinates in the order used by graphviz-1.5
54 sscanf(line.c_str(),"rect %s %d,%d %d,%d",url,&x1,&y1,&x2,&y2);
55
56 if (qstrcmp(url,"\\ref")==0 || qstrcmp(url,"@ref")==0)
57 {
58 isRef = true;
59 sscanf(line.c_str(),"rect %s %s %d,%d %d,%d",ref,url,&x1,&y1,&x2,&y2);
60 }
61
62 // sanity checks
63 if (y2<y1) { int temp=y2; y2=y1; y1=temp; }
64 if (x2<x1) { int temp=x2; x2=x1; x1=temp; }
65
66 t << "<area href=\"";
67
68 if ( isRef )
69 {
70 // handle doxygen \ref tag URL reference
71
72 auto parser { createDocParser() };
73 auto dfAst { createRef( *parser.get(), url, context ) };
74 auto dfAstImpl = dynamic_cast<const DocNodeAST*>(dfAst.get());
75 const DocRef *df = std::get_if<DocRef>(&dfAstImpl->root);
76 t << externalRef(relPath,df->ref(),TRUE);
77 if (!df->file().isEmpty()) t << addHtmlExtensionIfMissing(df->file());
78 if (!df->anchor().isEmpty()) t << "#" << df->anchor();
79 }
80 else
81 {
82 t << url;
83 }
84 t << "\" shape=\"rect\" coords=\""
85 << x1 << "," << y1 << "," << x2 << "," << y2 << "\""
86 << " alt=\"\"/>\n";
87 }
88 }
89
90 return true;
91}

93void writeMscGraphFromFile(const QCString &inFile,const QCString &outDir,
94 const QCString &outFile,MscOutputFormat format,
95 const QCString &srcFile,int srcLine
96 )
97{
98 QCString absOutFile = outDir;
99 absOutFile+=Portable::pathSeparator();
100 absOutFile+=outFile;
101
102 mscgen_format_t msc_format;
103 QCString imgName = absOutFile;
104 switch (format)
105 {
106 case MSC_BITMAP:
107 msc_format = mscgen_format_png;
108 imgName+=".png";
109 break;
110 case MSC_EPS:
111 msc_format = mscgen_format_eps;
112 imgName+=".eps";
113 break;
114 case MSC_SVG:
115 msc_format = mscgen_format_svg;
116 imgName+=".svg";
117 break;
118 default:
119 return;
120 }
121 int code;
122 if ((code=mscgen_generate(inFile.data(),imgName.data(),msc_format))!=0)
123 {
124 err_full(srcFile,srcLine,"Problems generating msc output (error=%s). Look for typos in you msc file %s\n",
125 mscgen_error2str(code),qPrint(inFile));
126 return;
127 }
128
129 if ( (format==MSC_EPS) && (Config_getBool(USE_PDFLATEX)) )
130 {
131 QCString epstopdfArgs(maxCmdLine);
132 epstopdfArgs.sprintf("\"%s.eps\" --outfile=\"%s.pdf\"",
133 qPrint(absOutFile),qPrint(absOutFile));
134 if (Portable::system("epstopdf",epstopdfArgs)!=0)
135 {
136 err_full(srcFile,srcLine,"Problems running epstopdf when processing '%s.eps'. Check your TeX installation!\n",
137 qPrint(absOutFile));
138 }
139 }
140
141 Doxygen::indexList->addImageFile(imgName);
142
143}

145static QCString getMscImageMapFromFile(const QCString& inFile, const QCString& outDir,
146 const QCString& relPath,const QCString& context,
147 bool writeSVGMap,const QCString &srcFile,int srcLine)
148{
149 QCString outFile = inFile + ".map";
150
151 int code;
152 if ((code=mscgen_generate(inFile.data(),outFile.data(),
153 writeSVGMap ? mscgen_format_svgmap : mscgen_format_pngmap))!=0)
154 {
155 err_full(srcFile,srcLine,"Problems generating msc output (error=%s). Look for typos in you msc file %s\n",
156 mscgen_error2str(code),qPrint(inFile));
157 return "";
158 }
159
160 TextStream t;
161 convertMapFile(t, outFile, relPath, context);
162
163 Dir().remove(outFile.str());
164
165 return t.str();
166}

168void writeMscImageMapFromFile(TextStream &t,const QCString &inFile,
169 const QCString &outDir,
170 const QCString &relPath,
171 const QCString &baseName,
172 const QCString &context,
173 MscOutputFormat format,
174 const QCString &srcFile,
175 int srcLine
176 )
177{
178 QCString mapName = baseName+".map";
179 t << "<img src=\"" << relPath << baseName << ".";
180 switch (format)
181 {
182 case MSC_BITMAP:
183 t << "png";
184 break;
185 case MSC_EPS:
186 t << "eps";
187 break;
188 case MSC_SVG:
189 t << "svg";
190 break;
191 default:
192 t << "unknown";
193 }
194 QCString imap = getMscImageMapFromFile(inFile,outDir,relPath,context,format==MSC_SVG,srcFile,srcLine);
195 if (!imap.isEmpty())
196 {
197 t << "\" alt=\""
198 << baseName << "\" border=\"0\" usemap=\"#" << mapName << "\"/>\n";
199 t << "<map name=\"" << mapName << "\" id=\"" << mapName << "\">" << imap << "</map>\n";
200 }
201 else
202 {
203 t << "\" alt=\"" << baseName << "\" border=\"0\"/>\n";
204 }
205}
