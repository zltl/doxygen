79static bool elemIsVisible(const XMLHandlers::Attributes &attrib,bool defVal=TRUE)
80{
81 QCString visible = XMLHandlers::value(attrib,"visible");
82 if (visible.isEmpty()) return defVal;
83 if (visible.at(0)=='$' && visible.length()>1)
84 {
85 QCString id = visible.mid(1);
86 const ConfigValues::Info *opt = ConfigValues::instance().get(id);
87 if (opt && opt->type==ConfigValues::Info::Bool)
88 {
89 return ConfigValues::instance().*(opt->value.b);
90 }
91 else if (opt && opt->type==ConfigValues::Info::String)
92 {
93 return opt->getBooleanRepresentation();
94 }
95 else if (!opt)
96 {
97 err("found unsupported value '%s' for visible attribute in layout file, reverting to '%s'\n",
98 qPrint(visible),(defVal?"yes":"no"));
99 return defVal;
100 }
101 }
102 QCString visibleLow = visible.lower();
103 if (visibleLow=="no" || visibleLow=="false" || visibleLow=="0") return FALSE;
104 else if (visibleLow=="yes" || visibleLow=="true" || visibleLow=="1") return TRUE;
105 else
106 {
107 err("found unsupported value '%s' for visible attribute in layout file, reverting to '%s'\n",
108 qPrint(visible),(defVal?"yes":"no"));
109 return defVal;
110 }
111}

120LayoutNavEntry *LayoutNavEntry::find(LayoutNavEntry::Kind kind,
121 const QCString &file) const
122{
123 LayoutNavEntry *result=0;
124 for (const auto &entry : m_children)
125 {
126 // depth first search, needed to find the entry furthest from the
127 // root in case an entry is in the tree twice
128 result = entry->find(kind,file);
129 if (result) return result;
130 if (entry->kind()==kind && (file==QCString() || entry->baseFile()==file))
131 {
132 return entry.get();
133 }
134 }
135 return result;
136}

138QCString LayoutNavEntry::url() const
139{
140 QCString url = baseFile().stripWhiteSpace();
141 if ((kind()!=LayoutNavEntry::User && kind()!=LayoutNavEntry::UserGroup) ||
142 (kind()==LayoutNavEntry::UserGroup && url.startsWith("usergroup")))
143 {
144 url = addHtmlExtensionIfMissing(url);
145 }
146 else if (url.startsWith("@ref ") || url.startsWith("\\ref "))
147 {
148 const Definition *d = 0;
149 QCString anchor;
150 bool found=FALSE;
151 if (resolveLink(QCString(),url.mid(5).stripWhiteSpace(),TRUE,&d,anchor))
152 {
153 if (d && d->isLinkable())
154 {
155 url=addHtmlExtensionIfMissing(d->getOutputFileBase());
156 if (!anchor.isEmpty())
157 {
158 url+="#"+anchor;
159 }
160 found=TRUE;
161 }
162 }
163 if (!found)
164 {
165 msg("explicit link request to '%s' in layout file '%s' could not be resolved\n",qPrint(url.mid(5)),qPrint(Config_getString(LAYOUT_FILE)));
166 }
167 }
168 //printf("LayoutNavEntry::url()=%s\n",qPrint(url));
169 return url;
170}

1642// Convert input to a title.
1643// The format of input can be a simple title "A title" or in case there are different
1644// titles for some programming languages they can take the following form:
1645// "A title|16=Another title|8=Yet Another title"
1646// where the number is a value of SrcLangExt in decimal notation (i.e. 16=Java, 8=IDL).
1647QCString extractLanguageSpecificTitle(const QCString &input,SrcLangExt lang)
1648{
1649 int i,s=0,e=input.find('|');
1650 if (e==-1) return input; // simple title case
1651 int e1=e;
1652 while (e!=-1) // look for 'number=title' pattern separated by '|'
1653 {
1654 s=e+1;
1655 e=input.find('|',s);
1656 i=input.find('=',s);
1657 assert(i>s);
1658 SrcLangExt key= static_cast<SrcLangExt>(input.mid(s,i-s).toUInt());
1659 if (key==lang) // found matching key
1660 {
1661 if (e==-1) e=input.length();
1662 return input.mid(i+1,e-i-1);
1663 }
1664 }
1665 return input.left(e1); // fallback, no explicit language key found
1666}
