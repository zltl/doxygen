44OutputList::OutputList(const OutputList &ol)
45{
46 m_id = ol.m_id;
47 m_codeGenList.setId(m_id);
48 for (const auto &og : ol.m_outputs)
49 {
50 auto docGen = og->clone();
51 docGen->codeGen()->setId(m_id);
52 m_codeGenList.add(docGen->codeGen());
53 m_outputs.emplace_back(std::move(docGen));
54 }
55}

172void OutputList::generateDoc(const QCString &fileName,int startLine,
173 const Definition *ctx,const MemberDef * md,
174 const QCString &docStr,bool indexWords,
175 bool isExample,const QCString &exampleName,
176 bool singleLine,bool linkFromIndex,
177 bool markdownSupport)
178{
179 int count=0;
180 if (docStr.isEmpty()) return;
181
182 for (const auto &og : m_outputs)
183 {
184 if (og->isEnabled()) count++;
185 }
186
187 if (count>0)
188 {
189 // we want to validate irrespective of the number of output formats
190 // specified as:
191 // - when only XML format there should be warnings as well (XML has its own write routines)
192 // - no formats there should be warnings as well
193 auto parser { createDocParser() };
194 auto ast { validatingParseDoc(*parser.get(),
195 fileName,startLine,
196 ctx,md,docStr,indexWords,isExample,exampleName,
197 singleLine,linkFromIndex,markdownSupport) };
198 if (ast) writeDoc(ast.get(),ctx,md,m_id);
199 }
200}

222void OutputList::parseText(const QCString &textStr)
223{
224 int count=0;
225 for (const auto &og : m_outputs)
226 {
227 if (og->isEnabled()) count++;
228 }
229
230 // we want to validate irrespective of the number of output formats
231 // specified as:
232 // - when only XML format there should be warnings as well (XML has its own write routines)
233 // - no formats there should be warnings as well
234 auto parser { createDocParser() };
235 auto textNode { validatingParseText(*parser.get(), textStr) };
236
237 if (textNode && count>0)
238 {
239 for (const auto &og : m_outputs)
240 {
241 if (og->isEnabled()) og->writeDoc(textNode.get(),0,0,m_id);
242 }
243 }
