408 QCString trWriteList(int numEntries)
409 {
410 // this function is used to produce a comma-separated list of items.
411 // use generateMarker(i) to indicate where item i should be put.
412 QCString result;
413 int i;
414 // the inherits list contain `numEntries' classes
415 for (i=0;i<numEntries;i++)
416 {
417 // use generateMarker to generate placeholders for the class links!
418 result+=generateMarker(i); // generate marker for entry i in the list
419 // (order is left to right)
420
421 if (i!=numEntries-1) // not the last entry, so we need a separator
422 {
423 if (i<numEntries-2) // not the fore last entry
424 result+=", ";
425 else // the fore last entry
426 result+="";
427 }
428 }
429 return result;
430 }
