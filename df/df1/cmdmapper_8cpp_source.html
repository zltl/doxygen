247int Mapper::map(const QCString &n)
248{
249 if (n.isEmpty()) return 0;
250 QCString name = n;
251 if (!m_cs) name=name.lower();
252 auto it = m_map.find(name.str());
253 return it!=m_map.end() ? it->second : 0;
254}
255
256QCString Mapper::find(const int n)
257{
258 for (const auto &kv : m_map)
259 {
260 int curVal = kv.second;
261 if (curVal == n || (curVal == (n | SIMPLESECT_BIT))) return kv.first.c_str();
262 }
263 return QCString();
264}
265
266Mapper::Mapper(const CommandMap *cm,bool caseSensitive) : m_cs(caseSensitive)
267{
268 const CommandMap *p = cm;
269 while (p->cmdName)
270 {
271 m_map.insert(std::make_pair(p->cmdName,p->cmdId));
272 p++;
273 }
274}
275
276void Mappers::freeMappers()
277{
278 delete cmdMapper; cmdMapper = 0;
279 delete htmlTagMapper; htmlTagMapper = 0;
280}
