164inline bool isPreformatted(const DocNodeVariant *n)
165{
166 return std::visit([](auto &&x)->decltype(auto) { return x.isPreformatted(); }, *n);
167}
168
169/*! Returns TRUE iff node n is a child of a preformatted node */
170inline bool insidePRE(const DocNodeVariant *n)
171{
172 while (n)
173 {
174 if (isPreformatted(n)) return TRUE;
175 n=parent(n);
176 }
177 return FALSE;
178}
179
180/*! Returns TRUE iff node n is a child of a html list item node */
181inline bool insideLI(const DocNodeVariant *n)
182{
183 while (n)
184 {
185 if (std::holds_alternative<DocHtmlListItem>(*n)) return TRUE;
186 n=parent(n);
187 }
188 return FALSE;
189}
190
191//---------------------------------------------------------------------------
192
193/*! Returns TRUE iff node n is a child of a unordered html list node */
194inline bool insideUL(const DocNodeVariant *n)
195{
196 while (n)
197 {
198 if (std::holds_alternative<DocHtmlList>(*n) &&
199 std::get<DocHtmlList>(*n).type()==DocHtmlList::Unordered)
200 {
201 return TRUE;
202 }
203 n=parent(n);
204 }
205 return FALSE;
206}
