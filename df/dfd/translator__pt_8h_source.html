685 /*! this function is used to produce a comma-separated list of items.
686 * use generateMarker(i) to indicate where item i should be put.
687 */
688 virtual QCString trWriteList(int numEntries)
689 {
690 QCString result;
691 int i;
692 // the inherits list contain `numEntries' classes
693 for (i=0;i<numEntries;i++)
694 {
695 // use generateMarker to generate placeholders for the class links!
696 result+=generateMarker(i); // generate marker for entry i in the list
697 // (order is left to right)
698
699 if (i!=numEntries-1) // not the last entry, so we need a separator
700 {
701 if (i<numEntries-2) // not the fore last entry
702 result+=", ";
703 else // the fore last entry
704 result+=" e ";
705 }
706 }
707 return result;
708 }

1830 * @param year Year in 4 digits
1831 * @param month Month of the year: 1=January
1832 * @param day Day of the Month: 1..31
1833 * @param dayOfWeek Day of the week: 1=Monday..7=Sunday
1834 * @param hour Hour of the day: 0..23
1835 * @param minutes Minutes in the hour: 0..59
1836 * @param seconds Seconds within the minute: 0..59
1837 * @param includeTime Include time in the result string?
1838 */
1839 virtual QCString trDateTime(int year,int month,int day,int dayOfWeek,
1840 int hour,int minutes,int seconds,
1841 DateTimeType includeTime)
1842 {
1843 static const char *days[] = { "Segunda","Terça","Quarta","Quinta","Sexta","Sábado","Domingo" };
1844 static const char *months[] = { "Janeiro","Fevereiro","Março","Abril","Maio","Junho","Julho","Agosto","Setembro","Outubro","Novembro","Dezembro" };
1845 QCString sdate;
1846 if (includeTime == DateTimeType::DateTime || includeTime == DateTimeType::Date)
1847 {
1848 sdate.sprintf("%s, %d de %s de %d",days[dayOfWeek-1],day,months[month-1],year);
1849 }
1850 if (includeTime == DateTimeType::DateTime) sdate += " ";
1851 if (includeTime == DateTimeType::DateTime || includeTime == DateTimeType::Time)
1852 {
1853 QCString stime;
1854 stime.sprintf("%.2d:%.2d:%.2d",hour,minutes,seconds);
1855 sdate+=stime;
1856 }
1857 return sdate;
