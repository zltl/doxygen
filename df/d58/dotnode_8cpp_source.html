99QCString escapeTooltip(const QCString &tooltip)
100{
101 if (tooltip.isEmpty()) return tooltip;
102 QCString result;
103 const char *p=tooltip.data();
104 char c;
105 while ((c=*p++))
106 {
107 switch(c)
108 {
109 case '"': result+="\\\""; break;
110 case '\\': result+="\\\\"; break;
111 default: result+=c; break;
112 }
113 }
114 return result;
115}

117static void writeBoxMemberList(TextStream &t,
118 char prot,const MemberList *ml,const ClassDef *scope,
119 bool &lineWritten,
120 bool isStatic=FALSE,const StringUnorderedSet *skipNames=nullptr)
121{
122 constexpr auto tr_start = "<TR><TD VALIGN=\"top\" CELLPADDING=\"1\" CELLSPACING=\"0\">";
123 constexpr auto tr_mid = "</TD><TD VALIGN=\"top\" ALIGN=\"LEFT\" CELLPADDING=\"1\" CELLSPACING=\"0\">";
124 constexpr auto tr_end = "</TD></TR>\n";
125 constexpr auto br = "<BR ALIGN=\"LEFT\"/>";
126 if (ml)
127 {
128 int totalCount=0;
129 for (const auto &mma : *ml)
130 {
131 if (mma->getClassDef()==scope &&
132 (skipNames==nullptr || skipNames->find(mma->name().str())==std::end(*skipNames)))
133 {
134 totalCount++;
135 }
136 }
137
138 int count=0;
139 auto dotUmlDetails = Config_getEnum(DOT_UML_DETAILS);
140 for (const auto &mma : *ml)
141 {
142 if (mma->getClassDef() == scope &&
143 (skipNames==nullptr || skipNames->find(mma->name().str())==std::end(*skipNames)))
144 {
145 int numFields = Config_getInt(UML_LIMIT_NUM_FIELDS);
146 if (numFields>0 && (totalCount>numFields*3/2 && count>=numFields))
147 {
148 t << tr_start << tr_mid << theTranslator->trAndMore(QCString().sprintf("%d",totalCount-count)) << tr_end;
149 lineWritten = true;
150 break;
151 }
152 else
153 {
154 t << tr_start << prot << tr_mid;
155 QCString label;
156 if (dotUmlDetails==DOT_UML_DETAILS_t::YES)
157 {
158 label+=mma->typeString();
159 label+=" ";
160 }
161 label+=mma->name();
162 if (!mma->isObjCMethod() && (mma->isFunction() || mma->isSlot() || mma->isSignal()))
163 {
164 if (dotUmlDetails==DOT_UML_DETAILS_t::YES)
165 {
166 label+=mma->argsString();
167 }
168 else
169 {
170 label+="()";
171 }
172 }
173 t << DotNode::convertLabel(label,true);
174 t << br << tr_end;
175 lineWritten = true;
176 count++;
177 }
178 }
179 }
180 // write member groups within the memberlist
181 for (const auto &mg : ml->getMemberGroupList())
182 {
183 if (!mg->members().empty())
184 {
185 writeBoxMemberList(t,prot,&mg->members(),scope,lineWritten,isStatic,skipNames);
186 }
187 }
188 }
189}

191QCString DotNode::convertLabel(const QCString &l, bool htmlLike)
192{
193 QCString bBefore("\\_/<({[: =-+@%#~?$"); // break before character set
194 QCString bAfter(">]),:;|"); // break after character set
195 QCString p(l);
196 if (p.isEmpty()) return QCString();
197 QCString result;
198 char c,pc=0;
199 uint idx = 0;
200 int len=p.length();
201 int charsLeft=len;
202 int sinceLast=0;
203 int foldLen = Config_getInt(DOT_WRAP_THRESHOLD); // ideal text length
204 QCString br;
205 if (htmlLike)
206 br = "<BR ALIGN=\"LEFT\"/>";
207 else
208 br = "\\l";
209 while (idx < p.length())
210 {
211 c = p[idx++];
212 char cs[2] = { c, 0 };
213 const char *replacement = cs;
214 if (htmlLike)
215 {
216 switch(c)
217 {
218 case '\\': replacement="\\\\"; break;
219 case '\n': replacement="\\n"; break;
220 case '<': replacement="&lt;"; break;
221 case '>': replacement="&gt;"; break;
222 case '"': replacement="&quot;"; break;
223 case '\'': replacement="&apos;"; break;
224 case '&': replacement="&amp;"; break;
225 }
226 }
227 else
228 {
229 switch(c)
230 {
231 case '\\': replacement="\\\\"; break;
232 case '\n': replacement="\\n"; break;
233 case '<': replacement="\\<"; break;
234 case '>': replacement="\\>"; break;
235 case '"': replacement="\\\""; break;
236 case '|': replacement="\\|"; break;
237 case '{': replacement="\\{"; break;
238 case '}': replacement="\\}"; break;
239 }
240 }
241 // Some heuristics to insert newlines to prevent too long
242 // boxes and at the same time prevent ugly breaks
243 if (c=='\n')
244 {
245 result+=replacement;
246 foldLen = (3*foldLen+sinceLast+2)/4;
247 sinceLast=1;
248 }
249 else if ((pc!=':' || c!=':') && charsLeft>foldLen/3 && sinceLast>foldLen && bBefore.contains(c))
250 {
251 result+=br;
252 result+=replacement;
253 foldLen = (foldLen+sinceLast+1)/2;
254 sinceLast=1;
255 }
256 else if (charsLeft>1+foldLen/4 && sinceLast>foldLen+foldLen/3 &&
257 !isupper(c) && isupper(p[idx]))
258 {
259 result+=replacement;
260 result+=br;
261 foldLen = (foldLen+sinceLast+1)/2;
262 sinceLast=0;
263 }
264 else if (charsLeft>foldLen/3 && sinceLast>foldLen && bAfter.contains(c) && (c!=':' || p[idx]!=':'))
265 {
266 result+=replacement;
267 result+=br;
268 foldLen = (foldLen+sinceLast+1)/2;
269 sinceLast=0;
270 }
271 else
272 {
273 result+=replacement;
274 sinceLast++;
275 }
276 charsLeft--;
277 pc=c;
278 }
279 if (htmlLike)
280 {
281 result = result.stripWhiteSpace();
282 }
283 return result;
284}

286static QCString stripProtectionPrefix(const QCString &s)
287{
288 if (!s.isEmpty() && (s[0]=='-' || s[0]=='+' || s[0]=='~' || s[0]=='#'))
289 {
290 return s.mid(1);
291 }
292 else
293 {
294 return s;
295 }

347void DotNode::deleteNode(DotNodeRefVector &deletedList)
348{
349 if (m_deleted) return; // avoid recursive loops in case the graph has cycles
350 m_deleted=TRUE;
351 // delete all parent nodes of this node
352 for (const auto &pn : m_parents)
353 {
354 pn->deleteNode(deletedList);
355 }
356 // delete all child nodes of this node
357 for (const auto &cn : m_children)
358 {
359 cn->deleteNode(deletedList);
360 }
361 // add this node to the list of deleted nodes.
362 deletedList.push_back(this);
363}

389void DotNode::writeLabel(TextStream &t, GraphType gt) const
390{
391 if (m_classDef && Config_getBool(UML_LOOK) && (gt==Inheritance || gt==Collaboration))
392 {
393 // Set shape to the plain type.
394 // the UML properties and methods are rendered using dot' HTML like table format
395 t << "shape=plain,label=";
396 // add names shown as relations to a set, so we don't show
397 // them as attributes as well
398 StringUnorderedSet arrowNames;
399 // for each edge
400 for (const auto &ei : m_edgeInfo)
401 {
402 if (!ei.label().isEmpty()) // labels joined by \n
403 {
404 int i;
405 int p=0;
406 QCString lab;
407 while ((i=ei.label().find('\n',p))!=-1)
408 {
409 lab = stripProtectionPrefix(ei.label().mid(p,i-p));
410 arrowNames.insert(lab.str());
411 p=i+1;
412 }
413 lab = stripProtectionPrefix(ei.label().right(ei.label().length()-p));
414 arrowNames.insert(lab.str());
415 }
416 }
417
418 constexpr auto hr_start = "<TR><TD COLSPAN=\"2\" CELLPADDING=\"1\" CELLSPACING=\"0\">";
419 constexpr auto hr_end = "</TD></TR>\n";
420 constexpr auto sep = "<HR/>\n";
421 constexpr auto empty_line = "<TR><TD COLSPAN=\"2\" CELLPADDING=\"1\" CELLSPACING=\"0\">&nbsp;</TD></TR>\n";
422 //printf("DotNode::writeBox for %s\n",qPrint(m_classDef->name()));
423 t << "<<TABLE CELLBORDER=\"0\" BORDER=\"1\">";
424 t << hr_start << convertLabel(m_label,true) << hr_end;
425 auto dotUmlDetails = Config_getEnum(DOT_UML_DETAILS);
426 if (dotUmlDetails!=DOT_UML_DETAILS_t::NONE)
427 {
428 bool lineWritten = false;
429 t << sep;
430 writeBoxMemberList(t,'+',m_classDef->getMemberList(MemberListType_pubAttribs),m_classDef,lineWritten,FALSE,&arrowNames);
431 writeBoxMemberList(t,'+',m_classDef->getMemberList(MemberListType_pubStaticAttribs),m_classDef,lineWritten,TRUE,&arrowNames);
432 writeBoxMemberList(t,'+',m_classDef->getMemberList(MemberListType_properties),m_classDef,lineWritten,FALSE,&arrowNames);
433 writeBoxMemberList(t,'~',m_classDef->getMemberList(MemberListType_pacAttribs),m_classDef,lineWritten,FALSE,&arrowNames);
434 writeBoxMemberList(t,'~',m_classDef->getMemberList(MemberListType_pacStaticAttribs),m_classDef,lineWritten,TRUE,&arrowNames);
435 writeBoxMemberList(t,'#',m_classDef->getMemberList(MemberListType_proAttribs),m_classDef,lineWritten,FALSE,&arrowNames);
436 writeBoxMemberList(t,'#',m_classDef->getMemberList(MemberListType_proStaticAttribs),m_classDef,lineWritten,TRUE,&arrowNames);
437 if (Config_getBool(EXTRACT_PRIVATE))
438 {
439 writeBoxMemberList(t,'-',m_classDef->getMemberList(MemberListType_priAttribs),m_classDef,lineWritten,FALSE,&arrowNames);
440 writeBoxMemberList(t,'-',m_classDef->getMemberList(MemberListType_priStaticAttribs),m_classDef,lineWritten,TRUE,&arrowNames);
441 }
442 if (!lineWritten) t << empty_line;
443 t << sep;
444 lineWritten = false;
445 writeBoxMemberList(t,'+',m_classDef->getMemberList(MemberListType_pubMethods),m_classDef,lineWritten);
446 writeBoxMemberList(t,'+',m_classDef->getMemberList(MemberListType_pubStaticMethods),m_classDef,lineWritten,TRUE);
447 writeBoxMemberList(t,'+',m_classDef->getMemberList(MemberListType_pubSlots),m_classDef,lineWritten);
448 writeBoxMemberList(t,'~',m_classDef->getMemberList(MemberListType_pacMethods),m_classDef,lineWritten);
449 writeBoxMemberList(t,'~',m_classDef->getMemberList(MemberListType_pacStaticMethods),m_classDef,lineWritten,TRUE);
450 writeBoxMemberList(t,'#',m_classDef->getMemberList(MemberListType_proMethods),m_classDef,lineWritten);
451 writeBoxMemberList(t,'#',m_classDef->getMemberList(MemberListType_proStaticMethods),m_classDef,lineWritten,TRUE);
452 writeBoxMemberList(t,'#',m_classDef->getMemberList(MemberListType_proSlots),m_classDef,lineWritten);
453 if (Config_getBool(EXTRACT_PRIVATE))
454 {
455 writeBoxMemberList(t,'-',m_classDef->getMemberList(MemberListType_priMethods),m_classDef,lineWritten);
456 writeBoxMemberList(t,'-',m_classDef->getMemberList(MemberListType_priStaticMethods),m_classDef,lineWritten,TRUE);
457 writeBoxMemberList(t,'-',m_classDef->getMemberList(MemberListType_priSlots),m_classDef,lineWritten);
458 }
459 if (m_classDef->getLanguage()!=SrcLangExt_Fortran)
460 {
461 for (const auto &mg : m_classDef->getMemberGroups())
462 {
463 if (!mg->members().empty())
464 {
465 writeBoxMemberList(t,'*',&mg->members(),m_classDef,lineWritten,FALSE,&arrowNames);
466 }
467 }
468 }
469 if (!lineWritten) t << empty_line;
470 }
471 t << "</TABLE>>\n";
472 }
473 else if (Config_getString(DOT_NODE_ATTR).contains("shape=plain"))
474 {
475 t << "label=";
476 if (m_isRoot)
477 t << "<<b>" << convertToXML(m_label) << "</b>>";
478 else if (m_truncated == Truncated)
479 t << "<<i>" << convertToXML(m_label) << "</i>>";
480 else
481 t << '"' << convertLabel(m_label) << '"';
482 }
483 else // standard look
484 {
485 t << "label=" << '"' << convertLabel(m_label) << '"';
486 }
487}

489void DotNode::writeUrl(TextStream &t) const
490{
491 if (m_url.isEmpty())
492 return;
493 int tagPos = m_url.findRev('$');
494 t << ",URL=\"";
495 QCString noTagURL = m_url;
496 if (tagPos!=-1)
497 {
498 t << m_url.left(tagPos);
499 noTagURL = m_url.mid(tagPos);
500 }
501 int anchorPos = noTagURL.findRev('#');
502 if (anchorPos==-1)
503 {
504 t << addHtmlExtensionIfMissing(noTagURL) << "\"";
505 }
506 else
507 {
508 t << addHtmlExtensionIfMissing(noTagURL.left(anchorPos))
509 << noTagURL.right(noTagURL.length() - anchorPos) << "\"";
510 }
511}

513void DotNode::writeBox(TextStream &t,
514 GraphType gt,
515 GraphOutputFormat /*format*/,
516 bool hasNonReachableChildren) const
517{
518 const char *labCol;
519 const char *fillCol = "white";
520 if (m_classDef)
521 {
522 if (m_classDef->hasDocumentation() && hasNonReachableChildren)
523 {
524 labCol = "red";
525 fillCol = "#FFF0F0";
526 }
527 else if (m_classDef->hasDocumentation() && !hasNonReachableChildren)
528 labCol = "gray40";
529 else if (!m_classDef->hasDocumentation() && hasNonReachableChildren)
530 labCol = "orangered";
531 else // (!m_classDef->hasDocumentation() && !hasNonReachableChildren)
532 {
533 labCol = "grey75";
534 if (m_classDef->templateMaster() && m_classDef->templateMaster()->hasDocumentation())
535 labCol = "gray40";
536 }
537 }
538 else
539 {
540 labCol = m_url.isEmpty() ? "grey60" : // non link
541 (hasNonReachableChildren ? "red" : "grey40");
542 fillCol = m_url.isEmpty() ? "#E0E0E0" :
543 (hasNonReachableChildren ? "#FFF0F0" : "white");
544 }
545 t << " Node" << m_number << " [";
546 writeLabel(t,gt);
547 t << ",height=0.2,width=0.4";
548 if (m_isRoot)
549 {
550 t << ",color=\"gray40\", fillcolor=\"grey60\", style=\"filled\", fontcolor=\"black\"";
551 }
552 else
553 {
554 t << ",color=\"" << labCol << "\"";
555 t << ", fillcolor=\"" << fillCol << "\"";
556 t << ", style=\"filled\"";
557 writeUrl(t);
558 }
559 if (!m_tooltip.isEmpty())
560 {
561 t << ",tooltip=\"" << escapeTooltip(m_tooltip) << "\"";
562 }
563 else
564 {
565 t << ",tooltip=\" \""; // space in tooltip is required otherwise still something like 'Node0' is used
566 }
567 t << "];\n";
568}

570void DotNode::writeArrow(TextStream &t,
571 GraphType gt,
572 GraphOutputFormat format,
573 const DotNode *cn,
574 const EdgeInfo *ei,
575 bool topDown,
576 bool pointBack) const
577{
578 t << " Node";
579 if (topDown)
580 t << cn->number();
581 else
582 t << m_number;
583 t << " -> Node";
584 if (topDown)
585 t << m_number;
586 else
587 t << cn->number();
588 t << " [";
589
590 const EdgeProperties *eProps = Config_getBool(UML_LOOK) ? &umlEdgeProps : &normalEdgeProps;
591 QCString aStyle = eProps->arrowStyleMap[ei->color()];
592 bool umlUseArrow = aStyle=="odiamond";
593
594 if (pointBack && !umlUseArrow) t << "dir=\"back\",";
595 t << "color=\"" << eProps->edgeColorMap[ei->color()] << "\",";
596 t << "style=\"" << eProps->edgeStyleMap[ei->style()] << "\"";
597 if (!ei->label().isEmpty())
598 {
599 t << ",label=\" " << convertLabel(ei->label()) << "\",fontcolor=\"grey\" ";
600 }
601 if (Config_getBool(UML_LOOK) &&
602 eProps->arrowStyleMap[ei->color()] &&
603 (gt==Inheritance || gt==Collaboration)
604 )
605 {
606 bool rev = pointBack;
607 if (umlUseArrow) rev=!rev; // UML use relates has arrow on the start side
608 if (rev)
609 t << ",arrowtail=\"" << eProps->arrowStyleMap[ei->color()] << "\"";
610 else
611 t << ",arrowhead=\"" << eProps->arrowStyleMap[ei->color()] << "\"";
612 }
613
614 t << "];\n";
615}

617void DotNode::write(TextStream &t,
618 GraphType gt,
619 GraphOutputFormat format,
620 bool topDown,
621 bool toChildren,
622 bool backArrows) const
623{
624 //printf("DotNode::write(%d) name=%s this=%p written=%d visible=%d\n",m_distance,qPrint(m_label),this,m_written,m_visible);
625 if (m_written) return; // node already written to the output
626 if (!m_visible) return; // node is not visible
627 writeBox(t,gt,format,m_truncated==Truncated);
628 m_written=TRUE;
629 if (toChildren)
630 {
631 auto it = m_edgeInfo.begin();
632 for (const auto &cn : m_children)
633 {
634 if (cn->isVisible())
635 {
636 //printf("write arrow %s%s%s\n",qPrint(label()),backArrows?"<-":"->",qPrint(cn->label()));
637 writeArrow(t,gt,format,cn,&(*it),topDown,backArrows);
638 }
639 cn->write(t,gt,format,topDown,toChildren,backArrows);
640 ++it;
641 }
642 }
643 else // render parents
644 {
645 for (const auto &pn : m_parents)
646 {
647 if (pn->isVisible())
648 {
649 const auto &children = pn->children();
650 auto child_it = std::find(children.begin(),children.end(),this);
651 size_t index = child_it - children.begin();
652 //printf("write arrow %s%s%s\n",qPrint(label()),backArrows?"<-":"->",qPrint(pn->label()));
653 writeArrow(t,
654 gt,
655 format,
656 pn,
657 &pn->edgeInfo()[index],
658 FALSE,
659 backArrows
660 );
661 }
662 pn->write(t,gt,format,TRUE,FALSE,backArrows);
663 }
664 }
665 //printf("end DotNode::write(%d) name=%s\n",distance,qPrint(m_label));
666}

668void DotNode::writeXML(TextStream &t,bool isClassGraph) const
669{
670 t << " <node id=\"" << m_number << "\">\n";
671 t << " <label>" << convertToXML(m_label) << "</label>\n";
672 if (!m_url.isEmpty())
673 {
674 QCString url(m_url);
675 int dollarPos = url.find('$');
676 if (dollarPos!=-1)
677 {
678 t << " <link refid=\"" << convertToXML(url.mid(dollarPos+1)) << "\"";
679 if (dollarPos>0)
680 {
681 t << " external=\"" << convertToXML(url.left(dollarPos)) << "\"";
682 }
683 t << "/>\n";
684 }
685 }
686 auto it = m_edgeInfo.begin();
687 for (const auto &childNode : m_children)
688 {
689 const EdgeInfo &edgeInfo = *it;
690 t << " <childnode refid=\"" << childNode->number() << "\" relation=\"";
691 if (isClassGraph)
692 {
693 switch(edgeInfo.color())
694 {
695 case EdgeInfo::Blue: t << "public-inheritance"; break;
696 case EdgeInfo::Green: t << "protected-inheritance"; break;
697 case EdgeInfo::Red: t << "private-inheritance"; break;
698 case EdgeInfo::Purple: t << "usage"; break;
699 case EdgeInfo::Orange: t << "template-instance"; break;
700 case EdgeInfo::Orange2: t << "type-constraint"; break;
701 case EdgeInfo::Grey: ASSERT(0); break;
702 }
703 }
704 else // include graph
705 {
706 t << "include";
707 }
708 t << "\">\n";
709 if (!edgeInfo.label().isEmpty())
710 {
711 int p=0;
712 int ni;
713 while ((ni=edgeInfo.label().find('\n',p))!=-1)
714 {
715 t << " <edgelabel>"
716 << convertToXML(edgeInfo.label().mid(p,ni-p))
717 << "</edgelabel>\n";
718 p=ni+1;
719 }

777 t << " <edgelabel>"
