76class Trace
77{
78 public:
79 Trace(const QCString &func) : m_func(func)
80 {
81 if (Debug::isFlagSet(Debug::Markdown))
82 {
83 fprintf(IOSTREAM,"> %s\n",qPrint(func));
84 s_indent++;
85 }
86 }
87 Trace(const QCString &func,const QCString &data) : m_func(func)
88 {
89 if (Debug::isFlagSet(Debug::Markdown))
90 {
91 indent();
92 char data_s[DATA_BUFSIZE*2+1] = ""; // worst case each input char outputs 2 chars + 0 terminator.
93 int j=0;
94 if (!data.isEmpty())
95 {
96 for (int i=0;i<DATA_BUFSIZE;i++)
97 {
98 char c=data[i];
99 if (c==0) break;
100 else if (c=='\n') { data_s[j++]='\\'; data_s[j++]='n'; }
101 else if (c=='\t') { data_s[j++]='\\'; data_s[j++]='t'; }
102 else if (c=='\r') { data_s[j++]='\\'; data_s[j++]='r'; }
103 else if (c=='\\') { data_s[j++]='\\'; data_s[j++]='\\'; }
104 else data_s[j++]=c;
105 }
106 }
107 data_s[j++]=0;
108 fprintf(IOSTREAM,"> %s data=[%s...]\n",qPrint(func),data_s);
109 s_indent++;
110 }
111 }
112 ~Trace()
113 {
114 if (Debug::isFlagSet(Debug::Markdown))
115 {
116 s_indent--;
117 indent();
118 if (m_resultSet)
119 {
120 fprintf(IOSTREAM,"< %s result=%s\n",qPrint(m_func),qPrint(m_resultValue));
121 }
122 else
123 {
124 fprintf(IOSTREAM,"< %s\n",qPrint(m_func));
125 }
126 }
127 }

129 {
130 if (Debug::isFlagSet(Debug::Markdown))
131 {
132 indent();
133 fprintf(IOSTREAM,": %s: ",qPrint(m_func));
134 va_list args;
135 va_start(args,fmt);
136 vfprintf(IOSTREAM, fmt, args);
137 va_end(args);
138 }
139 }

248// test if the next characters in data represent a new line (which can be character \n or string \ilinebr).
249// returns 0 if no newline is found, or the number of characters that make up the newline if found.
250inline int isNewline(const char *data)
251{
252 // normal newline
253 if (data[0] == '\n') return 1;
254 // artificial new line from ^^ in ALIASES
255 if (data[0] == '\\' && qstrncmp(data+1,"ilinebr ",7)==0) return data[8]==' ' ? 9 : 8;
256 return 0;
257}

259// escape double quotes in string
260static QCString escapeDoubleQuotes(const QCString &s)
261{
262 TRACE(s);
263 if (s.isEmpty()) return s;
264 GrowBuf growBuf;
265 const char *p=s.data();
266 char c,pc='\0';
267 while ((c=*p++))
268 {
269 switch (c)
270 {
271 case '"': if (pc!='\\') { growBuf.addChar('\\'); } growBuf.addChar(c); break;
272 default: growBuf.addChar(c); break;
273 }
274 pc=c;
275 }
276 growBuf.addChar(0);
277 TRACE_RESULT(growBuf.get());
278 return growBuf.get();
279}
280// escape characters that have a special meaning later on.
281static QCString escapeSpecialChars(const QCString &s)
282{
283 TRACE(s);
284 if (s.isEmpty()) return s;
285 bool insideQuote=FALSE;
286 GrowBuf growBuf;
287 const char *p=s.data();
288 char c,pc='\0';
289 while ((c=*p++))
290 {
291 switch (c)
292 {
293 case '"': if (pc!='\\') { insideQuote=!insideQuote; } growBuf.addChar(c); break;
294 case '<':
295 case '>': if (!insideQuote)
296 {
297 growBuf.addChar('\\');
298 growBuf.addChar(c);
299 if ((p[0] == ':') && (p[1] == ':'))
300 {
301 growBuf.addChar('\\');
302 growBuf.addChar(':');
303 p++;
304 }
305 }
306 else
307 {
308 growBuf.addChar(c);
309 }
310 break;
311 case '\\': if (!insideQuote) { growBuf.addChar('\\'); } growBuf.addChar('\\'); break;
312 case '@': if (!insideQuote) { growBuf.addChar('\\'); } growBuf.addChar('@'); break;
313 // commented out next line due to regression when using % to suppress a link
314 //case '%': if (!insideQuote) { growBuf.addChar('\\'); } growBuf.addChar('%'); break;
315 case '#': if (!insideQuote) { growBuf.addChar('\\'); } growBuf.addChar('#'); break;
316 case '$': if (!insideQuote) { growBuf.addChar('\\'); } growBuf.addChar('$'); break;
317 case '&': if (!insideQuote) { growBuf.addChar('\\'); } growBuf.addChar('&'); break;
318 default: growBuf.addChar(c); break;
319 }
320 pc=c;
321 }
322 growBuf.addChar(0);
323 TRACE_RESULT(growBuf.get());
324 return growBuf.get();
325}

360static QCString getFilteredImageAttributes(const char *fmt, const QCString &attrs)
361{
362 StringVector attrList = split(attrs.str(),",");
363 for (const auto &attr_ : attrList)
364 {
365 QCString attr = QCString(attr_).stripWhiteSpace();
366 int i = attr.find(':');

374 }

377 return attr;
378 }
379 }

383// Check if data contains a block command. If so returned the command
384// that ends the block. If not an empty string is returned.
385// Note When offset>0 character position -1 will be inspected.
386//
387// Checks for and skip the following block commands:
388// {@code .. { .. } .. }
389// \dot .. \enddot
390// \code .. \endcode
391// \msc .. \endmsc
392// \f$..\f$
393// \f(..\f)
394// \f[..\f]
395// \f{..\f}
396// \verbatim..\endverbatim
397// \iliteral..\endiliteral
398// \latexonly..\endlatexonly
399// \htmlonly..\endhtmlonly
400// \xmlonly..\endxmlonly
401// \rtfonly..\endrtfonly
402// \manonly..\endmanonly
403// \startuml..\enduml
404QCString Markdown::isBlockCommand(const char *data,int offset,int size)
405{
406 TRACE(data);
407
408 using EndBlockFunc = QCString (*)(const std::string &,bool,char);
409
410 static const auto getEndBlock = [](const std::string &blockName,bool,char) -> QCString
411 {
412 return "end"+blockName;
413 };
414 static const auto getEndCode = [](const std::string &blockName,bool openBracket,char) -> QCString
415 {
416 return openBracket ? QCString("}") : "end"+blockName;
417 };
418 static const auto getEndUml = [](const std::string &blockName,bool,char) -> QCString
419 {
420 return "enduml";
421 };
422 static const auto getEndFormula = [](const std::string &blockName,bool,char nextChar) -> QCString
423 {
424 switch (nextChar)
425 {
426 case '$': return "f$";
427 case '(': return "f)";
428 case '[': return "f]";
429 case '{': return "f}";
430 }
431 return "";
432 };
433
434 // table mapping a block start command to a function that can return the matching end block string
435 static const std::unordered_map<std::string,EndBlockFunc> blockNames =
436 {
437 { "dot", getEndBlock },
438 { "code", getEndCode },
439 { "icode", getEndBlock },
440 { "msc", getEndBlock },
441 { "verbatim", getEndBlock },
442 { "iverbatim", getEndBlock },
443 { "iliteral", getEndBlock },
444 { "latexonly", getEndBlock },
445 { "htmlonly", getEndBlock },
446 { "xmlonly", getEndBlock },
447 { "rtfonly", getEndBlock },
448 { "manonly", getEndBlock },
449 { "docbookonly", getEndBlock },
450 { "startuml", getEndUml },
451 { "f", getEndFormula }
452 };
453
454 bool openBracket = offset>0 && data[-1]=='{';
455 bool isEscaped = offset>0 && (data[-1]=='\\' || data[-1]=='@');
456 if (isEscaped) return QCString();
457
458 int end=1;
459 while (end<size && (data[end]>='a' && data[end]<='z')) end++;
460 if (end==1) return QCString();
461 std::string blockName(data+1,end-1);
462 auto it = blockNames.find(blockName);
463 QCString result;
464 if (it!=blockNames.end()) // there is a function assigned
465 {
466 result = it->second(blockName, openBracket, end<size ? data[end] : 0);
467 }
468 TRACE_RESULT(result);
469 return result;
470}

478 static const auto endOfLine = [](const char *data_,int offset_,int size_) -> int
479 {
480 // skip until the end of line (allowing line continuation characters)
481 char lc = 0;
482 char c;
483 while (offset_<size_ && ((c=data_[offset_])!='\n' || lc=='\\'))
484 {
485 if (c=='\\') lc='\\'; // last character was a line continuation
486 else if (c!=' ') lc=0; // rest line continuation
487 offset_++;
488 }
489 return offset_;
490 };
491
492 static const auto endOfLabel = [](const char *data_,int offset_,int size_) -> int
493 {
494 if (offset_<size_ && data_[offset_]==' ') // we expect a space before the label
495 {
496 char c;
497 offset_++;
498 // skip over spaces
499 while (offset_<size_ && data_[offset_]==' ') offset_++;
500 // skip over label
501 while (offset_<size_ && (c=data_[offset_])!=' ' && c!='\\' && c!='@' && c!='\n') offset_++;
502 return offset_;
503 }
504 return 0;
505 };
506
507 static const auto endOfParam = [](const char *data_,int offset_,int size_) -> int
508 {
509 int index=offset_;
510 if (index<size_ && data_[index]==' ') // skip over optional spaces
511 {
512 index++;
513 while (index<size_ && data_[index]==' ') index++;
514 }
515 if (index<size_ && data_[index]=='[') // find matching ']'
516 {
517 index++;
518 char c;
519 while (index<size_ && (c=data_[index])!=']' && c!='\n') index++;
520 if (index==size_ || data_[index]!=']') return 0; // invalid parameter
521 offset_=index+1; // part after [...] is the parameter name
522 }
523 return endOfLabel(data_,offset_,size_);
524 };
525
526 static const auto endOfFuncLike = [](const char *data_,int offset_,int size_,bool allowSpaces) -> int
527 {
528 if (offset_<size_ && data_[offset_]==' ') // we expect a space before the name
529 {
530 char c=0;

532 // skip over spaces
533 while (offset_<size_ && data_[offset_]==' ')

535 offset_++;
536 }
537 // skip over name (and optionally type)
538 while (offset_<size_ && (c=data_[offset_])!='\n' && (allowSpaces || c!=' ') && c!='(')
539 {
540 offset_++;
541 }
542 if (c=='(') // find the end of the function
543 {
544 int count=1;
545 offset_++;
546 while (offset_<size_ && (c=data_[offset_++]))
547 {
548 if (c=='(') count++;
549 else if (c==')') count--;
550 if (count==0) return offset_;
551 }
552 }
553 return offset_;
554 }
555 return 0;
556 };

560 return endOfFuncLike(data_,offset_,size_,true);

677/** looks for the next emph char, skipping other constructs, and
678 * stopping when either it is found, or we are at the end of a paragraph.
679 */
680int Markdown::findEmphasisChar(const char *data, int size, char c, int c_size)
681{
682 TRACE(data);
683 int i = 1;
684
685 while (i<size)
686 {
687 while (i<size && data[i]!=c && data[i]!='`' &&
688 data[i]!='\\' && data[i]!='@' &&
689 !(data[i]=='/' && data[i-1]=='<') && // html end tag also ends emphasis
690 data[i]!='\n') i++;
691 //printf("findEmphasisChar: data=[%s] i=%d c=%c\n",data,i,data[i]);
692
693 // not counting escaped chars or characters that are unlikely
694 // to appear as the end of the emphasis char
695 if (ignoreCloseEmphChar(i-1))
696 {
697 i++;
698 continue;
699 }
700 else
701 {
702 // get length of emphasis token
703 int len = 0;
704 while (i+len<size && data[i+len]==c)
705 {
706 len++;
707 }
708
709 if (len>0)
710 {
711 if (len!=c_size || (i<size-len && isIdChar(i+len))) // to prevent touching some_underscore_identifier
712 {
713 i=i+len;
714 continue;
715 }
716 TRACE_RESULT(i);
717 return i; // found it
718 }
719 }
720
721 // skipping a code span
722 if (data[i]=='`')
723 {
724 int snb=0;
725 while (i<size && data[i]=='`') snb++,i++;
726
727 // find same pattern to end the span
728 int enb=0;
729 while (i<size && enb<snb)
730 {
731 if (data[i]=='`') enb++;
732 if (snb==1 && data[i]=='\'') break; // ` ended by '
733 i++;
734 }
735 }
736 else if (data[i]=='@' || data[i]=='\\')
737 { // skip over blocks that should not be processed
738 QCString endBlockName = isBlockCommand(data+i,i,size-i);
739 if (!endBlockName.isEmpty())
740 {
741 i++;
742 int l = endBlockName.length();
743 while (i<size-l)
744 {
745 if ((data[i]=='\\' || data[i]=='@') && // command
746 data[i-1]!='\\' && data[i-1]!='@') // not escaped
747 {
748 if (qstrncmp(&data[i+1],endBlockName.data(),l)==0)
749 {
750 break;
751 }
752 }
753 i++;
754 }
755 }
756 else if (i<size-1 && isIdChar(i+1)) // @cmd, stop processing, see bug 690385
757 {
758 TRACE_RESULT(0);
759 return 0;
760 }
761 else
762 {
763 i++;
764 }
765 }
766 else if (data[i-1]=='<' && data[i]=='/') // html end tag invalidates emphasis
767 {
768 TRACE_RESULT(0);
769 return 0;
770 }
771 else if (data[i]=='\n') // end * or _ at paragraph boundary
772 {
773 i++;
774 while (i<size && data[i]==' ') i++;
775 if (i>=size || data[i]=='\n') { TRACE_RESULT(0); return 0; } // empty line -> paragraph
776 }
777 else // should not get here!
778 {
779 i++;
780 }
781
782 }
783 TRACE_RESULT(0);
784 return 0;

983/** Process a HTML tag. Note that <pre>..</pre> are treated specially, in
984 * the sense that all code inside is written unprocessed
985 */
986int Markdown::processHtmlTagWrite(const char *data,int offset,int size,bool doWrite)
987{

989 if (offset>0 && data[-1]=='\\') { TRACE_RESULT(0); return 0; } // escaped <

997 convertStringFragment(tagName,data+1,i-1);
998 if (tagName.lower()=="pre") // found <pre> tag
999 {
1000 bool insideStr=FALSE;
1001 while (i<size-6)
1002 {
1003 char c=data[i];
1004 if (!insideStr && c=='<') // potential start of html tag
1005 {
1006 if (data[i+1]=='/' &&
1007 tolower(data[i+2])=='p' && tolower(data[i+3])=='r' &&
1008 tolower(data[i+4])=='e' && tolower(data[i+5])=='>')

1015 }

1019 }
1020 else if (c=='"')

1022 insideStr=TRUE;
1023 }
1024 i++;
1025 }
1026 }
1027 else // some other html tag
1028 {
1029 if (l>0 && i<size)
1030 {
1031 if (data[i]=='/' && i<size-1 && data[i+1]=='>') // <bla/>

1033 //printf("Found htmlTag={%s}\n",qPrint(QCString(data).left(i+2)));

1067 }
1068 }
1069 }
1070 }
1071 //printf("Not a valid html tag\n");
1072 TRACE_RESULT(0);
1073 return 0;

1869 i++;
