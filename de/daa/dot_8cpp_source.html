42static void setDotFontPath(const QCString &path)
43{
44 ASSERT(g_dotFontPath.isEmpty());
45 g_dotFontPath = Portable::getenv("DOTFONTPATH");
46 QCString newFontPath = Config_getString(DOT_FONTPATH);
47 if (!newFontPath.isEmpty() && !path.isEmpty())
48 {
49 newFontPath.prepend(path+Portable::pathListSeparator());
50 }
51 else if (newFontPath.isEmpty() && !path.isEmpty())
52 {
53 newFontPath=path;
54 }
55 else
56 {
57 Portable::unsetenv("DOTFONTPATH");
58 return;
59 }
60 Portable::setenv("DOTFONTPATH",newFontPath);
61}

69 else

72 }

84DotManager::DotManager() : m_runners(), m_filePatchers()
85{
86 m_queue = new DotRunnerQueue;
87 int i;
88 int dotNumThreads = Config_getInt(DOT_NUM_THREADS);
89 if (dotNumThreads!=1)
90 {
91 for (i=0;i<dotNumThreads;i++)
92 {
93 DotWorkerThreadPtr thread(new DotWorkerThread(m_queue));
94 thread->start();
95 if (thread->isRunning())
96 {
97 m_workers.push_back(std::move(thread));
98 }
99 else // no more threads available!
100 {
101 }
102 }
103 ASSERT(m_workers.size()>0);
104 }
105}

112DotRunner* DotManager::createRunner(const QCString &absDotName, const QCString& md5Hash)
113{
114 std::lock_guard<std::mutex> lock(g_dotManagerMutex);
115 DotRunner* rv = nullptr;
116 auto const runit = m_runners.find(absDotName.str());
117 if (runit == m_runners.end())
118 {
119 auto insobj = std::make_unique<DotRunner>(absDotName, md5Hash);
120 rv = insobj.get();
121 m_runners.emplace(absDotName.str(), std::move(insobj));
122 }
123 else
124 {
125 // we have a match
126 if (md5Hash != runit->second->getMd5Hash())
127 {
128 err("md5 hash does not match for two different runs of %s !\n", qPrint(absDotName));
129 }
130 rv = runit->second.get();
131 }
132 assert(rv);
133 return rv;
134}

143 auto rv = m_filePatchers.emplace(std::make_pair(fileName.str(), fileName));
144 assert(rv.second);

148bool DotManager::run() const
149{
150 size_t numDotRuns = m_runners.size();
151 size_t numFilePatchers = m_filePatchers.size();
152 if (numDotRuns+numFilePatchers>1)
153 {
154 if (m_workers.size()==0)
155 {
156 msg("Generating dot graphs in single threaded mode...\n");
157 }
158 else
159 {
160 msg("Generating dot graphs using %zu parallel threads...\n",std::min(numDotRuns+numFilePatchers,m_workers.size()));
161 }
162 }
163 size_t i=1;
164
165 bool setPath=FALSE;
166 if (Config_getBool(GENERATE_HTML))
167 {
168 setDotFontPath(Config_getString(HTML_OUTPUT));
169 setPath=TRUE;
170 }
171 else if (Config_getBool(GENERATE_LATEX))
172 {
173 setDotFontPath(Config_getString(LATEX_OUTPUT));
174 setPath=TRUE;
175 }
176 else if (Config_getBool(GENERATE_RTF))
177 {
178 setDotFontPath(Config_getString(RTF_OUTPUT));
179 setPath=TRUE;
180 }
181 else if (Config_getBool(GENERATE_DOCBOOK))
182 {
183 setDotFontPath(Config_getString(DOCBOOK_OUTPUT));
184 setPath=TRUE;
185 }
186 // fill work queue with dot operations
187 size_t prev=1;
188 if (m_workers.size()==0) // no threads to work with
189 {
190 for (auto & dr : m_runners)
191 {
192 msg("Running dot for graph %zu/%zu\n",prev,numDotRuns);
193 dr.second->run();
194 prev++;
195 }
196 }
197 else // use multiple threads to run instances of dot in parallel
198 {
199 for (auto & dr: m_runners)
200 {
201 m_queue->enqueue(dr.second.get());
202 }
203 // wait for the queue to become empty
204 while ((i=m_queue->size())>0)
205 {
206 i = numDotRuns - i;
207 while (i>=prev)
208 {
209 msg("Running dot for graph %zu/%zu\n",prev,numDotRuns);
210 prev++;
211 }
212 Portable::sleep(100);
213 }
214 while (numDotRuns>=prev)
215 {
216 msg("Running dot for graph %zu/%zu\n",prev,numDotRuns);
217 prev++;
218 }
219 // signal the workers we are done
220 for (i=0;i<m_workers.size();i++)
221 {
222 m_queue->enqueue(0); // add terminator for each worker
223 }
224 // wait for the workers to finish
225 for (i=0;i<m_workers.size();i++)
226 {
227 m_workers.at(i)->wait();
228 }
229 }
230 if (setPath)
231 {
232 unsetDotFontPath();
233 }
234
235 // patch the output file and insert the maps and figures
236 i=1;
237 // since patching the svg files may involve patching the header of the SVG
238 // (for zoomable SVGs), and patching the .html files requires reading that
239 // header after the SVG is patched, we first process the .svg files and
240 // then the other files.
241 for (auto & fp : m_filePatchers)
242 {
243 if (fp.second.isSVGFile())
244 {
245 msg("Patching output file %zu/%zu\n",i,numFilePatchers);
246 if (!fp.second.run()) return FALSE;
247 i++;
248 }
249 }
250 for (auto& fp : m_filePatchers)
251 {
252 if (!fp.second.isSVGFile())
253 {
254 msg("Patching output file %zu/%zu\n",i,numFilePatchers);
255 if (!fp.second.run()) return FALSE;
256 i++;
257 }
258 }
259 return TRUE;
260}

264void writeDotGraphFromFile(const QCString &inFile,const QCString &outDir,
265 const QCString &outFile,GraphOutputFormat format,
266 const QCString &srcFile,int srcLine)
267{
268 Dir d(outDir.str());
269 if (!d.exists())
270 {
271 term("Output dir %s does not exist!\n",qPrint(outDir));
272 }
273
274 QCString imgExt = getDotImageExtension();
275 QCString imgName = QCString(outFile)+"."+imgExt;
276 QCString absImgName = QCString(d.absPath())+"/"+imgName;
277 QCString absOutFile = QCString(d.absPath())+"/"+outFile;
278
279 DotRunner dotRun(inFile);
280 if (format==GOF_BITMAP)
281 {
282 dotRun.addJob(Config_getEnumAsString(DOT_IMAGE_FORMAT),absImgName,srcFile,srcLine);
283 }
284 else // format==GOF_EPS
285 {
286 if (Config_getBool(USE_PDFLATEX))
287 {
288 dotRun.addJob("pdf",absOutFile+".pdf",srcFile,srcLine);
289 }
290 else
291 {
292 dotRun.addJob("ps",absOutFile+".eps",srcFile,srcLine);
293 }
294 }
295
296 dotRun.preventCleanUp();
297 if (!dotRun.run())
298 {
299 return;
300 }
301
302 Doxygen::indexList->addImageFile(imgName);
303
304}

306/*! Writes user defined image map to the output.
307 * \param t text stream to write to
308 * \param inFile just the basename part of the filename
309 * \param outDir output directory
310 * \param relPath relative path the to root of the output dir
311 * \param baseName the base name of the output files
312 * \param context the scope in which this graph is found (for resolving links)
313 * \param graphId a unique id for this graph, use for dynamic sections
314 * \param srcFile the source file
315 * \param srcLine the line number in the source file
316 */
317void writeDotImageMapFromFile(TextStream &t,
318 const QCString &inFile, const QCString &outDir,
319 const QCString &relPath, const QCString &baseName,
320 const QCString &context,int graphId,
321 const QCString &srcFile,int srcLine)
322{
323
324 Dir d(outDir.str());
325 if (!d.exists())
326 {
327 term("Output dir %s does not exist!\n",qPrint(outDir));
328 }
329
330 QCString mapName = baseName+".map";
331 QCString imgExt = getDotImageExtension();
332 QCString imgName = baseName+"."+imgExt;
333 QCString absOutFile = QCString(d.absPath())+"/"+mapName;
334
335 DotRunner dotRun(inFile);
336 dotRun.addJob(MAP_CMD,absOutFile,srcFile,srcLine);
337 dotRun.preventCleanUp();
338 if (!dotRun.run())
339 {
340 return;
341 }
342
343 if (imgExt=="svg") // vector graphics
344 {
345 QCString svgName = outDir+"/"+baseName+".svg";
346 DotFilePatcher::writeSVGFigureLink(t,relPath,baseName,svgName);
347 DotFilePatcher patcher(svgName);
348 patcher.addSVGConversion("",TRUE,context,TRUE,graphId);
349 patcher.run();
350 }
351 else // bitmap graphics
352 {
353 TextStream tt;
354 t << "<img src=\"" << relPath << imgName << "\" alt=\""
355 << imgName << "\" border=\"0\" usemap=\"#" << mapName << "\"/>\n";
356 DotFilePatcher::convertMapFile(tt, absOutFile, relPath ,TRUE, context);
357 if (!tt.empty())
358 {
359 t << "<map name=\"" << mapName << "\" id=\"" << mapName << "\">";
360 t << tt.str();
361 t << "</map>\n";
362 }
363 }
364 d.remove(absOutFile.str());
365}
