75/** Abstract node interface with type information. */
76class DocNode
77{
78 public:
79 /*! Creates a new node */
80 DocNode(DocParser *parser,DocNodeVariant *parent) : m_parser(parser), m_parent(parent) {}
81
82 // allow nodes to be moved but not copied
83 DocNode(const DocNode &) = delete;
84 DocNode &operator=(const DocNode &) = delete;
85 DocNode(DocNode &&) = default;
86 DocNode &operator=(DocNode &&) = default;
87 ~DocNode() = default;
88
89 /*! Returns the parent of this node or 0 for the root node. */
90 DocNodeVariant *parent() { return m_parent; }
91 const DocNodeVariant *parent() const { return m_parent; }
92
93 DocParser *parser() { return m_parser; }
94 const DocParser *parser() const { return m_parser; }
95
96 /*! Sets a new parent for this node. */
97 void setParent(DocNodeVariant *parent) { m_parent = parent; }
98
99 /*! Returns TRUE iff this node is inside a preformatted section */
100 bool isPreformatted() const { return m_insidePre; }
101
102 protected:
103 /*! Sets whether or not this item is inside a preformatted section */
104 void setInsidePreformatted(bool p) { m_insidePre = p; }
105 enum RefType { Unknown, Anchor, Section, Table };
106 private:
107 bool m_insidePre = false;
108 DocParser *m_parser;
109 DocNodeVariant *m_parent;
110};

119

125 * holds a T, otherwise nullptr is returned.
