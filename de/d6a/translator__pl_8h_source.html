627 /*! this function is used to produce a comma-separated list of items.
628 * use generateMarker(i) to indicate where item i should be put.
629 */
630 QCString trWriteList(int numEntries)
631 {
632 QCString result;
633 int i;
634 // the inherits list contain `numEntries' classes
635 for (i=0;i<numEntries;i++)
636 {
637 // use generateMarker to generate placeholders for the class links!
638 result+=generateMarker(i); // generate marker for entry i in the list
639 // (order is left to right)
640
641 if (i!=numEntries-1) // not the last entry, so we need a separator
642 {
643 if (i<numEntries-2) // not the fore last entry
644 result+=", ";
645 else // the fore last entry
646 result+=" i ";
647 }
648 }
649 return result;
650 }
