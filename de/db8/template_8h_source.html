36/** @defgroup template_api Template API
37 *
38 * This is the API for a
39 * <a href="https://www.djangoproject.com/">Django</a>
40 * compatible template system written in C++.
41 * It is somewhat inspired by Stephen Kelly's
42 * <a href="http://www.gitorious.org/grantlee/pages/Home">Grantlee</a>.
43 *
44 * A template is simply a text file.
45 * A template contains \b variables, which get replaced with values when the
46 * template is evaluated, and \b tags, which control the logic of the template.
47 *
48 * Variables look like this: `{{ variable }}`
49 * When the template engine encounters a variable, it evaluates that variable and
50 * replaces it with the result. Variable names consist of any combination of
51 * alphanumeric characters and the underscore ("_").
52 * Use a dot (.) to access attributes of a structured variable.
53 *
54 * One can modify variables for display by using \b filters, for example:
55 * `{{ value|default:"nothing" }}`
56 *
57 * Tags look like this: `{% tag %}`. Tags are more complex than variables:
58 * Some create text in the output, some control flow by performing loops or logic,
59 * and some load external information into the template to be used by later variables.
60 *
61 * To comment-out part of a line in a template, use the comment syntax:
62 * `{# comment text #}`.
63 *
64 * Supported Django tags:
65 * - `for ... empty ... endfor`
66 * - `if ... else ... endif`
67 * - `block ... endblock`
68 * - `extend`
69 * - `include`
70 * - `with ... endwith`
71 * - `spaceless ... endspaceless`
72 * - `cycle`
73 *
74 * Extension tags:
75 * - `create` which instantiates a template and writes the result to a file.
76 * The syntax is `{% create 'filename' from 'template' %}`.
77 * - `recursetree`
78 * - `markers`
79 * - `msg` ... `endmsg`
80 * - `set`
81 *
82 * Supported Django filters:
83 * - `default`
84 * - `length`
85 * - `add`
86 * - `divisibleby`
87 *
88 * Extension filters:
89 * - `stripPath`
90 * - `nowrap`
91 * - `prepend`
92 * - `append`
93 *
94 * @{
95 */

97/** @brief Variant type which can hold one value of a fixed set of types. */
98class TemplateVariant
99{
100 public:
101 /** Type representing a function call in a template */
102 using FunctionDelegate = std::function<TemplateVariant(const std::vector<TemplateVariant>&)>;
103
104 /** Symbolic names for the possible types that this variant can hold. */
105 using VariantT = std::variant<std::monostate, // index==0, Invalid/default type
106 bool, // index==1: Type::Bool
107 int, // index==2: Type::Int
108 QCString, // index==3: Type::String
109 TemplateStructIntfPtr, // index==4: Type::Struct
110 TemplateListIntfPtr, // index==5: Type::List
111 FunctionDelegate, // index==6: Type::Function
112 TemplateStructIntfWeakPtr // index==7: Type::WeakStruct
113 >;
114
115 enum class Type : uint8_t
116 {
117 None = 0,
118 Bool = 1,
119 Int = 2,
120 String = 3,
121 Struct = 4,
122 List = 5,
123 Function = 6,
124 WeakStruct = 7
125 };
126
127 /** Constructs an invalid variant. */
128 TemplateVariant() {}
129
130 /** Constructs a new variant with a boolean value \a b. */
131 explicit TemplateVariant(bool b) { m_variant = b; }
132
133 /** Constructs a new variant with a integer value \a v. */
134 TemplateVariant(int v) { m_variant = v; }
135
136 /** Constructs a new variant with a integer value \a v. */
137 TemplateVariant(unsigned int v) { m_variant = static_cast<int>(v); }
138
139 /** Constructs a new variant with a integer value \a v.
140 * We use SFINAE to avoid a compiler error in case size_t already matches the 'unsigned int' overload.
141 */
142 template<typename T,
143 typename std::enable_if<std::is_same<T,size_t>::value,T>::type* = nullptr
144 >
145 TemplateVariant(T v) { m_variant = static_cast<int>(v); }
146
147 /** Constructs a new variant with a string value \a s. */
148 TemplateVariant(const char *s,bool raw=FALSE) : m_raw(raw) { m_variant = QCString(s); }
149
150 /** Constructs a new variant with a string value \a s. */
151 TemplateVariant(const QCString &s,bool raw=FALSE) : m_raw(raw) { m_variant = s; }
152
153 /** Constructs a new variant with a string value \a s. */
154 TemplateVariant(const std::string &s,bool raw=FALSE) : m_raw(raw) { m_variant = QCString(s); }
155
156 /** Constructs a new variant with a struct value \a s.
157 * @note. The variant will hold a counting reference to the object.
158 */
159 TemplateVariant(TemplateStructIntfPtr s) { m_variant = s; }
160
161 /** Constructs a new variant with a list value \a l.
162 * @note. The variant will hold a counting reference to the object.
163 */
164 TemplateVariant(TemplateListIntfPtr l) { m_variant = l; }
165
166 /** Constructs a new variant with a struct value \a s.
167 * @note. The variant will hold a non-counting reference to the object.
168 */
169 TemplateVariant(TemplateStructIntfWeakPtr s) { m_variant = s; }
170
171 /** Constructs a new variant which represents a method call
172 * @param[in] delegate FunctionDelegate object to invoke when
173 * calling call() on this variant.
174 * @note Use TemplateVariant::FunctionDelegate::fromMethod() and
175 * TemplateVariant::FunctionDelegate::fromFunction() to create
176 * FunctionDelegate objects.
177 */
178 TemplateVariant(FunctionDelegate delegate) { m_variant = delegate; }
179
180 /** Destroys the Variant object */
181 ~TemplateVariant() = default;
182
183 /** Constructs a copy of the variant, \a v,
184 * passed as the argument to this constructor.
185 */
186 TemplateVariant(const TemplateVariant &v) = default;
187
188 /** Moves the contents of variant \a v into this variant.
189 * variant \a v will become invalid
190 */
191 TemplateVariant(TemplateVariant &&v);
192
193 /** Assigns the value of the variant \a v to this variant. */
194 TemplateVariant &operator=(const TemplateVariant &v) = default;
195
196 /** Move the value of the variant \a v into this variant.
197 * Variant \a v will become invalid */
198 TemplateVariant &operator=(TemplateVariant &&v);
199
200 /** Compares this QVariant with v and returns true if they are equal;
201 * otherwise returns false.
202 */
203 bool operator==(TemplateVariant &other) const;
204
205 /** Returns the variant as a string. */
206 QCString toString() const;
207
208 /** Returns the variant as a boolean. */
209 bool toBool() const;
210
211 /** Returns the variant as an integer. */
212 int toInt() const;
213
214 /** Returns TRUE if the variant holds a valid value, or FALSE otherwise */
215 constexpr bool isValid() const { return std::holds_alternative<std::monostate>(m_variant); }
216 /** Returns TRUE if the variant holds a boolean value */
217 constexpr bool isBool() const { return std::holds_alternative<bool>(m_variant); }
218 /** Returns TRUE if the variant holds an integer value */
219 constexpr bool isInt() const { return std::holds_alternative<int>(m_variant); }
220 /** Returns TRUE if the variant holds a string value */
221 constexpr bool isString() const { return std::holds_alternative<QCString>(m_variant); }
222 /** Returns TRUE if the variant holds a struct value */
223 constexpr bool isStruct() const { return std::holds_alternative<TemplateStructIntfPtr>(m_variant); }
224 /** Returns TRUE if the variant holds a list value */
225 constexpr bool isList() const { return std::holds_alternative<TemplateListIntfPtr>(m_variant); }
226 /** Returns TRUE if the variant holds a function value */
227 constexpr bool isFunction() const { return std::holds_alternative<FunctionDelegate>(m_variant); }
228 /** Returns TRUE if the variant holds a struct value */
229 constexpr bool isWeakStruct() const { return std::holds_alternative<TemplateStructIntfWeakPtr>(m_variant); }
230
231 /** Returns the pointer to list referenced by this variant
232 * or 0 if this variant does not have list type.
233 */
234 TemplateListIntfPtr toList();
235 const TemplateListIntfPtr toList() const;
236
237 /** Returns the pointer to struct referenced by this variant
238 * or 0 if this variant does not have struct type.
239 */
240 TemplateStructIntfPtr toStruct();
241 const TemplateStructIntfPtr toStruct() const;
242
243 /** Return the result of apply this function with \a args.
244 * Returns an empty string if the variant type is not a function.
245 */
246 TemplateVariant call(const std::vector<TemplateVariant> &args = std::vector<TemplateVariant>());
247
248 /** Sets whether or not the value of the Variant should be
249 * escaped or written as-is (raw).
250 * @param[in] b TRUE means write as-is, FALSE means apply escaping.
251 */
252 void setRaw(bool b) { m_raw = b; }
253
254 /** Returns whether or not the value of the Value is raw.
255 * @see setRaw()
256 */
257 constexpr bool raw() const { return m_raw; }
258
259 /** Returns the type held by this variant */
260 constexpr Type type() const { return static_cast<Type>(m_variant.index()); }
261
262 /** Returns a string representation of this variant's type */
263 const char *typeAsString() const;
264
265 private:
266 QCString listToString() const;
267 QCString structToString() const;
268
269 VariantT m_variant;
270 bool m_raw = false;
271};

425/** @brief Abstract interface for a template context.
426 *
427 * A Context consists of a stack of dictionaries.
428 * A dictionary consists of a mapping of string keys onto TemplateVariant values.
429 * A key is searched starting with the dictionary at the top of the stack
430 * and searching downwards until it is found. The stack is used to create
431 * local scopes.
432 * @note This object must be created by TemplateEngine::createContext()
433 */
434class TemplateContext
435{
436 public:
437 virtual ~TemplateContext() = default;
438
439 /** Push a new scope on the stack. */
440 virtual void push() = 0;
441
442 /** Pop the current scope from the stack. */
443 virtual void pop() = 0;
444
445 /** Sets a value in the current scope.
446 * @param[in] name The name of the value; the key in the dictionary.
447 * @param[in] v The value associated with the key.
448 * @note When a given key is already present,
449 * its value will be replaced by \a v
450 */
451 virtual void set(const QCString &name,const TemplateVariant &v) = 0;
452
453 /** Gets the value for a given key
454 * @param[in] name The name of key.
455 * @returns The value, which can be an invalid variant in case the
456 * key was not found.
457 */
458 virtual TemplateVariant get(const QCString &name) const = 0;
459
460 /** Returns a pointer to the value corresponding to a given key.
461 * @param[in] name The name of key.
462 * @returns A pointer to the value, or 0 in case the key was not found.
463 */
464 virtual const TemplateVariant *getRef(const QCString &name) const = 0;
465
466 /** When files are created (i.e. by {% create ... %}) they written
467 * to the directory \a dir.
468 */
469 virtual void setOutputDirectory(const QCString &dir) = 0;
470
471 /** Sets the interface that will be used for escaping the result
472 * of variable expansion before writing it to the output.
473 */
474 virtual void setEscapeIntf(const QCString &extension, std::unique_ptr<TemplateEscapeIntf> intf) = 0;
475
476 /** Sets the interface that will be used inside a spaceless block
477 * to remove any redundant whitespace.
478 */
479 virtual void setSpacelessIntf(std::unique_ptr<TemplateSpacelessIntf> intf) = 0;
480};
